--@author Ally for Hire / Merydian9
--@model models/jaanus/wiretool/wiretool_range.mdl
--@shared
--@include allylib/simple_server-side_wire_helper.txt
--@include allylib/general_algorithms.txt

-- Version 1.5.1
-- Uses built-in SSSWH networking for driver-only state sync.

INPUTS = INPUTS or {}
SHARED = SHARED or {}
SETTINGS = SETTINGS or {}

INPUTS.Chair = chip()
INPUTS.MainParent = chip()
INPUTS.SecondaryParent = chip()
INPUTS.TertiaryParent = chip()
INPUTS.Filter = {}

SETTINGS.CamWorldPos1 = Vector(0, 0, 120)
SETTINGS.CamRelPos1 = Vector(0, 0, 0)
SETTINGS.CamDist1 = 300

SETTINGS.CamWorldPos2 = Vector(0, 0, 160)
SETTINGS.CamRelPos2 = Vector(0, 0, 0)
SETTINGS.CamDist2 = 0

SETTINGS.CamWorldPos3 = Vector(0, 0, 0)
SETTINGS.CamRelPos3 = Vector(250, 0, 0)
SETTINGS.CamDist3 = 0

DRIVER_CHECK_INTERVAL = 0.5

SHARED.Driver = nil
SHARED.Owner = owner()
SHARED.Player = player()
SHARED.EntId = chip():entIndex()

MSG_MAIN_OUTPUTS = 1
MSG_STATE = 2
MSG_FOV = 3
CLEAR_DRIVER_NET = SHARED.EntId .. " Driver Clear"
ZERO_VECTOR = Vector()
ZERO_ANGLE = Angle()

require("allylib/general_algorithms.txt")
require("allylib/simple_server-side_wire_helper.txt")

if SERVER then
    syncInputCommands(true)
    updateInputs("Chair:ENTITY MainParent:ENTITY SecondaryParent:ENTITY TertiaryParent:ENTITY Filter:ARRAY")
    updateOutputs(
        "HitPos:VECTOR CamDir:VECTOR CamAng:ANGLE CamPos:VECTOR " ..
        "CamState:NORMAL CamFOV:NORMAL CamActive:NORMAL"
    )

    Outputs.HitPos:update(ZERO_VECTOR)
    Outputs.CamDir:update(ZERO_VECTOR)
    Outputs.CamAng:update(ZERO_ANGLE)
    Outputs.CamPos:update(ZERO_VECTOR)
    Outputs.CamState:update(1)
    Outputs.CamFOV:update(100)
    Outputs.CamActive:update(0)

    addNetworkedVariable("Driver")
    addNetworkedVariable("ChairState")
    addNetworkedVariable("MainParentState")
    addNetworkedVariable("SecondaryParentState")
    addNetworkedVariable("TertiaryParentState")
    addNetworkedVariable("FilterState")

    local function refreshServerInputs()
        INPUTS.Chair = sanitizeEntityInput(InputValues.Chair, chip())
        INPUTS.MainParent = sanitizeEntityInput(InputValues.MainParent, chip())
        INPUTS.SecondaryParent = sanitizeEntityInput(InputValues.SecondaryParent, chip())
        INPUTS.TertiaryParent = sanitizeEntityInput(InputValues.TertiaryParent, chip())
        INPUTS.Filter = sanitizeTraceFilter(InputValues.Filter)
    end

    local function sendDriverClear(ply)
        if ply == nil or ply == chip() then return end

        net.start(CLEAR_DRIVER_NET)
        net.send(ply)
    end

    local function clearCameraOutputs()
        Outputs.HitPos:update(ZERO_VECTOR)
        Outputs.CamDir:update(ZERO_VECTOR)
        Outputs.CamAng:update(ZERO_ANGLE)
        Outputs.CamPos:update(ZERO_VECTOR)
        Outputs.CamState:update(1)
        Outputs.CamFOV:update(100)
        Outputs.CamActive:update(0)
    end

    local function hasCameraDriver()
        return SHARED.Driver ~= nil and SHARED.Driver ~= chip()
    end

    local function hasMainParent()
        return INPUTS.MainParent ~= nil and INPUTS.MainParent ~= chip()
    end

    local function refreshCameraActiveOutput()
        if hasCameraDriver() and hasMainParent() then
            Outputs.CamActive:update(1)
            return true
        end

        clearCameraOutputs()
        return false
    end

    local function queueDriverSnapshot()
        local driverVar = addNetworkedVariable("Driver")
        if driverVar ~= nil then
            driverVar.forceMeta = true
        end
        updateNetworkedVariable("Driver", SHARED.Driver)
        updateNetworkedVariable("ChairState", INPUTS.Chair)
        updateNetworkedVariable("MainParentState", INPUTS.MainParent)
        updateNetworkedVariable("SecondaryParentState", INPUTS.SecondaryParent)
        updateNetworkedVariable("TertiaryParentState", INPUTS.TertiaryParent)
        updateNetworkedVariable("FilterState", INPUTS.Filter)
    end

    local function setDriver(ply)
        local previousDriver = SHARED.Driver

        if previousDriver ~= nil and previousDriver ~= chip() and previousDriver ~= ply then
            sendDriverClear(previousDriver)
        end

        if ply ~= nil then
            SHARED.Driver = ply
            networkTargetSet(ply)
            refreshCameraActiveOutput()
            queueDriverSnapshot()
        else
            SHARED.Driver = chip()
            clearCameraOutputs()
            networkTargetSet({})
        end
    end

    local function handleChairChange(value)
        INPUTS.Chair = sanitizeEntityInput(value, chip())

        if SHARED.Driver ~= nil and SHARED.Driver ~= chip() then
            updateNetworkedVariable("ChairState", INPUTS.Chair)
        end
    end

    local function handleFilterChange(value)
        INPUTS.Filter = sanitizeTraceFilter(value)
        if SHARED.Driver ~= nil and SHARED.Driver ~= chip() then
            updateNetworkedVariable("FilterState", INPUTS.Filter)
        end
    end

    Inputs.Chair:addTrigger(handleChairChange)
    Inputs.MainParent:addTrigger(function(value)
        INPUTS.MainParent = sanitizeEntityInput(value, chip())
        if SHARED.Driver ~= nil and SHARED.Driver ~= chip() then
            updateNetworkedVariable("MainParentState", INPUTS.MainParent)
        end
        refreshCameraActiveOutput()
    end)
    Inputs.SecondaryParent:addTrigger(function(value)
        INPUTS.SecondaryParent = sanitizeEntityInput(value, chip())
        if SHARED.Driver ~= nil and SHARED.Driver ~= chip() then
            updateNetworkedVariable("SecondaryParentState", INPUTS.SecondaryParent)
        end
    end)
    Inputs.TertiaryParent:addTrigger(function(value)
        INPUTS.TertiaryParent = sanitizeEntityInput(value, chip())
        if SHARED.Driver ~= nil and SHARED.Driver ~= chip() then
            updateNetworkedVariable("TertiaryParentState", INPUTS.TertiaryParent)
        end
    end)
    Inputs.Filter:addTrigger(handleFilterChange)

    refreshServerInputs()
    addDriverDetection(Inputs.Chair, function(driver)
        setDriver(driver)
    end, DRIVER_CHECK_INTERVAL)

    net.receive(SHARED.EntId .. " Server Update", function(_, ply)
        if SHARED.Driver ~= ply then return end

        local msgType = net.readInt(8)
        if not hasMainParent() then
            clearCameraOutputs()
            return
        end

        if msgType == MSG_MAIN_OUTPUTS then
            local hitPos = net.readVector()
            local camDir = net.readVector()
            local camAng = net.readAngle()
            local camPos = net.readVector()
            Outputs.HitPos:update(hitPos)
            Outputs.CamDir:update(camDir)
            Outputs.CamAng:update(camAng)
            Outputs.CamPos:update(camPos)
            Outputs.CamActive:update(1)
        elseif msgType == MSG_STATE then
            Outputs.CamState:update(net.readInt(8))
        elseif msgType == MSG_FOV then
            Outputs.CamFOV:update(net.readInt(8))
        end
    end)

end

if CLIENT then
    CamWorldPos = SETTINGS.CamWorldPos1
    CamRelPos = SETTINGS.CamRelPos1
    CamDist = SETTINGS.CamDist1
    CamParent = INPUTS.MainParent
    CamFOV = 100

    CamAng = Angle()
    CamHitPos = Vector()
    PrevCamState = 1
    CamState = 1

    LastScroll = timer.curtime()
    NextSend = timer.curtime()

    ZoomHeld = false
    PanMul = 1
    ClientActiveState = false
    ClientHooksInstalled = false
    local calcViewHookId = SHARED.EntId .. " CalcView"
    local inputPressedHookId = SHARED.EntId .. " InputPressed"
    local inputReleasedHookId = SHARED.EntId .. " InputReleased"
    local mouseMovedHookId = SHARED.EntId .. " MouseMoved"
    local mouseWheeledHookId = SHARED.EntId .. " MouseWheeled"

    local function refreshClientInputs()
        INPUTS.Chair = sanitizeEntityInput(NetworkedVariableValues.ChairState, chip())
        INPUTS.MainParent = sanitizeEntityInput(NetworkedVariableValues.MainParentState, chip())
        INPUTS.SecondaryParent = sanitizeEntityInput(NetworkedVariableValues.SecondaryParentState, chip())
        INPUTS.TertiaryParent = sanitizeEntityInput(NetworkedVariableValues.TertiaryParentState, chip())
        INPUTS.Filter = sanitizeTraceFilter(NetworkedVariableValues.FilterState)
        SHARED.Driver = NetworkedVariableValues.Driver or chip()
    end

    local function computeClientActiveState()
        refreshClientInputs()

        if SHARED.Driver == chip() then return false end
        if SHARED.Driver ~= SHARED.Player then return false end
        if INPUTS.MainParent == chip() then return false end
        return true
    end

    local function shouldHit(ent)
        if not isValid(ent) then return false end
        if ent:getOwner() == SHARED.Owner then return false end

        for i = 1, #INPUTS.Filter do
            if INPUTS.Filter[i] == ent then
                return false
            end
        end

        return true
    end

    local function sendState(msgType, valueA, valueB, valueC, valueD)
        net.start(SHARED.EntId .. " Server Update")
            net.writeInt(msgType, 8)

            if msgType == MSG_MAIN_OUTPUTS then
                net.writeVector(valueA)
                net.writeVector(valueB)
                net.writeAngle(valueC)
                net.writeVector(valueD)
            else
                net.writeInt(valueA, 8)
            end
        net.send()
    end

    local function applyState(state)
        refreshClientInputs()
        if state == 1 then
            CamWorldPos = SETTINGS.CamWorldPos1
            CamRelPos = SETTINGS.CamRelPos1
            CamDist = SETTINGS.CamDist1
            CamParent = INPUTS.MainParent
        elseif state == 2 then
            CamWorldPos = SETTINGS.CamWorldPos2
            CamRelPos = SETTINGS.CamRelPos2
            CamDist = SETTINGS.CamDist2
            CamParent = INPUTS.SecondaryParent
        elseif state == 3 then
            CamWorldPos = SETTINGS.CamWorldPos3
            CamRelPos = SETTINGS.CamRelPos3
            CamDist = SETTINGS.CamDist3
            CamParent = INPUTS.TertiaryParent
        end
    end

    local function resetCam()
        CamFOV = 100
        CamState = 1
        PanMul = 1
        ZoomHeld = false
        applyState(CamState)
    end

    local function uninstallClientHooks()
        if not ClientHooksInstalled then return end

        hook.remove("CalcView", calcViewHookId)
        hook.remove("InputPressed", inputPressedHookId)
        hook.remove("InputReleased", inputReleasedHookId)
        hook.remove("MouseMoved", mouseMovedHookId)
        hook.remove("MouseWheeled", mouseWheeledHookId)
        ClientHooksInstalled = false
    end

    local modifyState

    local function installClientHooks()
        if ClientHooksInstalled then return end

        hook.add("CalcView", calcViewHookId, function()
            if not isValid(CamParent) then
                return
            end

            local now = timer.curtime()
            local forward = CamAng:getForward()
            local origin = CamParent:localToWorld(CamRelPos) + CamWorldPos - forward * CamDist

            if NextSend < now then
                NextSend = now + 0.05
                local traceData = trace.line(origin, origin + forward * 999999, shouldHit)
                if traceData and traceData.HitPos then
                    CamHitPos = traceData.HitPos
                else
                    CamHitPos = origin + forward * 999999
                end
                sendState(MSG_MAIN_OUTPUTS, CamHitPos, forward, normalizeAngleCopy(CamAng), origin)
            end

            return {
                origin = origin,
                angles = CamAng,
                fov = CamFOV
            }
        end)

        hook.add("InputPressed", inputPressedHookId, function(button)
            if button == 79 then
                if INPUTS.TertiaryParent == chip() then return end

                if CamState ~= 3 then
                    PrevCamState = CamState
                    CamState = 3
                else
                    CamState = PrevCamState
                end

                modifyState()
            elseif button == 32 then
                if (CamState + 1 == 2 and INPUTS.SecondaryParent == chip()) or CamState == 3 then
                    return
                end

                CamState = CamState + 1
                if CamState > 2 then CamState = 1 end
                modifyState()
            elseif button == 12 then
                CamFOV = 27.27
                ZoomHeld = true
                PanMul = CamFOV / 100 * 2
                sendState(MSG_FOV, math.round(CamFOV))
            end
        end)

        hook.add("InputReleased", inputReleasedHookId, function(button)
            if button == 12 then
                ZoomHeld = false
                CamFOV = 100
                PanMul = 1
                sendState(MSG_FOV, math.round(CamFOV))
            end
        end)

        hook.add("MouseMoved", mouseMovedHookId, function(x, y)
            CamAng.pitch = clampValue(CamAng.pitch + (y / 40) * PanMul, -89, 89)
            CamAng.yaw = CamAng.yaw - (x / 40) * PanMul
        end)

        hook.add("MouseWheeled", mouseWheeledHookId, function(delta)
            local now = timer.curtime()
            if (now - LastScroll) < 0.01 then return end

            CamFOV = clampValue(CamFOV / 4 ^ delta, 1, 100)
            PanMul = CamFOV / 100 * (ZoomHeld and 2 or 1)

            LastScroll = now
            sendState(MSG_FOV, math.round(CamFOV))
        end)

        ClientHooksInstalled = true
    end

    local function syncClientActivation()
        local active = computeClientActiveState()

        if active and not ClientActiveState then
            ClientActiveState = true
            resetCam()
            installClientHooks()
            return
        end

        if not active and ClientActiveState then
            ClientActiveState = false
            uninstallClientHooks()
            resetCam()
            return
        end

        if active then
            installClientHooks()
        else
            uninstallClientHooks()
        end
    end

    modifyState = function()
        applyState(CamState)
        sendState(MSG_STATE, CamState)
    end

    NetworkedVariables.Driver:addTrigger(function()
        syncClientActivation()
    end)
    NetworkedVariables.ChairState:addTrigger(function()
        syncClientActivation()
    end)
    NetworkedVariables.MainParentState:addTrigger(function()
        syncClientActivation()
    end)
    NetworkedVariables.SecondaryParentState:addTrigger(function()
        syncClientActivation()
    end)
    NetworkedVariables.TertiaryParentState:addTrigger(function()
        syncClientActivation()
    end)
    NetworkedVariables.FilterState:addTrigger(function()
        syncClientActivation()
    end)
    syncClientActivation()

    net.receive(CLEAR_DRIVER_NET, function()
        SHARED.Driver = chip()
        syncClientActivation()
        resetCam()
    end)
end
