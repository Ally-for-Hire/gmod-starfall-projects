-- Seat-driver camera controller. Helper owns wire, input mirroring, driver detection, and NetVar sync.

require("allylib/general_algorithms.txt")
require("allylib/simple_server-side_wire_helper.txt")

local CHIP = chip()
local OWNER = owner()
local PLAYER = player()
local ENT_ID = CHIP:entIndex()

local DRIVER_CHECK_INTERVAL = 0.5
local DEFAULT_FOV = 100
local ZOOM_FOV = 27.27
local TRACE_LENGTH = 999999
local SEND_INTERVAL = 0.05
local SCROLL_INTERVAL = 0.01
local MOUSE_SCALE = 40

local MSG_MAIN_OUTPUTS = 1
local MSG_STATE = 2
local MSG_FOV = 3
local NET_UPDATE = ENT_ID .. " Server Update"
local ZERO_VECTOR = Vector()
local ZERO_ANGLE = Angle()

local CameraModes = {
    {
        parent = "MainParent",
        worldPos = Vector(0, 0, 140),
        relPos = Vector(-50, 0, 0),
        dist = 350
    },
    {
        parent = "SecondaryParent",
        worldPos = Vector(0, 0, 160),
        relPos = Vector(-50, 0, 0),
        dist = 0
    },
    {
        parent = "TertiaryParent",
        worldPos = Vector(0, 0, 0),
        relPos = Vector(250, 0, 0),
        dist = 0
    }
}

if SERVER then
    local Driver = CHIP

    syncInputCommands(true)
    updateInputs("Chair:ENTITY MainParent:ENTITY SecondaryParent:ENTITY TertiaryParent:ENTITY Filter:ARRAY")
    updateOutputs(
        "HitPos:VECTOR CamDir:VECTOR CamAng:ANGLE CamPos:VECTOR " ..
        "CamState:NORMAL CamFOV:NORMAL CamActive:NORMAL"
    )

    addNetworkedVariable("Driver")
    Inputs.MainParent:enableNetworking(true)
    Inputs.SecondaryParent:enableNetworking(true)
    Inputs.TertiaryParent:enableNetworking(true)
    Inputs.Filter:enableNetworking(true)

    local function parentInput(name)
        return sanitizeEntityInput(InputValues[name], CHIP)
    end

    local function driverActive()
        return Driver ~= nil and Driver ~= CHIP and parentInput("MainParent") ~= CHIP
    end

    local function clearOutputs()
        Outputs.HitPos:update(ZERO_VECTOR)
        Outputs.CamDir:update(ZERO_VECTOR)
        Outputs.CamAng:update(ZERO_ANGLE)
        Outputs.CamPos:update(ZERO_VECTOR)
        Outputs.CamState:update(1)
        Outputs.CamFOV:update(DEFAULT_FOV)
        Outputs.CamActive:update(0)
    end

    local function updateActiveOutput()
        if driverActive() then
            Outputs.CamActive:update(1)
        else
            clearOutputs()
        end
    end

    local function sendDriverSnapshot(ply)
        updateNetworkedVariables({
            Driver = Driver,
            MainParent = parentInput("MainParent"),
            SecondaryParent = parentInput("SecondaryParent"),
            TertiaryParent = parentInput("TertiaryParent"),
            Filter = sanitizeTraceFilter(InputValues.Filter)
        })
        if ply ~= nil and ply ~= CHIP then sendNetworkedVariablesTo(ply) end
    end

    local function setDriver(ply)
        local previousDriver = Driver

        if previousDriver ~= nil and previousDriver ~= CHIP and previousDriver ~= ply then
            Driver = CHIP
            updateNetworkedVariable("Driver", Driver, 0)
            sendNetworkedVariablesTo(previousDriver)
        end

        Driver = ply or CHIP
        updateActiveOutput()

        if Driver ~= CHIP then
            networkTargetSet(Driver)
            sendDriverSnapshot(Driver)
        else
            updateNetworkedVariable("Driver", Driver, 0)
            if previousDriver ~= nil and previousDriver ~= CHIP then
                sendNetworkedVariablesTo(previousDriver)
            end
            networkTargetSet({})
        end
    end

    clearOutputs()

    Inputs.MainParent:addTrigger(function()
        updateActiveOutput()
    end)

    addDriverDetection(Inputs.Chair, function(driver)
        setDriver(driver)
    end, DRIVER_CHECK_INTERVAL)

    net.receive(NET_UPDATE, function(_, ply)
        if Driver ~= ply then return end
        if parentInput("MainParent") == CHIP then
            clearOutputs()
            return
        end

        local msgType = net.readInt(8)

        if msgType == MSG_MAIN_OUTPUTS then
            Outputs.HitPos:update(net.readVector())
            Outputs.CamDir:update(net.readVector())
            Outputs.CamAng:update(net.readAngle())
            Outputs.CamPos:update(net.readVector())
            Outputs.CamActive:update(1)
        elseif msgType == MSG_STATE then
            Outputs.CamState:update(net.readInt(8))
        elseif msgType == MSG_FOV then
            Outputs.CamFOV:update(net.readInt(8))
        end
    end)
end

if CLIENT then
    local Active = false
    local HooksInstalled = false
    local Mode = 1
    local PrevMode = 1
    local FOV = DEFAULT_FOV
    local PanMul = 1
    local ZoomHeld = false
    local CamAng = Angle()
    local NextSend = timer.curtime()
    local LastScroll = timer.curtime()

    local calcViewHookId = ENT_ID .. " CalcView"
    local inputPressedHookId = ENT_ID .. " InputPressed"
    local inputReleasedHookId = ENT_ID .. " InputReleased"
    local mouseMovedHookId = ENT_ID .. " MouseMoved"
    local mouseWheeledHookId = ENT_ID .. " MouseWheeled"

    local function parentForMode(mode)
        return sanitizeEntityInput(NetVars[CameraModes[mode].parent]:get(), CHIP)
    end

    local function cameraActive()
        return NetVars.Driver:get() == PLAYER and parentForMode(1) ~= CHIP
    end

    local function traceFilter(ent)
        if not isValid(ent) then return false end
        if ent:getOwner() == OWNER then return false end

        local filter = sanitizeTraceFilter(NetVars.Filter:get())
        for i = 1, #filter do
            if filter[i] == ent then return false end
        end

        return true
    end

    local function sendState(msgType, valueA, valueB, valueC, valueD)
        net.start(NET_UPDATE)
            net.writeInt(msgType, 8)

            if msgType == MSG_MAIN_OUTPUTS then
                net.writeVector(valueA)
                net.writeVector(valueB)
                net.writeAngle(valueC)
                net.writeVector(valueD)
            else
                net.writeInt(valueA, 8)
            end
        net.send()
    end

    local function resetCamera()
        Mode = 1
        PrevMode = 1
        FOV = DEFAULT_FOV
        PanMul = 1
        ZoomHeld = false
    end

    local function uninstallHooks()
        if not HooksInstalled then return end

        hook.remove("CalcView", calcViewHookId)
        hook.remove("InputPressed", inputPressedHookId)
        hook.remove("InputReleased", inputReleasedHookId)
        hook.remove("MouseMoved", mouseMovedHookId)
        hook.remove("MouseWheeled", mouseWheeledHookId)
        HooksInstalled = false
    end

    local function installHooks()
        if HooksInstalled then return end

        hook.add("CalcView", calcViewHookId, function()
            local camera = CameraModes[Mode]
            local parent = parentForMode(Mode)
            if not isValid(parent) then return end

            local forward = CamAng:getForward()
            local origin = parent:localToWorld(camera.relPos) + camera.worldPos - forward * camera.dist
            local now = timer.curtime()

            if NextSend < now then
                NextSend = now + SEND_INTERVAL
                local traceData = trace.line(origin, origin + forward * TRACE_LENGTH, traceFilter)
                local hitPos = traceData and traceData.HitPos or origin + forward * TRACE_LENGTH
                sendState(MSG_MAIN_OUTPUTS, hitPos, forward, normalizeAngleCopy(CamAng), origin)
            end

            return {
                origin = origin,
                angles = CamAng,
                fov = FOV
            }
        end)

        hook.add("InputPressed", inputPressedHookId, function(button)
            if button == 79 then
                if parentForMode(3) == CHIP then return end

                if Mode ~= 3 then
                    PrevMode = Mode
                    Mode = 3
                else
                    Mode = PrevMode
                end
                sendState(MSG_STATE, Mode)
            elseif button == 32 then
                if Mode == 3 or Mode == 1 and parentForMode(2) == CHIP then return end

                Mode = Mode + 1
                if Mode > 2 then Mode = 1 end
                sendState(MSG_STATE, Mode)
            elseif button == 12 then
                FOV = ZOOM_FOV
                ZoomHeld = true
                PanMul = FOV / DEFAULT_FOV * 2
                sendState(MSG_FOV, math.round(FOV))
            end
        end)

        hook.add("InputReleased", inputReleasedHookId, function(button)
            if button ~= 12 then return end

            ZoomHeld = false
            FOV = DEFAULT_FOV
            PanMul = 1
            sendState(MSG_FOV, math.round(FOV))
        end)

        hook.add("MouseMoved", mouseMovedHookId, function(x, y)
            CamAng.pitch = clampValue(CamAng.pitch + y / MOUSE_SCALE * PanMul, -89, 89)
            CamAng.yaw = CamAng.yaw - x / MOUSE_SCALE * PanMul
        end)

        hook.add("MouseWheeled", mouseWheeledHookId, function(delta)
            local now = timer.curtime()
            if now - LastScroll < SCROLL_INTERVAL then return end

            FOV = clampValue(FOV / 4 ^ delta, 1, DEFAULT_FOV)
            PanMul = FOV / DEFAULT_FOV * (ZoomHeld and 2 or 1)
            LastScroll = now
            sendState(MSG_FOV, math.round(FOV))
        end)

        HooksInstalled = true
    end

    local function syncActivation()
        local shouldBeActive = cameraActive()
        if shouldBeActive == Active then
            if Active then installHooks() end
            return
        end

        Active = shouldBeActive
        if Active then
            resetCamera()
            installHooks()
        else
            uninstallHooks()
            resetCamera()
        end
    end

    NetVars.Driver:addTrigger(syncActivation)
    NetVars.MainParent:addTrigger(syncActivation)
    NetVars.SecondaryParent:addTrigger(syncActivation)
    NetVars.TertiaryParent:addTrigger(syncActivation)
    syncActivation()
end
