function transposeMatrix (a) -- assuming a 3x3 matrix
    return {
        {a[1][1], a[2][1], a[3][1]}, 
        {a[1][2], a[2][2], a[3][2]}, 
        {a[1][3], a[2][3], a[3][3]}
    }
end
