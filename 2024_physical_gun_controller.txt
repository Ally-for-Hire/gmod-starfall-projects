--@server

require("allylib/general_algorithms.txt")
require("allylib/simple_server-side_wire_helper.txt")

updateInputs("Aimer:ENTITY HitPos:VECTOR LockTurret:NORMAL TargetPos:VECTOR LeadPos:VECTOR TargetName:STRING")
updateOutputs("GunTarget:VECTOR")

DT = game.getTickInterval()
ROTATE_SPEED = Angle(240) * DT * 5
MAX_ELEVATION = 75
MAX_DEPRESSION = 10
LOCK_DISTANCE = 100000

KP = 450
KI = 15
KD = 40

Abs = math.abs

Gun = nil
Inertia = Angle()
Integral = Angle()
TurretLocked = true
LockedPos = chip():worldToLocal(chip():getPos() + chip():getForward() * LOCK_DISTANCE)
LastLockValue = 0
LockInitialized = false

local function clampAngle(ang, lowAng, highAng)
    return Angle(
        math.clamp(ang.pitch, lowAng.pitch, highAng.pitch),
        math.clamp(ang.yaw, lowAng.yaw, highAng.yaw),
        math.clamp(ang.roll, lowAng.roll, highAng.roll)
    )
end

local function zeroRoll(ang)
    return Angle(ang.pitch, ang.yaw, 0)
end

local function negateAngle(ang)
    return Angle(-ang.pitch, -ang.yaw, -ang.roll)
end

local function updateGunState(value)
    Gun = value
    if not Gun or not Gun:isValid() then
        Inertia = Angle()
        LockInitialized = false
        return false
    end

    local gunInertia = Gun:getInertia()
    Inertia = Angle(gunInertia.y, gunInertia.z, 0)
    LockInitialized = false
    return true
end

local function getHoldAimPos(currentGun)
    if currentGun and currentGun:isValid() then
        return getGunOrigin(currentGun) + currentGun:getForward() * LOCK_DISTANCE
    end
    return chip():getPos() + chip():getForward() * LOCK_DISTANCE
end

local function getUnlockedTargetSnapshot(currentGun)
    local hitPos = InputValues.HitPos or Vector()
    local targetPos = InputValues.TargetPos or Vector()
    local leadPos = InputValues.LeadPos or Vector()
    local targetName = InputValues.TargetName or ""

    if targetName ~= "" and hasVectorValue(leadPos) then
        if hasVectorValue(targetPos) then
            return leadPos, targetPos
        end
        return leadPos, leadPos
    end

    if hasVectorValue(hitPos) then
        return hitPos, hitPos
    end

    if hasVectorValue(targetPos) then
        return targetPos, targetPos
    end

    local holdAimPos = getHoldAimPos(currentGun)
    return holdAimPos, holdAimPos
end

local function getTargetSnapshot(currentGun)
    if TurretLocked then
        local lockedWorldPos = chip():localToWorld(LockedPos)
        return lockedWorldPos, lockedWorldPos
    end

    return getUnlockedTargetSnapshot(currentGun)
end

local function captureLockPos(currentGun)
    LockedPos = chip():worldToLocal(getHoldAimPos(currentGun))
    LockInitialized = true
end

Inputs.Aimer:addTrigger(function(value)
    updateGunState(value)
end)

Inputs.LockTurret:addTrigger(function(value)
    local pressed = value ~= 0 and LastLockValue == 0
    LastLockValue = value
    if not pressed then return end

    TurretLocked = not TurretLocked
    if TurretLocked then
        captureLockPos(Gun)
    end
end)

hook.add("tick", "main", function()
    if InputValues.Aimer ~= Gun then
        updateGunState(InputValues.Aimer)
    end
    if not Gun or not Gun:isValid() or Gun:isFrozen() then
        Integral = Angle()
        Outputs.GunTarget:update(Vector())
        return
    end
    if TurretLocked and not LockInitialized then
        captureLockPos(Gun)
    end

    local aimPos, gunTarget = getTargetSnapshot(Gun)
    Outputs.GunTarget:update(gunTarget)

    local toTarget = aimPos - getGunOrigin(Gun)
    if not hasVectorValue(toTarget) then
        Integral = Angle()
        return
    end

    local gunAng = toTarget:getAngle()
    local chipAngs = chip():localToWorldAngles(Angle())
    local chipPitch = chipAngs.pitch
    local relPitch = gunAng.pitch > 90 and -(360 - gunAng.pitch) or gunAng.pitch
    relPitch = math.clamp(relPitch, -MAX_ELEVATION + chipPitch, MAX_DEPRESSION + chipPitch)
    gunAng.pitch = gunAng.pitch > 90 and relPitch + 360 or relPitch

    local force = zeroRoll(Gun:worldToLocalAngles(gunAng))

    if Abs(force.yaw) > 45 then
        force.pitch = 0
        Integral = Angle()
    elseif Abs(force.pitch) + Abs(force.yaw) < 5 then
        Integral = Integral + force
        Integral = clampAngle(Integral, negateAngle(ROTATE_SPEED), ROTATE_SPEED)
    else
        Integral = Angle()
    end

    force = clampAngle(force, negateAngle(ROTATE_SPEED), ROTATE_SPEED)

    local angVel = zeroRoll(Gun:getAngleVelocityAngle())
    local output = force * KP + Integral * KI - angVel * KD
    Gun:applyAngForce(output * Inertia)
end)
