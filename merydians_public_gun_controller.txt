require("allylib/general_algorithms.txt")
require("allylib/simple_server-side_wire_helper.txt")
require("allylib/merys_integrated_ballistics_computer.txt")

updateInputs("Active:NORMAL M1:NORMAL Gun:ENTITY Base:ENTITY HitPos:VECTOR")
updateOutputs("TurretActive:NORMAL HAimPos:VECTOR VAimPos:VECTOR TTA:NORMAL Fire:NORMAL")

CurTime = timer.curtime
ZERO_VEC = Vector()

ACTIVE_INTERVAL = 0.05
IDLE_INTERVAL = 0.2

NextThink = 0

local function wakeThink()
    NextThink = 0
end

Inputs.Gun:addTrigger(function(gun)
    if gun and gun:isValid() and CFBA_SetGun then
        CFBA_SetGun(gun)
    end

    wakeThink()
end)

Inputs.Base:addTrigger(wakeThink)
Inputs.HitPos:addTrigger(wakeThink)
Inputs.Active:addTrigger(wakeThink)

local function main()
    local now = CurTime()
    local gun = InputValues.Gun
    local base = InputValues.Base
    local hitPos = InputValues.HitPos or ZERO_VEC
    local manualFire = (InputValues.M1 or 0) ~= 0
    local active = (InputValues.Active or 0) ~= 0
    local gunValid = gun and gun:isValid()
    local canSolve = active
        and gunValid
        and base
        and base:isValid()
        and hitPos:getLength() > 0

    Outputs.TurretActive:update(active and 1 or 0)
    Outputs.Fire:update(manualFire and 1 or 0)

    if now < NextThink then
        return
    end

    local aimPos = hitPos
    local tta = 0

    if canSolve and gun:acfReady() and CFBA_SolveStatic then
        local _, staticAimPos, staticTta = CFBA_SolveStatic(base, hitPos)
        if hasVectorValue(staticAimPos) then
            aimPos = staticAimPos
        end
        if type(staticTta) == "number" then
            tta = staticTta
        end
    end

    Outputs.HAimPos:update(aimPos)
    Outputs.VAimPos:update(aimPos)
    Outputs.TTA:update(tta)

    NextThink = now + (active and ACTIVE_INTERVAL or IDLE_INTERVAL)
end

hook.add("think", "SF_MPGC", main)
