-- Constants / state

G  = physenv.getGravity()           -- world gravity (GU/s^2)
U  = 39.3701                        -- m -> GU

local G2D       = Vector2(0, G[3])  -- vertical accel in 2D
local GZ        = -G[3]
local TIME_STEP = game.getTickInterval()  -- finer dt for accuracy
local VALIDGUN  = false

local QUOTA_RATIO_MAX    = 0.92     -- throttle generation if near quota
local STEP_INTERVAL      = 0.05     -- seconds between generation slices
local MAX_STEPS_PER_TICK = 6
local MAX_DISTANCE_GU    = 50000
local MAX_DATASET_POINTS = 768
local MIN_DISTANCE_STEP  = 0.5
local LOW_ARC_EPS        = 2
local LOW_ARC_DESCENTS   = 3

local EPS_MV_ABS     = 0.01
local EPS_MV_REL     = 1e-6
local EPS_DRAG_ABS   = 1e-6
local EPS_DRAG_REL   = 1e-4
local EPS_MUZZLE_ABS = 0.01
local EPS_MUZZLE_REL = 1e-6

local muzzleVelocity = 0            -- m/s from ACF
local dragCoef       = 0            -- ACF-exported drag coefficient (scaled later)
local MUZZLE_DIST    = 0            -- GU, origin -> muzzle
local ammoType       = ""

local DATASET    = {}               -- monotonic lookup { {distance = D, time = T}, ... }
local COEFFS     = nil              -- kept as readiness flag / lookup alias for API compatibility
local TIME_SCALE = 1

local TARGET_SAMPLE_POS_EPS = 1
local TARGET_SAMPLE_VEL_EPS = 1

local TRACK_ACCEL         = Vector()
local TRACK_TURN_OMEGA    = Vector()
local TRACK_SAMPLE_POS    = Vector()
local TRACK_SAMPLE_VEL    = Vector()
local TRACK_SAMPLE_TIME   = 0
local TRACK_SAMPLE_DT     = 0
local ZERO_VEC            = Vector()

local GEN_ANGLE        = 0          -- degrees
local GEN_STEP_DEFAULT = 0.01      -- degrees
local GEN_STEP_GROWTH  = 1.5
local GEN_STEP_MAX     = 0.25
local GEN_STEP         = GEN_STEP_DEFAULT
local GEN_MAX_DISTANCE = 0
local GEN_DONE         = false
local GEN_TOKEN        = 0
local GEN_LAST_DISTANCE = 0
local GEN_DESCENT_COUNT = 0

local D_MIN   = 0
local D_MAX   = 0
local D_SCALE = 1

local MAX_TIME = 6                  -- s, safety cap

local DEBUG_ENABLED = false

------------------------------------------------------------
-- Debug helpers
------------------------------------------------------------

local function logStatus(...)
    print(Color(0, 255, 255), "[C-FBA]: ", Color(255, 255, 255), ...)
end

local function dprint(...)
    if not DEBUG_ENABLED then return end
    logStatus(...)
end

local function nearlyEqual(a, b, absEps, relEps)
    if a == b then return true end
    local diff = math.abs(a - b)
    if diff <= absEps then return true end
    local maxAbs = math.max(math.abs(a), math.abs(b))
    if maxAbs <= 0 then return diff <= absEps end
    return diff <= maxAbs * relEps
end

local function lerp(a, b, t)
    return a + (b - a) * t
end

local function simulateImpact(angleDeg, muzzleVelUnits, dragCoefUnits, muzzleDistUnits)
    local dt       = TIME_STEP
    local angleRad = math.rad(angleDeg)
    local dir      = Vector2(math.cos(angleRad), math.sin(angleRad))

    -- start at the muzzle
    local pos      = dir * muzzleDistUnits
    local vel      = dir * muzzleVelUnits
    local time     = 0
    local prevPos  = pos
    local prevTime = time

    while time < MAX_TIME and pos[2] > 0 do
        prevPos  = pos
        prevTime = time

        local speed = vel:getLength()
        if speed <= 0 then break end

        -- dragNum = (dragCoef * |v|^2) / 40
        -- then scaled by dt, clamped like acf_ballistics_lib
        local dragNum = (dragCoefUnits * speed * speed) / 40
        local clamp   = speed * 0.9
        dragNum       = math.min(dragNum * dt, clamp)
        local dragVec = vel:getNormalized() * dragNum

        pos  = pos + vel * dt
        vel  = vel + (G2D * dt - dragVec)
        time = time + dt
    end

    -- interpolate when we cross back to muzzle height (y = 0)
    if pos[2] <= 0 and prevPos[2] > 0 then
        local denom = prevPos[2] - pos[2]
        if math.abs(denom) > 1e-6 then
            local frac      = prevPos[2] / denom
            local impactPos = prevPos + (pos - prevPos) * frac
            local impactT   = prevTime + (time - prevTime) * frac
            return impactT, impactPos[1]
        end
    end

    return nil, nil
end

-- Lookup building / interpolation
------------------------------------------------------------

local function fitCurve()
    local n = #DATASET
    if n < 2 then
        COEFFS     = nil
        TIME_SCALE = 1
        dprint("Not enough points to build lookup (have ", n, ")")
        return
    end

    D_MIN = math.huge
    D_MAX = 0
    for i = 1, n do
        local D = DATASET[i].distance
        if D < D_MIN then D_MIN = D end
        if D > D_MAX then D_MAX = D end
    end

    if D_MAX <= 0 or D_MIN == math.huge then
        COEFFS     = nil
        TIME_SCALE = 1
        dprint("buildLookup: invalid distance range")
        return
    end

    D_SCALE = D_MAX
    COEFFS = DATASET
    dprint("Lookup done: points=", n,
           " D_MIN=", D_MIN, " D_MAX=", D_MAX)
    logStatus("Data Generation finished for " .. I.Gun:acfAmmoType())
end

local function interpolateDistanceToTime(D)
    local count = #DATASET
    if count <= 0 then return 0 end
    if D <= DATASET[1].distance then return DATASET[1].time end
    if D >= DATASET[count].distance then return DATASET[count].time end

    local low = 1
    local high = count
    while high - low > 1 do
        local mid = math.floor((low + high) * 0.5)
        if DATASET[mid].distance < D then
            low = mid
        else
            high = mid
        end
    end

    local a = DATASET[low]
    local b = DATASET[high]
    local span = b.distance - a.distance
    if span <= 0 then return a.time end
    local frac = (D - a.distance) / span
    return lerp(a.time, b.time, frac)
end

local function interpolateTimeToDistance(T)
    local count = #DATASET
    if count <= 0 then return 0 end
    if T <= DATASET[1].time then return DATASET[1].distance end
    if T >= DATASET[count].time then return DATASET[count].distance end

    local low = 1
    local high = count
    while high - low > 1 do
        local mid = math.floor((low + high) * 0.5)
        if DATASET[mid].time < T then
            low = mid
        else
            high = mid
        end
    end

    local a = DATASET[low]
    local b = DATASET[high]
    local span = b.time - a.time
    if span <= 0 then return a.distance end
    local frac = (T - a.time) / span
    return lerp(a.distance, b.distance, frac)
end

local function clampValue(value, minValue, maxValue)
    if value < minValue then return minValue end
    if value > maxValue then return maxValue end
    return value
end

local function vectorDeltaLen(a, b)
    return (a - b):getLength()
end

local function getTangentialAccel(vel, accel)
    local speed = vel:getLength()
    if speed <= 0 then
        return ZERO_VEC
    end

    local dir = vel / speed
    return dir * dir:dot(accel)
end

local function rotateAroundAxis(vec, axis, ang)
    local cosAng = math.cos(ang)
    local sinAng = math.sin(ang)
    return vec * cosAng
        + axis:cross(vec) * sinAng
        + axis * axis:dot(vec) * (1 - cosAng)
end

local function predictConstantTurnOffset(vel, turnOmega, accel, dt)
    if dt <= 0 then
        return ZERO_VEC
    end

    local omegaMag = turnOmega:getLength()
    local speed = vel:getLength()
    if speed <= 0 then
        return ZERO_VEC
    end

    local tangentialAccel = getTangentialAccel(vel, accel)
    if omegaMag <= 0 then
        return vel * dt + tangentialAccel * (0.5 * dt * dt)
    end

    local axis = turnOmega / omegaMag
    local velParallel = axis * axis:dot(vel)
    local velPerp = vel - velParallel

    local ang = omegaMag * dt
    local sinAng = math.sin(ang)
    local cosAng = math.cos(ang)
    local arcOffset = (
        velParallel * dt
        + velPerp * (sinAng / omegaMag)
        + axis:cross(velPerp) * ((1 - cosAng) / omegaMag)
    )

    return arcOffset + tangentialAccel * (0.5 * dt * dt)
end

local function advanceTargetState(pos, vel, accel, turnOmega, dt)
    if dt <= 0 then
        return pos, vel
    end

    local nextPos = pos + predictConstantTurnOffset(vel, turnOmega, accel, dt)
    local omegaMag = turnOmega:getLength()
    local speed = vel:getLength()
    if speed <= 0 then
        return nextPos, vel
    end

    local tangentialAccel = getTangentialAccel(vel, accel)
    if omegaMag <= 0 then
        return nextPos, vel + tangentialAccel * dt
    end

    local axis = turnOmega / omegaMag
    local rotatedVel = rotateAroundAxis(vel, axis, omegaMag * dt)
    local rotatedSpeed = rotatedVel:getLength()
    if rotatedSpeed <= 0 then
        return nextPos, rotatedVel
    end

    local tangentialDelta = tangentialAccel:dot(vel / speed) * dt
    local nextVel = rotatedVel + (rotatedVel / rotatedSpeed) * tangentialDelta
    return nextPos, nextVel
end

local function resetTargetTracking()
    TRACK_ACCEL = ZERO_VEC
    TRACK_TURN_OMEGA = ZERO_VEC
    TRACK_SAMPLE_POS = ZERO_VEC
    TRACK_SAMPLE_VEL = ZERO_VEC
    TRACK_SAMPLE_TIME = 0
    TRACK_SAMPLE_DT = 0
end

local function updateTargetTracking(targetPos, targetVel, now)
    if TRACK_SAMPLE_TIME <= 0 then
        TRACK_SAMPLE_POS = targetPos
        TRACK_SAMPLE_VEL = targetVel
        TRACK_SAMPLE_TIME = now
        TRACK_ACCEL = ZERO_VEC
        TRACK_TURN_OMEGA = ZERO_VEC
        return targetPos, targetVel
    end

    if vectorDeltaLen(targetPos, TRACK_SAMPLE_POS) > TARGET_SAMPLE_POS_EPS
    or vectorDeltaLen(targetVel, TRACK_SAMPLE_VEL) > TARGET_SAMPLE_VEL_EPS then
        local dt = now - TRACK_SAMPLE_TIME
        if dt > 0 then
            TRACK_SAMPLE_DT = dt
            TRACK_ACCEL = (targetVel - TRACK_SAMPLE_VEL) / dt

            local lastSpeed = TRACK_SAMPLE_VEL:getLength()
            local curSpeed = targetVel:getLength()
            local rawTurnOmega = ZERO_VEC
            if lastSpeed > 0 and curSpeed > 0 then
                local prevDir = TRACK_SAMPLE_VEL / lastSpeed
                local curDir = targetVel / curSpeed
                local crossDir = prevDir:cross(curDir)
                local crossLen = crossDir:getLength()
                if crossLen > 0 then
                    local axis = crossDir / crossLen
                    local turnAngle = math.acos(clampValue(prevDir:dot(curDir), -1, 1))
                    rawTurnOmega = axis * (turnAngle / dt)
                end
            end
            TRACK_TURN_OMEGA = rawTurnOmega
        else
            TRACK_ACCEL = ZERO_VEC
            TRACK_TURN_OMEGA = ZERO_VEC
        end

        TRACK_SAMPLE_POS = targetPos
        TRACK_SAMPLE_VEL = targetVel
        TRACK_SAMPLE_TIME = now
        return advanceTargetState(targetPos, targetVel, TRACK_ACCEL, TRACK_TURN_OMEGA, TRACK_SAMPLE_DT * 0.5)
    end

    local age = now - TRACK_SAMPLE_TIME + TRACK_SAMPLE_DT * 0.5
    if age < 0 then
        age = 0
    end
    return advanceTargetState(targetPos, targetVel, TRACK_ACCEL, TRACK_TURN_OMEGA, age)
end

local function solveIntercept(base, targetPos, targetVel)
    if not VALIDGUN or not isValid(I.Gun) then
        return targetPos, targetPos, 0
    end
    if not base or not base:isValid() then
        return targetPos, targetPos, 0
    end
    if not ballisticDistance then
        return targetPos, targetPos, 0
    end

    local gun = I.Gun
    local gpos = gun:getMassCenterW() or gun:getPos()
    local dist = gpos:getDistance(targetPos)
    local bvel = base:getVelocity() or ZERO_VEC
    local baseToGun = targetPos - gpos
    local low = 0
    local high = 6
    local baseTime = ballisticTime(dist) or 0
    if baseTime > 0 then
        high = baseTime * 1.5
        if high < 0.5 then high = 0.5 end
        if high > 6 then high = 6 end
    end

    local function evalDiff(t)
        local tSq = t * t
        local drop = Vector(0, 0, 0.5 * GZ * tSq)
        local targetOffset = predictConstantTurnOffset(targetVel, TRACK_TURN_OMEGA, TRACK_ACCEL, t) - bvel * t
        local tlen = (baseToGun + targetOffset + drop):getLength()
        local slen = ballisticDistance(t)
        return slen - tlen
    end

    local slenHigh = ballisticDistance(high)
    if slenHigh <= 0 then
        return targetPos, targetPos, 0
    end

    local diffLow = evalDiff(low)
    local diffHigh = evalDiff(high)
    while diffHigh < 0 do
        local nextHigh = high * 1.5
        if nextHigh <= high then
            break
        end

        local nextSlenHigh = ballisticDistance(nextHigh)
        if nextSlenHigh <= slenHigh then
            break
        end

        high = nextHigh
        slenHigh = nextSlenHigh
        diffHigh = evalDiff(high)
    end

    if diffHigh < 0 then
        return targetPos, targetPos, 0
    end

    local tta = (low + high) * 0.5
    while high - low > 0.05 do
        local nextT = nil
        if diffHigh ~= diffLow then
            local secant = high - diffHigh * (high - low) / (diffHigh - diffLow)
            if secant > low and secant < high then
                nextT = secant
            end
        end
        if not nextT then
            nextT = (low + high) * 0.5
        end

        local diff = evalDiff(nextT)
        if diff < 0 then
            low = nextT
            diffLow = diff
        else
            high = nextT
            diffHigh = diff
        end

        tta = (low + high) * 0.5
    end

    local ttaSq = tta * tta
    local drop = Vector(0, 0, 0.5 * GZ * ttaSq)
    local targetOffset = predictConstantTurnOffset(targetVel, TRACK_TURN_OMEGA, TRACK_ACCEL, tta) - bvel * tta
    local predPos = targetPos + predictConstantTurnOffset(targetVel, TRACK_TURN_OMEGA, TRACK_ACCEL, tta)
    local aimPos = targetPos + targetOffset + drop
    return predPos, aimPos, tta
end

local function finishDataset()
    GEN_DONE = true

    if not VALIDGUN or not isValid(I.Gun) then
        dprint("finishDataset: gun invalid or not valid")
        return DATASET
    end

    if #DATASET > 0 then
        table.sort(DATASET, function(a, b)
            return a.distance < b.distance
        end)

        local lookup = {
            { distance = 0, time = 0 }
        }

        for i = 1, #DATASET do
            local sample = DATASET[i]
            local last = lookup[#lookup]
            if sample.distance > last.distance + MIN_DISTANCE_STEP and sample.time > last.time then
                lookup[#lookup + 1] = sample
            end
        end

        DATASET = lookup
    end

    dprint("Dataset done for ", I.Gun:acfName(),
           " ammo=", I.Gun:acfAmmoType(),
           " pts=", #DATASET,
           " maxD=", math.floor(GEN_MAX_DISTANCE))

    fitCurve()
    return DATASET
end

local function isOverBudget()
    local maxQuota = chip():getQuotaMax()
    if maxQuota <= 0 then return false end
    return (chip():getQuotaUsed() / maxQuota) > QUOTA_RATIO_MAX
end

local function stepDatasetGeneration(token)
    if token ~= GEN_TOKEN then return end
    if GEN_DONE or not VALIDGUN or not isValid(I.Gun) then return end

    local mv = muzzleVelocity * U
    local dc = dragCoef        -- already scaled (acfDragCoef * U)
    local md = MUZZLE_DIST
    if mv <= 0 then GEN_DONE = true return end

    local stepsThisTick, maxSteps = 0, MAX_STEPS_PER_TICK

    while stepsThisTick < maxSteps and not GEN_DONE do
        if isOverBudget() then break end
        if #DATASET >= MAX_DATASET_POINTS then
            GEN_DONE = true
            break
        end
        if GEN_ANGLE > 89 then
            GEN_DONE = true
            break
        end

        local t, d = simulateImpact(GEN_ANGLE, mv, dc, md)

        if t and d then
            if d > MAX_DISTANCE_GU then
                GEN_DONE = true
                break
            end

            if d + LOW_ARC_EPS < GEN_MAX_DISTANCE then
                GEN_DESCENT_COUNT = GEN_DESCENT_COUNT + 1
                if GEN_DESCENT_COUNT >= LOW_ARC_DESCENTS then
                    GEN_DONE = true
                    break
                end
            else
                GEN_DESCENT_COUNT = 0
                if d > GEN_MAX_DISTANCE then
                    GEN_MAX_DISTANCE = d
                end
                if d > GEN_LAST_DISTANCE + MIN_DISTANCE_STEP then
                    DATASET[#DATASET + 1] = { distance = d, time = t }
                    GEN_LAST_DISTANCE = d
                end
            end
        end

        GEN_ANGLE = GEN_ANGLE + GEN_STEP
        -- bound step growth so we still have samples at long range
        GEN_STEP  = math.min(GEN_STEP * GEN_STEP_GROWTH, GEN_STEP_MAX)
        stepsThisTick = stepsThisTick + 1
    end

    if GEN_DONE then
        finishDataset()
    else
        timer.simple(STEP_INTERVAL, function() stepDatasetGeneration(token) end)
    end
end

local function generateDataset()
    if not VALIDGUN or not isValid(I.Gun) then
        dprint("generateDataset: no valid gun")
        return
    end

    DATASET          = {}
    COEFFS           = nil
    TIME_SCALE       = 1
    GEN_ANGLE        = 0
    GEN_STEP         = GEN_STEP_DEFAULT
    GEN_MAX_DISTANCE = 0
    GEN_DONE         = false
    GEN_LAST_DISTANCE = 0
    GEN_DESCENT_COUNT = 0
    D_MIN            = 0
    D_MAX            = 0
    D_SCALE          = 1

    dprint("Starting dataset for ", I.Gun:acfName(),
           " ammo=", I.Gun:acfAmmoType())

    GEN_TOKEN = GEN_TOKEN + 1
    local token = GEN_TOKEN
    stepDatasetGeneration(token)
end

local function computeMuzzleDistance(gun)
    local idx = gun:lookupAttachment("muzzle")
    if idx and idx > 0 then
        local md = gun:getAttachment(idx)
        if md then
            local muzzlePos   = Vector(md[1], md[2], md[3])
            local massCenterW = gun:getMassCenterW()
            if massCenterW then
                return massCenterW:getDistance(muzzlePos)
            end
            return gun:localToWorld(Vector()):getDistance(muzzlePos)
        end
    end
    return 0
end

function updateValues()
    if not VALIDGUN then return end

    local gun = I.Gun
    if not isValid(gun) then
        VALIDGUN = false
        dprint("updateValues: gun invalid")
        return
    end

    local newMV = gun:acfMuzzleVel() or 0
    local newDrag = (gun:acfDragCoef() or 0) * U
    local newAmmoType = gun:acfAmmoType() or ""
    local newMuzzleDist = computeMuzzleDistance(gun)

    local mvChanged = not nearlyEqual(newMV, muzzleVelocity, EPS_MV_ABS, EPS_MV_REL)
    local dragChanged = not nearlyEqual(newDrag, dragCoef, EPS_DRAG_ABS, EPS_DRAG_REL)
    local muzzleChanged = not nearlyEqual(newMuzzleDist, MUZZLE_DIST, EPS_MUZZLE_ABS, EPS_MUZZLE_REL)

    local changed = mvChanged
        or dragChanged
        or (newAmmoType ~= ammoType)
        or muzzleChanged

    if changed then
        muzzleVelocity = newMV
        dragCoef       = newDrag
        ammoType       = newAmmoType
        MUZZLE_DIST    = newMuzzleDist

        dprint("updateValues: MV=", muzzleVelocity,
               " DragCoef=", dragCoef,
               " MuzzleDist=", MUZZLE_DIST,
               " AmmoType=", ammoType)

        generateDataset()
    end

    timer.simple(0.5, updateValues)
end

local function initGun(value)
    VALIDGUN = false
    resetTargetTracking()
    if not isValid(value) then
        dprint("initGun: value not valid")
        return
    end
    if value:getClass() ~= "acf_gun" then
        dprint("initGun: class is not acf_gun (got ", value:getClass(), ")")
        return
    end

    VALIDGUN = true
    I.Gun    = value
    dprint("initGun: gun set, class=", value:getClass())
    updateValues()
end

-- used by your controller (@include + require)
function CFBA_SetGun(value)
    initGun(value)
end

function CFBA_SetDebug(on)
    DEBUG_ENABLED = (on and true) or false
    dprint("CFBA_SetDebug: DEBUG_ENABLED=", DEBUG_ENABLED)
end

function CFBA_ResetTargetTracking()
    resetTargetTracking()
end

function CFBA_SolveIntercept(base, targetPos, targetVel)
    local now = timer.curtime()
    local currentPos, currentVel = updateTargetTracking(targetPos or ZERO_VEC, targetVel or ZERO_VEC, now)
    return solveIntercept(base, currentPos, currentVel)
end

------------------------------------------------------------
-- Public API: time / drop
------------------------------------------------------------

function ballisticTime(D)
    if not COEFFS or not D or D <= 0 then
        return 0
    end
    if D_MAX <= 0 then
        return 0
    end

    -- clamp distance into fitted domain
    if D < D_MIN then D = D_MIN end
    if D > D_MAX then D = D_MAX end

    local t = interpolateDistanceToTime(D)
    if t < 0 then t = 0 end

    dprint("ballisticTime: D=", D, " -> t=", t)
    return t
end

function ballisticDistance(T)
    if not COEFFS or not T or T <= 0 then
        return 0
    end
    if D_MAX <= 0 then
        return 0
    end

    local tMax = DATASET[#DATASET].time
    if T >= tMax then
        return D_MAX
    end

    local dist = interpolateTimeToDistance(T)
    dprint("ballisticDistance: T=", T, " -> D=", dist)
    return dist
end

function ballisticDropScale(t)
    -- still just base gravity, no extra shaping for now
    return 1
end
