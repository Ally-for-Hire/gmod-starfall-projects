--[[
Direct-global wire helper layered over simple_server-side_wire_helper.txt.

SSSWH remains the compatibility backend. This helper changes the front-end style:
- `updateInputs("Base:ENTITY RWR:ENTITY")` creates globals `Base` and `RWR`.
- `networkInputs("Base RWR", interval)` chooses which input globals replicate to clients.
- `addNetworkedVariable("TargetState", interval, {})` creates a global `TargetState`.
- Assigning `TargetState = {...}` on the server queues the networked variable update.
- `inputClk(name, fn)` and `netVarClk(name, fn)` keep triggers name-based.

Do not declare direct globals as `local`. Reads and writes must go through `_G`.
]]--

local base = require("simple_server-side_wire_helper.txt")

DirectWireValues = DirectWireValues or {}
DirectWireRegistered = DirectWireRegistered or {}
DirectWireNetworked = DirectWireNetworked or {}
DirectWireKinds = DirectWireKinds or {}
DirectWireTriggers = DirectWireTriggers or {}
DirectWireNetVarTriggers = DirectWireNetVarTriggers or {}
DirectWireInstalled = DirectWireInstalled or false

local BaseUpdateInputs = updateInputs
local BaseAddNetworkedVariable = addNetworkedVariable
local BaseSetNetworkedVariableInterval = setNetworkedVariableInterval
local BaseUpdateNetworkedVariable = updateNetworkedVariable

local DEFAULT_TABLE_NAMES = {
    "filter", "list", "data", "ids", "positions", "contacts", "targetstate", "table"
}

local DEFAULT_VECTOR_NAMES = {
    "pos", "vel", "vector", "aim"
}

local DEFAULT_BOOL_NAMES = {
    "active", "enabled", "ready", "lock"
}

local DEFAULT_STRING_NAMES = {
    "name", "text", "status", "type"
}

local DEFAULT_NUMBER_NAMES = {
    "count", "amount", "size", "armor", "state", "fov", "id"
}

local function nameContainsAny(name, patterns)
    name = string.lower(tostring(name or ""))
    for i = 1, #patterns do
        if string.find(name, patterns[i]) then return true end
    end
    return false
end

local function defaultEntityValue()
    local value = chip()
    if value and value.getPos and value.getForward then return value end
    return nil
end

local function isEntityDefaultName(name)
    name = string.lower(tostring(name or ""))
    if string.find(name, "ready$")
        or string.find(name, "ammotype$")
        or string.find(name, "ammocount$")
        or string.find(name, "ammocountmax$")
        or string.find(name, "magsize$")
        or string.find(name, "armor$")
        or string.find(name, "armormax$") then return false end

    return string.find(name, "driver")
        or string.find(name, "chair")
        or string.find(name, "parent")
        or string.find(name, "owner")
        or string.find(name, "entity")
        or string.match(name, "^gun%d*$")
        or name == "base"
end

local function defaultForName(name)
    local lowered = string.lower(tostring(name or ""))
    if isEntityDefaultName(lowered) then return defaultEntityValue() end
    if nameContainsAny(lowered, DEFAULT_TABLE_NAMES) then return {} end
    if nameContainsAny(lowered, DEFAULT_VECTOR_NAMES) then return Vector() end
    if string.find(lowered, "ang") or string.find(lowered, "angle") then return Angle() end
    if nameContainsAny(lowered, DEFAULT_BOOL_NAMES) then return false end
    if nameContainsAny(lowered, DEFAULT_STRING_NAMES) then return "" end
    if nameContainsAny(lowered, DEFAULT_NUMBER_NAMES) then return 0 end
    return 0
end

local function defaultForPort(name, portType)
    portType = string.upper(tostring(portType or "NORMAL"))
    if portType == "ENTITY" then return defaultEntityValue() end
    if portType == "VECTOR" then return Vector() end
    if portType == "ANGLE" then return Angle() end
    if portType == "ARRAY" or portType == "TABLE" then return {} end
    if portType == "STRING" then return "" end
    if portType == "BOOLEAN" or portType == "BOOL" then return false end
    if portType == "WIRELINK" then return nil end
    return defaultForName(name)
end

local function parseNames(value)
    local names = {}
    if value == nil or value == true then
        for name in pairs(DirectWireRegistered) do names[#names + 1] = name end
        return names
    end

    if type(value) == "string" then
        for token in string.gmatch(value, "[^,%s]+") do
            local name = string.match(token, "^([^:]+)")
            if name and name ~= "" then names[#names + 1] = name end
        end
        return names
    end

    if type(value) == "table" then
        for key, name in pairs(value) do
            if type(name) == "string" and name ~= "" then
                names[#names + 1] = name
            elseif type(key) == "string" and key ~= "" then
                names[#names + 1] = key
            end
        end
    end
    return names
end

local function runDirectTriggers(store, name, value)
    local list = store[name]
    if not list then return end
    for i = 1, #list do
        xpcall(list[i], function(err) print("[SWH] trigger error: ", err) end, value, name)
    end
end

local setDirectValue

local function installGlobalAccessors()
    if DirectWireInstalled then return end
    DirectWireInstalled = true

    local oldMeta = getmetatable(_G) or {}
    local oldIndex = oldMeta.__index
    local oldNewIndex = oldMeta.__newindex

    oldMeta.__index = function(tbl, key)
        if DirectWireRegistered[key] then return DirectWireValues[key] end
        if type(oldIndex) == "function" then return oldIndex(tbl, key) end
        if type(oldIndex) == "table" then return oldIndex[key] end
        return nil
    end

    oldMeta.__newindex = function(tbl, key, value)
        if DirectWireRegistered[key] then
            setDirectValue(key, value, "assign")
            return
        end

        if type(oldNewIndex) == "function" then
            oldNewIndex(tbl, key, value)
        elseif type(oldNewIndex) == "table" then
            oldNewIndex[key] = value
        else
            rawset(tbl, key, value)
        end
    end

    setmetatable(_G, oldMeta)
end

local function registerDirectName(name, defaultValue, kind)
    if type(name) ~= "string" or name == "" then return false end
    installGlobalAccessors()

    DirectWireRegistered[name] = true
    DirectWireKinds[name] = kind or DirectWireKinds[name] or "value"
    if DirectWireValues[name] == nil then DirectWireValues[name] = defaultValue end
    rawset(_G, name, nil)
    return true
end

setDirectValue = function(name, value, source)
    if not DirectWireRegistered[name] then return false end

    DirectWireValues[name] = value
    if DirectWireKinds[name] == "input" then InputValues[name] = value end

    local willNetwork = source == "assign" and SERVER and DirectWireNetworked[name]
    if willNetwork then
        BaseUpdateNetworkedVariable(name, value)
    end

    if source ~= "trigger" and source ~= "netvar" and not willNetwork then
        runDirectTriggers(DirectWireTriggers, name, value)
    end
    return true
end

function inputClk(name, callback)
    if type(name) ~= "string" or type(callback) ~= "function" then return nil end
    local input = Inputs[name]
    if input and input.addTrigger then return input:addTrigger(callback) end

    local list = InputTriggers[name] or {}
    InputTriggers[name] = list
    list[#list + 1] = callback
    return #list
end

function netVarClk(name, callback)
    if type(name) ~= "string" or type(callback) ~= "function" then return nil end
    return NetVars[name]:addTrigger(callback)
end

function changedClk(name, callback)
    if type(name) ~= "string" or type(callback) ~= "function" then return nil end
    local list = DirectWireTriggers[name] or {}
    DirectWireTriggers[name] = list
    list[#list + 1] = callback
    return #list
end

function updateInputs(str)
    local names, types, inputs = BaseUpdateInputs(str)
    for i = 1, #names do
        local name = names[i]
        registerDirectName(name, defaultForPort(name, types[i]), "input")
        setDirectValue(name, inputs[name]:get() or DirectWireValues[name], "input")

        if not DirectWireTriggers["__input:" .. name] then
            DirectWireTriggers["__input:" .. name] = true
            inputClk(name, function(value)
                setDirectValue(name, value, "input")
            end)
        end
    end
    return names, types, inputs
end

function networkInputs(namesOrAll, interval)
    local names = parseNames(namesOrAll)
    local count = 0
    for i = 1, #names do
        local name = names[i]
        local input = Inputs[name]
        if input and input.enableNetworking then
            DirectWireNetworked[name] = input:enableNetworking(true, interval)
            count = count + (DirectWireNetworked[name] and 1 or 0)
        end
    end
    return count
end

function networkInput(name, interval)
    return networkInputs({ name }, interval) == 1
end

function unnetworkInputs(namesOrAll)
    local names = parseNames(namesOrAll)
    local count = 0
    for i = 1, #names do
        local name = names[i]
        local input = Inputs[name]
        if input and input.enableNetworking then
            input:enableNetworking(false)
            DirectWireNetworked[name] = false
            count = count + 1
        end
    end
    return count
end

function addNetworkedVariable(name, interval, defaultValue)
    local entry = BaseAddNetworkedVariable(name, interval)
    if not entry then return nil end

    if defaultValue == nil then defaultValue = NetVarValues[name] end
    if defaultValue == nil then defaultValue = defaultForName(name) end
    registerDirectName(name, defaultValue, "netvar")
    DirectWireNetworked[name] = true

    if not DirectWireNetVarTriggers[name] then
        DirectWireNetVarTriggers[name] = NetVars[name]:addTrigger(function(value)
            DirectWireValues[name] = value
            runDirectTriggers(DirectWireTriggers, name, value)
        end) or true
    end

    if SERVER and defaultValue ~= nil and NetVarValues[name] == nil then
        BaseUpdateNetworkedVariable(name, defaultValue, interval)
    end
    return entry
end

function networkVariable(name, interval, defaultValue)
    return addNetworkedVariable(name, interval, defaultValue)
end

function setNetworkedVariableInterval(name, interval)
    return BaseSetNetworkedVariableInterval(name, interval)
end

function updateNetworkedVariable(name, value, interval)
    addNetworkedVariable(name, interval, value)
    setDirectValue(name, value, "netvar")
    return BaseUpdateNetworkedVariable(name, value, interval)
end

function updateNetworkedVariables(values, interval)
    if type(values) ~= "table" then return 0 end

    local count = 0
    for name, value in pairs(values) do
        if updateNetworkedVariable(name, value, interval) then count = count + 1 end
    end
    return count
end

function networkVariables(valuesOrNames, interval)
    if type(valuesOrNames) == "table" then
        local count = 0
        for name, value in pairs(valuesOrNames) do
            if type(name) == "string" then
                addNetworkedVariable(name, interval, value)
                count = count + 1
            elseif type(value) == "string" then
                addNetworkedVariable(value, interval)
                count = count + 1
            end
        end
        return count
    end

    local names = parseNames(valuesOrNames)
    local count = 0
    for i = 1, #names do
        addNetworkedVariable(names[i], interval)
        count = count + 1
    end
    return count
end

function directValue(name, defaultValue)
    return registerDirectName(name, defaultValue, "value")
end

function getDirectValue(name, fallback)
    local value = DirectWireValues[name]
    if value ~= nil then return value end
    return fallback
end

return base
