--@name Simple Server-Side Wire Helper
--@author Ally for Hire / Merydian9

--[[
Small utility that turns human-readable wire I/O strings into actual ports and
provides helper tables (`Inputs`, `Outputs`, `InputValues`) for quick access.

Optimized pass goals:
- lower parse overhead (single-pass parser, no token table)
- safer trigger execution (isolated callback failures)
- lightweight API additions while preserving compatibility

Function index:
- `updateInputs(str)`: creates or refreshes wire inputs from a compact port string.
- `updateOutputs(str)`: creates or refreshes wire outputs from a compact port string.
- `syncInputCommands(enabled)`: mirrors input values into globals matching their input names.
- `changed(value, keepChangedState, slot, epsilon)`: change detector with optional persistent dirty state.
- `resetChanged(slot)`: clears `changed(...)` state for one slot or all slots.
- `addInputTrigger(name, fn)`: registers a trigger callback on a named input.
- `safeWireCreate(entI, entO, inputName, outputName)`: protected `wire.create(...)`.
- `safeWireDelete(ent, inputName)`: protected `wire.delete(...)`.
- `minSendTime(...)`: estimates the minimum steady-state interval in seconds to keep sending the provided values indefinitely under the current send budget.
- `Inputs.<Name>:get()`: reads the cached input value.
- `Inputs.<Name>:isValid()`: true when that input is currently backed by a real wire/network update.
- `Inputs.<Name>:addTrigger(fn)`: registers a trigger callback on that input object on both realms.
- `Inputs.<Name>:removeTrigger(idx)`: removes a previously added trigger by index.
- `Inputs.<Name>:clearTriggers()`: removes all triggers for that input.
- `Outputs.<Name>:update(value)`: writes a cached output value to Wire.
- `InputValues.<Name>`: latest cached input payload for direct reads.
- `NetworkedVariables.<Name>:get()`: reads the latest networked value on either realm.
- `NetworkedVariables.<Name>:addTrigger(fn)`: registers a callback for that networked variable.
- `NetworkedVariables.<Name>:removeTrigger(idx)`: removes a previously added networked callback by index.
- `NetworkedVariables.<Name>:clearTriggers()`: removes all callbacks for that networked variable.
- `NetworkedVariables.<Name>:changed(keepChangedState, slot, epsilon)`: wraps `changed(...)` for that networked variable.
- `NetworkedVariables.<Name>:resetChanged(slot)`: wraps `resetChanged(...)` for that networked variable.
- `NetworkedVariables.<Name>:update(value)`: updates a custom networked variable on the server.
- `NetworkedVariables.<Name>:setInterval(interval)`: sets custom networked variable coalescing interval on the server.
- `NetworkedVariables.<Name>:setThrottleGroup(groupName, interval)`: puts the variable on a shared throttle.
- `getNetworkedVariableValue("Name", fallback)`: reads a custom networked variable with a nil fallback.
- `readNetworkedGunState("Gun1", stateTable)`: reads standard ACF gun state into a table.
- `readNetworkedPropArmorState("Prop1", stateTable)`: reads standard ACF prop armor state into a table.

Networking additions:
- `Inputs.<Name>:enableNetworking(bool, interval)`: enables delta networking for an input, with optional send coalescing interval in seconds.
- `addNetworkedVariable("Name", interval, throttleGroup)`: registers a custom named networked variable with optional coalescing interval.
- `addNetworkedGunState("Gun1", interval)`: registers standard ACF gun state variables.
- `addNetworkedPropArmorState("Prop1", interval)`: registers standard ACF prop armor state variables.
- `addNetworkedVariableTrigger(name, fn)`: registers a callback on a named networked variable.
- `setNetworkedVariableInterval("Name", interval)`: sets a custom networked variable coalescing interval.
- `setNetworkedVariableThrottleGroup("Name", groupName, interval)`: puts a variable on a shared throttle.
- `updateNetworkedVariable("Name", value, interval, throttleGroup)`: updates a custom networked variable.
- `updateNetworkedGunState("Gun1", inputOrGun, interval)`: updates standard ACF gun state variables.
- `updateNetworkedPropArmorState("Prop1", inputOrProp, interval)`: updates standard ACF prop armor state variables.
- `networkTargetSet(playerOrPlayers)`: sets broadcast, single-player, or multi-player network targets.
- `updateMaxSend(bitsPerMillisecond)`: overrides the automatic fair-share limiter.
- `networkEnableDebug(bool)`: enables chat/debug prints for buffer, throttle, and send-limit events.
- `NetworkedVariableValues.<Name>`: latest received or queued networked value.

Notes:
- First send transmits metadata plus value so the client can attach a read path.
- Later sends only transmit the variable id and the latest value.
- Tables are serialized recursively without `net.writeTable`.
- `ClientInitialized` sends the full current state without applying the local cap.
- On the client, `Inputs.<Name>` is created lazily so triggers can be registered before metadata arrives.
- Default bandwidth budgeting is derived from Starfall's actual per-player net burst,
  split across `NetworkBusyChipShareCount` busy chips by default.
]]--

local networkedVariableProto = {}
networkedVariableProto.__index = networkedVariableProto

function networkedVariableProto:get()
    return NetworkedVariableValues[self.name]
end

function networkedVariableProto:addTrigger(fn)
    if _type(fn) ~= "function" then return nil end

    local name = self.name
    local list = NetworkedVariableTriggers[name]
    if list == nil then
        list = {}
        NetworkedVariableTriggers[name] = list
    end

    local idx = #list + 1
    list[idx] = fn
    return idx
end

function networkedVariableProto:removeTrigger(idx)
    if _type(idx) ~= "number" then return false end

    local list = NetworkedVariableTriggers[self.name]
    if not list or not list[idx] then return false end

    _table_remove(list, idx)
    if #list == 0 then
        NetworkedVariableTriggers[self.name] = nil
    end

    return true
end

function networkedVariableProto:clearTriggers()
    NetworkedVariableTriggers[self.name] = nil
end

function networkedVariableProto:changed(keepChangedState, slot, epsilon)
    return changed(self:get(), keepChangedState, slot or ("__netvar:" .. self.name), epsilon)
end

function networkedVariableProto:resetChanged(slot)
    return resetChanged(slot or ("__netvar:" .. self.name))
end

function networkedVariableProto:update(value)
    if not SERVER or self.sourceCode ~= NET_SOURCE_CUSTOM then
        return false
    end

    return updateNetworkedVariable(self.name, value)
end

function networkedVariableProto:setInterval(interval)
    if not SERVER or self.sourceCode ~= NET_SOURCE_CUSTOM then
        return false
    end

    return setNetworkedVariableInterval(self.name, interval)
end

function networkedVariableProto:setThrottleGroup(groupName, interval)
    if not SERVER or self.sourceCode ~= NET_SOURCE_CUSTOM then
        return false
    end

    return setNetworkedVariableThrottleGroup(self.name, groupName, interval)
end

local function ensureNetworkedVariableObject(entry)
    if not entry then return nil end

    if getmetatable(entry) ~= networkedVariableProto then
        _setmetatable(entry, networkedVariableProto)
    end

    return entry
end

local function getOrCreateNetworkedVariableObject(name)
    local entry = _rawget(NetworkedVariables, name)
    if entry then
        return ensureNetworkedVariableObject(entry)
    end

    entry = ensureNetworkedVariableObject({
        name = name,
        enabled = false,
        sourceCode = nil
    })

    _rawset(NetworkedVariables, name, entry)
    return entry
end

local function normalizeNetworkedGunPrefix(prefix)
    if _type(prefix) == "number" then
        return "Gun" .. prefix
    end
    if _type(prefix) == "string" and prefix ~= "" then
        return prefix
    end
    return nil
end

local function normalizeNetworkedPropPrefix(prefix)
    if _type(prefix) == "number" then
        return "Prop" .. prefix
    end
    if _type(prefix) == "string" and prefix ~= "" then
        return prefix
    end
    return nil
end

function getNetworkedVariableValue(name, fallback)
    if _type(name) ~= "string" or name == "" then return fallback end

    local value = getOrCreateNetworkedVariableObject(name):get()
    if value == nil then
        return fallback
    end

    return value
end

function readNetworkedGunState(prefix, state)
    prefix = normalizeNetworkedGunPrefix(prefix)
    if not prefix then return state end

    state = state or {}
    state.Ready = getNetworkedVariableValue(prefix .. "Ready", false) == true
    state.AmmoType = getNetworkedVariableValue(prefix .. "AmmoType", state.AmmoType or "")
    state.AmmoCount = getNetworkedVariableValue(prefix .. "AmmoCount", state.AmmoCount or 0)
    state.AmmoCountMax = getNetworkedVariableValue(prefix .. "AmmoCountMax", state.AmmoCountMax or 0)

    return state
end

function readNetworkedPropArmorState(prefix, state)
    prefix = normalizeNetworkedPropPrefix(prefix)
    if not prefix then return state end

    state = state or {}
    state.Armor = getNetworkedVariableValue(prefix .. "Armor", state.Armor or 0)

    return state
end

local function fireNetworkedVariableTriggers(name, value, errPrefix)
    local triggers = NetworkedVariableTriggers[name]
    if not triggers then return end

    for i = 1, #triggers do
        local fn = triggers[i]
        if fn then
            local ok, msg = _xpcall(fn, function(err)
                return errPrefix .. _tostring(err)
            end, value)
            if not ok then _print(msg) end
        end
    end
end

local function isSequentialTable(tbl)
    local count = 0
    local maxKey = 0
    for key in _pairs(tbl) do
        if _type(key) ~= "number" or key < 1 or key % 1 ~= 0 then
            return false, 0
        end

        count = count + 1
        if key > maxKey then
            maxKey = key
        end
    end

    return maxKey == count, count
end

local function isInteger(value)
    return _type(value) == "number" and value == _math_floor(value)
end

local function isColorTable(value)
    if _type(value) ~= "table" then return false end
    if _type(value.r) ~= "number" or _type(value.g) ~= "number" or _type(value.b) ~= "number" then
        return false
    end

    return value.a == nil or _type(value.a) == "number"
end

local function normalizedType(value)
    local valueType = _type(value)

    if valueType == "Vector" then return "vector" end
    if valueType == "Angle" then return "angle" end
    if valueType == "Entity" then return "entity" end
    if valueType == "Player" then return "entity" end
    if valueType == "Vehicle" then return "entity" end
    if valueType == "Weapon" then return "entity" end
    if valueType == "Npc" then return "entity" end
    if valueType == "NPC" then return "entity" end
    if valueType == "VMatrix" then return "matrix" end

    return valueType
end

local function cloneValue(value)
    local valueType = _type(value)
    if valueType ~= "table" then
        return value
    end

    local clone = {}
    for key, innerValue in _pairs(value) do
        clone[key] = cloneValue(innerValue)
    end

    return clone
end

local function cloneValueIfTable(value)
    if _type(value) == "table" then
        return cloneValue(value)
    end

    return value
end

local function deepValuesEqual(a, b, epsilon)
    if a == b and normalizedType(a) ~= "table" then
        return true
    end

    local typeA = normalizedType(a)
    if typeA ~= normalizedType(b) then
        return false
    end

    if epsilon and typeA == "number" then
        return _math_abs(a - b) <= epsilon
    end

    if typeA == "vector" then
        return a.x == b.x and a.y == b.y and a.z == b.z
    end

    if typeA == "angle" then
        return a.pitch == b.pitch and a.yaw == b.yaw and a.roll == b.roll
    end

    if typeA ~= "table" then
        return a == b
    end

    local sequentialA, countA = isSequentialTable(a)
    local sequentialB, countB = isSequentialTable(b)
    if sequentialA ~= sequentialB or countA ~= countB then
        return false
    end

    if sequentialA then
        for i = 1, countA do
            if not deepValuesEqual(a[i], b[i], epsilon) then
                return false
            end
        end

        return true
    end

    for key, valueA in _pairs(a) do
        if not deepValuesEqual(valueA, b[key], epsilon) then
            return false
        end
    end

    for key in _pairs(b) do
        if a[key] == nil then
            return false
        end
    end

    return true
end

local function sortedMapKeys(tbl)
    local keys

    if _table_getKeys then
        keys = _table_getKeys(tbl)
    else
        keys = {}
        for key in _pairs(tbl) do
            keys[#keys + 1] = key
        end
    end

    _table_sort(keys, function(a, b)
        local typeA = _type(a)
        local typeB = _type(b)
        if typeA ~= typeB then
            return typeA < typeB
        end

        if typeA == "number" then
            return a < b
        end

        return _tostring(a) < _tostring(b)
    end)

    return keys
end

local function outputValuesEqual(a, b)
    if a == b and normalizedType(a) ~= "table" then
        return true
    end

    local typeA = normalizedType(a)
    if typeA ~= normalizedType(b) then
        return false
    end

    if typeA == "vector" then
        return a.x == b.x and a.y == b.y and a.z == b.z
    end

    if typeA == "angle" then
        return a.pitch == b.pitch and a.yaw == b.yaw and a.roll == b.roll
    end

    if typeA ~= "table" then
        return a == b
    end

    local sequentialA, countA = isSequentialTable(a)
    local sequentialB, countB = isSequentialTable(b)
    if sequentialA ~= sequentialB or countA ~= countB then
        return false
    end

    if sequentialA then
        for i = 1, countA do
            if not outputValuesEqual(a[i], b[i]) then
                return false
            end
        end

        return true
    end

    for key, valueA in _pairs(a) do
        if not outputValuesEqual(valueA, b[key]) then
            return false
        end
    end

    for key in _pairs(b) do
        if a[key] == nil then
            return false
        end
    end

    return true
end

local function buildValueDescriptor(value)
    local valueType = normalizedType(value)

    if valueType == "boolean" then
        return { code = TYPE_BOOL }
    end

    if valueType == "number" then
        if isInteger(value) then
            if value >= 0 and value <= 4294967295 then
                return { code = TYPE_UINT }
            end

            if value >= -2147483648 and value <= 2147483647 then
                return { code = TYPE_INT }
            end
        end

        if value >= -2147483648 and value <= 2147483647 then
            return { code = TYPE_FLOAT }
        end

        return { code = TYPE_DOUBLE }
    end

    if valueType == "string" then
        return { code = TYPE_STRING }
    end

    if valueType == "vector" then
        return { code = TYPE_VECTOR }
    end

    if valueType == "angle" then
        return { code = TYPE_ANGLE }
    end

    if valueType == "entity" then
        return { code = TYPE_ENTITY }
    end

    if valueType == "matrix" or valueType == "vmatrix" then
        return { code = TYPE_MATRIX }
    end

    if valueType == "table" then
        if isColorTable(value) then
            return { code = TYPE_COLOR }
        end

        local sequential, count = isSequentialTable(value)
        if sequential then
            local elements = {}
            for i = 1, count do
                local descriptor = buildValueDescriptor(value[i])
                if not descriptor then return nil end
                elements[i] = descriptor
            end

            return { code = TYPE_ARRAY, count = count, elements = elements }
        end

        local entries, keys = {}, sortedMapKeys(value)
        for i = 1, #keys do
            local key = keys[i]
            local keyDescriptor = buildValueDescriptor(key)
            local valueDescriptor = buildValueDescriptor(value[key])
            if not keyDescriptor or not valueDescriptor then return nil end

            entries[i] = {
                keyDescriptor = keyDescriptor,
                keyValue = cloneValue(key),
                valueDescriptor = valueDescriptor
            }
        end

        return { code = TYPE_MAP, count = #entries, entries = entries }
    end

    return nil
end

local function descriptorsEqual(a, b)
    if a == b then return true end
    if not a or not b or a.code ~= b.code then return false end

    if a.code == TYPE_ARRAY then
        if a.count ~= b.count then return false end
        for i = 1, a.count do
            if not descriptorsEqual(a.elements[i], b.elements[i]) then
                return false
            end
        end
        return true
    end

    if a.code == TYPE_MAP then
        if a.count ~= b.count then return false end
        for i = 1, a.count do
            local entryA = a.entries[i]
            local entryB = b.entries[i]
            if not entryB then return false end
            if not descriptorsEqual(entryA.keyDescriptor, entryB.keyDescriptor) then return false end
            if not descriptorsEqual(entryA.valueDescriptor, entryB.valueDescriptor) then return false end
            if not deepValuesEqual(entryA.keyValue, entryB.keyValue) then return false end
        end
        return true
    end

    return true
end

local function writeValueByDescriptor(value, descriptor)
    local code = descriptor.code

    if code == TYPE_BOOL then
        net.writeBool(value == true)
        return
    end

    if code == TYPE_INT then
        net.writeInt(value, 32)
        return
    end

    if code == TYPE_UINT then
        net.writeUInt(value, 32)
        return
    end

    if code == TYPE_FLOAT then
        net.writeFloat(value)
        return
    end

    if code == TYPE_DOUBLE then
        net.writeDouble(value)
        return
    end

    if code == TYPE_STRING then
        net.writeString(value)
        return
    end

    if code == TYPE_VECTOR then
        net.writeVector(value)
        return
    end

    if code == TYPE_ANGLE then
        net.writeAngle(value)
        return
    end

    if code == TYPE_ENTITY then
        if value ~= nil and isValid(value) then
            net.writeEntity(value)
        else
            net.writeEntity(chip())
        end
        return
    end

    if code == TYPE_COLOR then
        net.writeColor(value)
        return
    end

    if code == TYPE_MATRIX then
        net.writeMatrix(value)
        return
    end

    if code == TYPE_UINT64 then
        net.writeUInt64(value)
        return
    end

    if code == TYPE_ARRAY then
        for i = 1, descriptor.count do
            writeValueByDescriptor(value[i], descriptor.elements[i])
        end
        return
    end

    if code == TYPE_MAP then
        for i = 1, descriptor.count do
            local entry = descriptor.entries[i]
            writeValueByDescriptor(value[entry.keyValue], entry.valueDescriptor)
        end
    end
end

local function readValueByDescriptor(descriptor)
    local code = descriptor.code

    if code == TYPE_BOOL then
        return net.readBool()
    end

    if code == TYPE_INT then
        return net.readInt(32)
    end

    if code == TYPE_UINT then
        return net.readUInt(32)
    end

    if code == TYPE_FLOAT then
        return net.readFloat()
    end

    if code == TYPE_DOUBLE then
        return net.readDouble()
    end

    if code == TYPE_STRING then
        return net.readString()
    end

    if code == TYPE_VECTOR then
        return net.readVector()
    end

    if code == TYPE_ANGLE then
        return net.readAngle()
    end

    if code == TYPE_ENTITY then
        return net.readEntity()
    end

    if code == TYPE_COLOR then
        return net.readColor()
    end

    if code == TYPE_MATRIX then
        return net.readMatrix()
    end

    if code == TYPE_UINT64 then
        return net.readUInt64()
    end

    if code == TYPE_ARRAY then
        local array = {}
        for i = 1, descriptor.count do
            array[i] = readValueByDescriptor(descriptor.elements[i])
        end
        return array
    end

    if code == TYPE_MAP then
        local map = {}
        for i = 1, descriptor.count do
            local entry = descriptor.entries[i]
            map[cloneValue(entry.keyValue)] = readValueByDescriptor(entry.valueDescriptor)
        end
        return map
    end
end

local function writeDescriptor(descriptor)
    net.writeUInt(descriptor.code, NETVAR_TYPE_BITS)

    if descriptor.code == TYPE_ARRAY then
        net.writeUInt(descriptor.count, 16)
        for i = 1, descriptor.count do
            writeDescriptor(descriptor.elements[i])
        end
        return
    end

    if descriptor.code == TYPE_MAP then
        net.writeUInt(descriptor.count, 16)
        for i = 1, descriptor.count do
            local entry = descriptor.entries[i]
            writeDescriptor(entry.keyDescriptor)
            writeValueByDescriptor(entry.keyValue, entry.keyDescriptor)
            writeDescriptor(entry.valueDescriptor)
        end
    end
end

local function readDescriptor()
    local code = net.readUInt(NETVAR_TYPE_BITS)
    local descriptor = { code = code }

    if code == TYPE_ARRAY then
        local count = net.readUInt(16)
        descriptor.count = count
        descriptor.elements = {}
        for i = 1, count do
            descriptor.elements[i] = readDescriptor()
        end
        return descriptor
    end

    if code == TYPE_MAP then
        local count = net.readUInt(16)
        descriptor.count = count
        descriptor.entries = {}
        for i = 1, count do
            local keyDescriptor = readDescriptor()
            local keyValue = readValueByDescriptor(keyDescriptor)
            local valueDescriptor = readDescriptor()
            descriptor.entries[i] = {
                keyDescriptor = keyDescriptor,
                keyValue = keyValue,
                valueDescriptor = valueDescriptor
            }
        end
    end

    return descriptor
end

local function estimateDescriptorBits(descriptor)
    local total = NETVAR_TYPE_BITS

    if descriptor.code == TYPE_ARRAY then
        total = total + 16
        for i = 1, descriptor.count do
            total = total + estimateDescriptorBits(descriptor.elements[i])
        end
        return total
    end

    if descriptor.code == TYPE_MAP then
        total = total + 16
        for i = 1, descriptor.count do
            local entry = descriptor.entries[i]
            total = total + estimateDescriptorBits(entry.keyDescriptor)
            total = total + estimateValueBits(entry.keyValue, entry.keyDescriptor)
            total = total + estimateDescriptorBits(entry.valueDescriptor)
        end
    end

    return total
end

estimateValueBits = function(value, descriptor)
    local code = descriptor.code

    if code == TYPE_BOOL then return 1 end
    if code == TYPE_INT or code == TYPE_UINT or code == TYPE_FLOAT then return 32 end
    if code == TYPE_DOUBLE or code == TYPE_UINT64 then return 64 end
    if code == TYPE_STRING then return (#value + 1) * 8 end
    if code == TYPE_VECTOR then return 96 end
    if code == TYPE_ANGLE then return 96 end
    if code == TYPE_ENTITY then return 48 end
    if code == TYPE_COLOR then return 32 end
    if code == TYPE_MATRIX then return 512 end

    if code == TYPE_ARRAY then
        local total = 0
        for i = 1, descriptor.count do
            total = total + estimateValueBits(value[i], descriptor.elements[i])
        end
        return total
    end

    if code == TYPE_MAP then
        local total = 0
        for i = 1, descriptor.count do
            local entry = descriptor.entries[i]
            total = total + estimateValueBits(value[entry.keyValue], entry.valueDescriptor)
        end
        return total
    end

    return 32
end

local function getEffectiveSendBitsPerSecond()
    if MaxSendBitsPerMs ~= nil then
        return MaxSendBitsPerMs * 1000
    end

    local shareCount = _math_max(1, NetworkBusyChipShareCount)
    return NetworkBurstRegenBitsPerSecond / shareCount
end

local function estimateSteadyStateValueBits(value)
    local descriptor = buildValueDescriptor(value)
    if not descriptor then
        return nil
    end

    return NETVAR_ID_BITS + estimateValueBits(value, descriptor)
end

function minSendTime(...)
    local totalBits = 0
    local count = select("#", ...)

    for i = 1, count do
        local value = select(i, ...)
        local estimatedBits = estimateSteadyStateValueBits(value)
        if not estimatedBits then
            return nil
        end

        totalBits = totalBits + estimatedBits
    end

    if totalBits <= 0 then
        return 0
    end

    local bitsPerSecond = getEffectiveSendBitsPerSecond()
    if bitsPerSecond <= 0 then
        return nil
    end

    return totalBits / bitsPerSecond
end

function updateMaxSend(bitsPerMillisecond)
    if bitsPerMillisecond == nil or bitsPerMillisecond == false then
        MaxSendBitsPerMs = nil
        if SERVER then
            NetworkSendBudgetCap = _math_floor(NetworkBurstLimitBits / _math_max(1, NetworkBusyChipShareCount))
            NetworkSendBudget = NetworkSendBudgetCap
            LastBudgetUpdate = _timer_curtime()
        end
        return true
    end

    if _type(bitsPerMillisecond) ~= "number" or bitsPerMillisecond <= 0 then
        return false
    end

    MaxSendBitsPerMs = bitsPerMillisecond
    if SERVER then
        NetworkSendBudgetCap = _math_max(1, _math_floor(bitsPerMillisecond * 1000))
        NetworkSendBudget = NetworkSendBudgetCap
        LastBudgetUpdate = _timer_curtime()
    end
    return true
end

        if target == nil then

            return nil, false
        end

    local function registerNetworkedVariable(name, sourceCode)
        local existing = _rawget(NetworkedVariables, name)
        if existing then
            if existing.sourceCode ~= nil and existing.sourceCode ~= sourceCode then
                return nil

            existing.sourceCode = sourceCode
            return ensureNetworkedVariableObject(existing)
        end

        local entry = {
            id = allocateNetworkedVariableId(),
            name = name,
            sourceCode = sourceCode,
            enabled = sourceCode == NET_SOURCE_CUSTOM,
            sendInterval = 0,
            pendingFlushAt = nil,
            descriptor = nil,
            forceMeta = true,
            currentValue = nil,
            averageSendInterval = 0,
            sendCount = 0,
            lastSentAt = nil

        _rawset(NetworkedVariables, name, entry)
        NetworkedVariableLookup[entry.id] = entry
        NetworkedVariableIds[name] = entry.id
        networkDebug("registered netvar '" .. name .. "' (id " .. entry.id .. ")")
        return entry
    end

    local function getThrottleGroup(groupName, interval)

        if not groupName then return nil end

        local group = NetworkThrottleGroups[groupName]
        if not group then
            group = {
                name = groupName,
                sendInterval = 0,

    local function groupHasPendingEntry(groupName)

        for i = 1, #PendingNetworkOrder do
            local entry = NetworkedVariableLookup[PendingNetworkOrder[i]]
            if entry and entry.pendingValue ~= nil and entry.throttleGroup == groupName then
                return true

        end

        return false

    local function clearThrottleGroupIfIdle(groupName)
        if not groupName then return end

        local group = NetworkThrottleGroups[groupName]
        if group and not groupHasPendingEntry(groupName) then
            group.pendingFlushAt = nil
        end
    end

    local function getEntrySendInterval(entry)
        local group = entry.throttleGroup and NetworkThrottleGroups[entry.throttleGroup]
        if group then
            return group.sendInterval or 0
        end

        return entry.sendInterval or 0
    end

    local function getEntryPendingFlushAt(entry)
        local group = entry.throttleGroup and NetworkThrottleGroups[entry.throttleGroup]
        if group then
            return group.pendingFlushAt
        end

        return entry.pendingFlushAt
    end

    local function schedulePendingFlush(entry)
        local interval = getEntrySendInterval(entry)
        if interval <= 0 then
            entry.pendingFlushAt = nil
            return
        end

        local now = _timer_curtime()
        local group = entry.throttleGroup and NetworkThrottleGroups[entry.throttleGroup]
        if group then
            if group.pendingFlushAt == nil or group.pendingFlushAt <= now then
                group.pendingFlushAt = now + interval
            end
            entry.pendingFlushAt = nil
            return
        end

        entry.pendingFlushAt = now + interval
    end

    local function disableFlushHookIfIdle()
        if FlushHookActive and #PendingNetworkOrder == 0 then
            hook.remove("tick", NETVAR_PREFIX .. "_Flush")
            FlushHookActive = false
        end
    end

    local function enableFlushHook()
        if FlushHookActive then return end

        hook.add("tick", NETVAR_PREFIX .. "_Flush", function()
            if #PendingNetworkOrder == 0 then
                disableFlushHookIfIdle()
                return
            end

            local now = _timer_curtime()
            replenishNetworkBudget(now)

            if now - LastFlushAt < NetworkFlushInterval then
                return
            end

            LastFlushAt = now
            flushNetworkQueue(false)
            disableFlushHookIfIdle()
        end)

        FlushHookActive = true
    end

    local function queueNetworkedVariable(entry, value)
        if not entry or not entry.enabled or value == nil then return false end

        local descriptor = buildValueDescriptor(value)
        if not descriptor then
            networkDebug("unsupported network type for '" .. entry.name .. "'; update dropped")
            return false
        end

        local previousValue = entry.currentValue
        local compareValue = previousValue
        if entry.pendingValue ~= nil then
            compareValue = entry.pendingValue
        end
        local descriptorChanged = not entry.descriptor or not descriptorsEqual(entry.descriptor, descriptor)
        local valueChanged = compareValue == nil or not deepValuesEqual(compareValue, value)

        if not valueChanged and not descriptorChanged and not entry.forceMeta then
            return false
        end

        local storedValue = cloneValueIfTable(value)
        local wasPending = entry.pendingValue ~= nil
        entry.currentValue = storedValue
        NetworkedVariableValues[entry.name] = cloneValueIfTable(value)

        if descriptorChanged then
            entry.descriptor = descriptor
            entry.forceMeta = true
            networkDebug("descriptor updated for '" .. entry.name .. "'; next send will include metadata")
        end

        entry.pendingValue = storedValue
        if not wasPending then
            local sendInterval = getEntrySendInterval(entry)
            if sendInterval > 0 then
                schedulePendingFlush(entry)
                networkDebug("buffered '" .. entry.name .. "' for " .. _tostring(sendInterval) .. "s coalescing window")
            else
                entry.pendingFlushAt = nil
                networkDebug("queued '" .. entry.name .. "' for next network flush")
            end
        else
            networkDebug("coalesced newer value for '" .. entry.name .. "' while pending")
        end

        if not PendingNetworkLookup[entry.id] then
            PendingNetworkLookup[entry.id] = true
            PendingNetworkOrder[#PendingNetworkOrder + 1] = entry.id
            enableFlushHook()
        end

        if previousValue == nil or not deepValuesEqual(previousValue, value) then
            fireNetworkedVariableTriggers(entry.name, value, "[SS-SWH] networked trigger error: ")
        end

        return true
    end

    normalizeNetworkInterval = function(interval)
        if interval == nil or interval == false then
            return 0
        end

        if _type(interval) == "number" and interval > 0 then
            return interval
        end

        return nil
    end

    local function setNetworkedVariableIntervalInternal(entry, interval)
        if not entry then return false end

        local nextInterval = normalizeNetworkInterval(interval)
        if nextInterval == nil then return false end

        local group = entry.throttleGroup and NetworkThrottleGroups[entry.throttleGroup]
        if group then
            group.sendInterval = nextInterval
            if entry.pendingValue ~= nil then
                schedulePendingFlush(entry)
            end
            return true
        end

        if entry.sendInterval == nextInterval then
            return true
        end

        entry.sendInterval = nextInterval

        if entry.pendingValue ~= nil then
            schedulePendingFlush(entry)
        end

        return true
    end

    local function setNetworkedVariableThrottleGroupInternal(entry, groupName, interval)
        if not entry then return false end

        groupName = normalizeThrottleGroupName(groupName)
        if groupName == false then return false end

        if not groupName then
            local oldGroup = entry.throttleGroup
            entry.throttleGroup = nil
            if interval ~= nil and setNetworkedVariableIntervalInternal(entry, interval) == false then
                entry.throttleGroup = oldGroup
                return false
            end
            clearThrottleGroupIfIdle(oldGroup)
            return true
        end

        local group = getThrottleGroup(groupName, interval)
        if not group then return false end

        local oldGroup = entry.throttleGroup
        entry.throttleGroup = groupName
        entry.pendingFlushAt = nil
        if entry.pendingValue ~= nil then
            schedulePendingFlush(entry)
        end
        clearThrottleGroupIfIdle(oldGroup)

        return true
    end

    local function sendNetworkedMeta(target, entry, value)
        if not entry.descriptor then return false end

        net.start(NETVAR_META)
        net.writeUInt(entry.id, NETVAR_ID_BITS)
        net.writeUInt(entry.sourceCode, 1)
        net.writeString(entry.name)
        writeDescriptor(entry.descriptor)
        writeValueByDescriptor(value, entry.descriptor)
        net.send(target)

        entry.forceMeta = false
        return true
    end

    local function sendNetworkedMetaSnapshot(target, entry, value, descriptor)
        if not descriptor then return false end

        net.start(NETVAR_META)
        net.writeUInt(entry.id, NETVAR_ID_BITS)
        net.writeUInt(entry.sourceCode, 1)
        net.writeString(entry.name)
        writeDescriptor(descriptor)
        writeValueByDescriptor(value, descriptor)
        net.send(target)
        return true
    end

    local function sendNetworkedUpdate(target, entry, value)
        if not entry.descriptor then return false end

        net.start(NETVAR_UPDATE)
        net.writeUInt(entry.id, NETVAR_ID_BITS)
        writeValueByDescriptor(value, entry.descriptor)
        net.send(target)
        return true
    end

    local function updateSendStats(entry, now)
        if entry.lastSentAt then
            local interval = now - entry.lastSentAt
            local count = entry.sendCount
            entry.averageSendInterval = ((entry.averageSendInterval * count) + interval) / (count + 1)
            entry.sendCount = count + 1
        else
            entry.averageSendInterval = 0
            entry.sendCount = 1
        end

        entry.lastSentAt = now
    end

    local function removeNetworkedVariableInternal(name)
        local entry = _rawget(NetworkedVariables, name)
        if not entry then return false end

        removePendingNetworkedVariable(entry.id)
        NetworkedVariableLookup[entry.id] = nil
        NetworkedVariableIds[name] = nil
        NetworkedVariableValues[name] = nil
        _rawset(NetworkedVariables, name, nil)
        clearThrottleGroupIfIdle(entry.throttleGroup)
        disableFlushHookIfIdle()

        local target, hasRecipients = getActiveSendTarget()
        if hasRecipients then
            net.start(NETVAR_REMOVE)
            net.writeUInt(entry.id, NETVAR_ID_BITS)
            net.send(target)
        end

        networkDebug("removed netvar '" .. name .. "'")

        return true
    end

    replenishNetworkBudget = function(now)
        local elapsed = now - LastBudgetUpdate
        if elapsed <= 0 then return end

        local bitsPerMs
        local budgetCap

        if MaxSendBitsPerMs ~= nil then
            bitsPerMs = MaxSendBitsPerMs
            budgetCap = NetworkSendBudgetCap or _math_max(1, _math_floor(bitsPerMs * 1000))
        else
            local shareCount = _math_max(1, NetworkBusyChipShareCount)
            bitsPerMs = (NetworkBurstRegenBitsPerSecond / 1000) / shareCount
            budgetCap = _math_floor(NetworkBurstLimitBits / shareCount)
        end

        NetworkSendBudget = _math_min(budgetCap, NetworkSendBudget + (elapsed * 1000 * bitsPerMs))
        LastBudgetUpdate = now
    end

    local function estimateUpdateBits(entry, value)
        return NETVAR_ID_BITS + estimateValueBits(value, entry.descriptor)
    end

    local function estimateMetaBits(entry, value)
        return NETVAR_ID_BITS + 1 + ((#entry.name + 1) * 8) + estimateDescriptorBits(entry.descriptor) + estimateValueBits(value, entry.descriptor)
    end

    local function resolveNetworkedEntryValue(entry)
        if entry.pendingValue ~= nil then return entry.pendingValue end
        if entry.currentValue ~= nil then return entry.currentValue end

        if entry.sourceCode == NET_SOURCE_INPUT then
            if InputValidity[entry.name] ~= true then return nil end
            return InputValues[entry.name]
        end

        if NetworkedVariableValues[entry.name] ~= nil then
            return NetworkedVariableValues[entry.name]
        end

        return entry.currentValue
    end

    local function sendNetworkedEntrySnapshot(ply, entry)
        if not entry.enabled then return false end

        local value = resolveNetworkedEntryValue(entry)
        if value == nil then return false end

        local descriptor = buildValueDescriptor(value)
        if not descriptor then return false end

        if not entry.descriptor or not descriptorsEqual(entry.descriptor, descriptor) then
            entry.descriptor = descriptor
            entry.forceMeta = true
        end

        return sendNetworkedMetaSnapshot(ply, entry, value, descriptor)
    end

    flushNetworkQueue = function(forceAll)
        local target, hasRecipients = getActiveSendTarget()
        if not hasRecipients then return end

        local now = _timer_curtime()
        replenishNetworkBudget(now)

        local availableBits
        if forceAll then
            availableBits = 2147483647
        else
            availableBits = NetworkSendBudget
        end
        local i = 1

        while i <= #PendingNetworkOrder do
            local id = PendingNetworkOrder[i]
            local entry = NetworkedVariableLookup[id]

            if not entry or entry.pendingValue == nil then
                PendingNetworkLookup[id] = nil
                _table_remove(PendingNetworkOrder, i)
            else
                local value = entry.pendingValue
                local pendingFlushAt = getEntryPendingFlushAt(entry)
                if not forceAll and pendingFlushAt and now < pendingFlushAt then
                    if not entry.debugIntervalPending then
                        entry.debugIntervalPending = true
                        networkDebug("holding '" .. entry.name .. "' until coalescing window closes")
                    end
                    i = i + 1
                else
                local estimatedBits = entry.forceMeta and estimateMetaBits(entry, value) or estimateUpdateBits(entry, value)

                if not forceAll and estimatedBits > availableBits then
                    if not entry.debugBudgetPending then
                        entry.debugBudgetPending = true
                        networkDebug("budget deferred '" .. entry.name .. "' to a later flush")
                    end
                    i = i + 1
                else
                    if entry.forceMeta then
                        sendNetworkedMeta(target, entry, value)
                        networkDebug("sent metadata sync for '" .. entry.name .. "'")
                    else
                        sendNetworkedUpdate(target, entry, value)
                        networkDebug("sent update for '" .. entry.name .. "'")
                    end

                    updateSendStats(entry, now)
                    entry.pendingValue = nil
                    entry.pendingFlushAt = nil
                    entry.debugBudgetPending = false
                    entry.debugIntervalPending = false
                    PendingNetworkLookup[id] = nil
                    _table_remove(PendingNetworkOrder, i)
                    clearThrottleGroupIfIdle(entry.throttleGroup)

                    if not forceAll and MaxSendBitsPerMs ~= nil then
                        NetworkSendBudget = _math_max(0, NetworkSendBudget - estimatedBits)
                    end

                    if not forceAll then
                        availableBits = _math_max(0, availableBits - estimatedBits)
                    end
                end
                end
            end
        end
    end

    local function queueInputUpdate(name, value)
        local entry = _rawget(NetworkedVariables, name)
        if entry and entry.sourceCode == NET_SOURCE_INPUT and entry.enabled then
            queueNetworkedVariable(entry, value)
        end
    end

    local function clearNetworkedInputValue(name)
        local entry = _rawget(NetworkedVariables, name)
        if not entry or entry.sourceCode ~= NET_SOURCE_INPUT or not entry.enabled then return end

        entry.currentValue = nil
        entry.pendingValue = nil
        entry.pendingFlushAt = nil
        entry.forceMeta = true
        NetworkedVariableValues[name] = nil
        removePendingNetworkedVariable(entry.id)
        clearThrottleGroupIfIdle(entry.throttleGroup)
        disableFlushHookIfIdle()

        local target, hasRecipients = getActiveSendTarget()
        if not hasRecipients then return end

        net.start(NETVAR_REMOVE)
        net.writeUInt(entry.id, NETVAR_ID_BITS)
        net.send(target)
    end

    local inputProto = {}
    inputProto.__index = inputProto

    function inputProto:get()
        return InputValues[self.name]
    end

    function inputProto:isValid()
        return InputValidity[self.name] == true
    end

    function inputProto:addTrigger(fn)
        if _type(fn) ~= "function" then return nil end

        local name = self.name
        local list = InputTriggers[name]
        if list == nil then
            list = {}
            InputTriggers[name] = list
        end

        local idx = #list + 1
        list[idx] = fn
        return idx
    end

    function inputProto:removeTrigger(idx)
        if _type(idx) ~= "number" then return false end

        local list = InputTriggers[self.name]
        if not list or not list[idx] then return false end

        _table_remove(list, idx)
        if #list == 0 then
            InputTriggers[self.name] = nil
        end

        return true
    end

    function inputProto:clearTriggers()
        InputTriggers[self.name] = nil
    end

    function inputProto:enableNetworking(enabled, interval)
        local entry = registerNetworkedVariable(self.name, NET_SOURCE_INPUT)
        if not entry then return false end

        local nextInterval = 0
        if interval == nil or interval == false then
            nextInterval = 0
        elseif _type(interval) == "number" and interval > 0 then
            nextInterval = interval
        else
            return false
        end

        local nextEnabled = enabled ~= false
        local wasEnabled = entry.enabled == true
        local intervalChanged = entry.sendInterval ~= nextInterval

        entry.enabled = nextEnabled
        entry.sendInterval = nextInterval
        self.networked = entry.enabled

        if entry.enabled then
            if not wasEnabled then
                entry.forceMeta = true
            end

            local currentValue = InputValues[self.name]
            if currentValue ~= nil and InputValidity[self.name] == true and (not wasEnabled or intervalChanged) then
                queueNetworkedVariable(entry, currentValue)
            elseif entry.pendingValue ~= nil and entry.sendInterval > 0 then
                entry.pendingFlushAt = _timer_curtime() + entry.sendInterval
            end
            networkDebug("enabled networking for input '" .. self.name .. "' with interval " .. _tostring(entry.sendInterval or 0))
        else
            removePendingNetworkedVariable(entry.id)
            entry.pendingValue = nil
            entry.pendingFlushAt = nil
            entry.debugBudgetPending = false
            entry.debugIntervalPending = false
            disableFlushHookIfIdle()
            networkDebug("disabled networking for input '" .. self.name .. "'")
        end

        return entry.enabled
    end

    local outputProto = {}
    outputProto.__index = outputProto

    function outputProto:update(value)
        local name = self.name
        local lastValue = LastOutputValues[name]
        if lastValue ~= nil and outputValuesEqual(value, lastValue) then
            return false
        end

        LastOutputValues[name] = cloneValue(value)
        _wire_ports[name] = value
        return true
    end

    local function splitPortString(portString)
        local names, types = {}, {}

        if _type(portString) ~= "string" then
            return names, types
        end

        local n = 0
        for token in _string_gmatch(portString, "[^,%s]+") do
            local name, portType = _string_match(token, "^([^:]+):?(.*)$")
            if name and name ~= "" then
                n = n + 1
                names[n] = name
                types[n] = (portType ~= nil and portType ~= "") and portType or "NORMAL"
            end
        end

        return names, types
    end

    local function onTriggerError(err)
        return "[SS-SWH] trigger error: " .. _tostring(err)
    end

    local function syncInputGlobal(name, value)
        if not SyncInputCommandsEnabled then return end

        if not SyncedInputCommandNames[name] then
            SyncedInputCommandNames[name] = true
            SyncedInputCommandBackup[name] = _G[name]
        end

        _G[name] = value
    end

    local function unsyncInputGlobal(name)
        if not SyncedInputCommandNames[name] then return end

        _G[name] = SyncedInputCommandBackup[name]
        SyncedInputCommandBackup[name] = nil
        SyncedInputCommandNames[name] = nil
    end

    local function doWireInput(name, value)
        if value == nil then return end

        local previousValue = InputValues[name]
        local previousValid = InputValidity[name] == true
        local inputValid = _wire.self():isWired(name)
        local changedValue = not deepValuesEqual(previousValue, value)
        local storedValue = cloneValueIfTable(value)

        InputValues[name] = storedValue
        InputValidity[name] = inputValid
        syncInputGlobal(name, storedValue)

        if inputValid and (changedValue or not previousValid) then
            queueInputUpdate(name, value)
        elseif not inputValid and previousValid then
            clearNetworkedInputValue(name)
        end

        local triggers = InputTriggers[name]
        if not triggers then return end

        for i = 1, #triggers do
            local fn = triggers[i]
            if fn then
                local ok, msg = _xpcall(fn, onTriggerError, value)
                if not ok then _print(msg) end
            end
        end
    end

    local function samePortLayout(namesA, typesA, namesB, typesB)
        local len = #namesA
        if len ~= #namesB or len ~= #typesA or len ~= #typesB then return false end

        for i = 1, len do
            if namesA[i] ~= namesB[i] or typesA[i] ~= typesB[i] then
                return false
            end
        end

        return true
    end

    local function copyPortLayout(srcNames, srcTypes, dstNames, dstTypes)
        for i = 1, #dstNames do
            dstNames[i] = nil
            dstTypes[i] = nil
        end

        for i = 1, #srcNames do
            dstNames[i] = srcNames[i]
            dstTypes[i] = srcTypes[i]
        end
    end

    local function clearSet(tbl)
        for key in _pairs(tbl) do
            tbl[key] = nil
        end
    end

    local function resolveChangedSlot(explicitSlot)
        if explicitSlot ~= nil then return explicitSlot end

        if _debug and _debug.getinfo then
            local info = _debug.getinfo(3, "Sl")
            if info then
                return (_tostring(info.short_src) .. ":" .. _tostring(info.currentline))
            end
        end

        return "__default_slot"
    end

    local function valuesEqual(a, b, epsilon)
        if a == b then return true end

        if epsilon and _type(a) == "number" and _type(b) == "number" then
            return _math_abs(a - b) <= epsilon
        end

        return false
    end

    function syncInputCommands(enabled)
        local turnOn = enabled == true

        if turnOn == SyncInputCommandsEnabled then
            return SyncInputCommandsEnabled
        end

        SyncInputCommandsEnabled = turnOn

        if SyncInputCommandsEnabled then
            for name, value in _pairs(InputValues) do
                syncInputGlobal(name, value)
            end
        else
            for name in _pairs(SyncedInputCommandNames) do
                unsyncInputGlobal(name)
            end
        end

        return SyncInputCommandsEnabled
    end

    function changed(value, keepChangedState, slot, epsilon)
        local key = resolveChangedSlot(slot)
        local state = ChangedState[key]

        if not state then
            ChangedState[key] = { last = value, pending = value, dirty = false }
            return true
        end

        if not valuesEqual(value, state.last, epsilon) then
            state.pending = value
            state.dirty = true
        end

        if not state.dirty then
            return false
        end

        if not keepChangedState then
            state.last = state.pending
            state.dirty = false
        end

        return true
    end

    function resetChanged(slot)
        if slot == nil then
            for _, state in _pairs(ChangedState) do
                if state.dirty then
                    state.last = state.pending
                    state.dirty = false
                end
            end
            return true
        end

        local key = resolveChangedSlot(slot)
        local state = ChangedState[key]
        if not state then return false end

        if state.dirty then
            state.last = state.pending
            state.dirty = false
        end

        return true
    end

    function updateInputs(str)
        local names, types = splitPortString(str)

        local changedLayout = not samePortLayout(names, types, LastInputNames, LastInputTypes)
        if not changedLayout then
            return names, types, Inputs, false
        end

        local keep = ReusableInputKeep
        clearSet(keep)
        for i = 1, #names do
            keep[names[i]] = true
        end

        for key in _pairs(InputTriggers) do
            if not keep[key] then InputTriggers[key] = nil end
        end

        for key in _pairs(InputValues) do
            if not keep[key] then
                InputValues[key] = nil
                InputValidity[key] = nil
                unsyncInputGlobal(key)
            end
        end

        for key, entry in _pairs(NetworkedVariables) do
            if entry.sourceCode == NET_SOURCE_INPUT and not keep[key] then
                removeNetworkedVariableInternal(key)
            end
        end

        for key in _pairs(Inputs) do
            Inputs[key] = nil
        end

        for i = 1, #names do
            local name = names[i]
            local existing = _rawget(NetworkedVariables, name)
            Inputs[name] = _setmetatable({
                name = name,
                networked = existing and existing.sourceCode == NET_SOURCE_INPUT and existing.enabled or false
            }, inputProto)
        end

        copyPortLayout(names, types, LastInputNames, LastInputTypes)
        _wire.adjustInputs(names, types)
        return names, types, Inputs, true
    end

    function updateOutputs(str)
        local names, types = splitPortString(str)

        local changedLayout = not samePortLayout(names, types, LastOutputNames, LastOutputTypes)
        if not changedLayout then
            return names, types, Outputs, false
        end

        for key in _pairs(Outputs) do
            Outputs[key] = nil
        end

        for key in _pairs(LastOutputValues) do
            LastOutputValues[key] = nil
        end

        for i = 1, #names do
            local name = names[i]
            Outputs[name] = _setmetatable({ name = name }, outputProto)
        end

        copyPortLayout(names, types, LastOutputNames, LastOutputTypes)
        _wire.adjustOutputs(names, types)
        return names, types, Outputs, true
    end

    function addInputTrigger(name, fn)
        local inputObj = Inputs[name]
        if not inputObj then return nil end
        return inputObj:addTrigger(fn)
    end

    function addNetworkedVariable(name, interval, throttleGroup)
        if _type(name) ~= "string" or name == "" then return nil end

        local entry = registerNetworkedVariable(name, NET_SOURCE_CUSTOM)
        if not entry then return nil end

        local wasEnabled = entry.enabled == true
        entry.enabled = true
        if not wasEnabled then
            entry.forceMeta = true
        end
        if interval ~= nil and setNetworkedVariableIntervalInternal(entry, interval) == false then
            return nil
        end
        if throttleGroup ~= nil and setNetworkedVariableThrottleGroupInternal(entry, throttleGroup, interval) == false then
            return nil
        end

        return entry
    end

    function addNetworkedGunState(prefix, interval, throttleGroup)
        prefix = normalizeNetworkedGunPrefix(prefix)
        if not prefix then return false end

        addNetworkedVariable(prefix .. "Ready", interval, throttleGroup)
        addNetworkedVariable(prefix .. "AmmoType", interval, throttleGroup)
        addNetworkedVariable(prefix .. "AmmoCount", interval, throttleGroup)
        addNetworkedVariable(prefix .. "AmmoCountMax", interval, throttleGroup)
        return true
    end

    function addNetworkedPropArmorState(prefix, interval, throttleGroup)
        prefix = normalizeNetworkedPropPrefix(prefix)
        if not prefix then return false end

        addNetworkedVariable(prefix .. "Armor", interval, throttleGroup)
        return true
    end

    function addNetworkedVariableTrigger(name, fn)
        if _type(name) ~= "string" or name == "" then return nil end
        local entry = getOrCreateNetworkedVariableObject(name)
        return entry:addTrigger(fn)
    end

    function setNetworkedVariableInterval(name, interval)
        if _type(name) ~= "string" or name == "" then return false end

        local entry = addNetworkedVariable(name)
        if not entry then return false end

        return setNetworkedVariableIntervalInternal(entry, interval)
    end

    function setNetworkedVariableThrottleGroup(name, groupName, interval)
        if _type(name) ~= "string" or name == "" then return false end

        local entry = addNetworkedVariable(name)
        if not entry then return false end

        return setNetworkedVariableThrottleGroupInternal(entry, groupName, interval)
    end

    function updateNetworkedVariable(name, value, interval, throttleGroup)
        if _type(name) ~= "string" or name == "" or value == nil then
            return false
        end

        local entry = addNetworkedVariable(name, interval, throttleGroup)
        if not entry then
            return false
        end

        return queueNetworkedVariable(entry, value)
    end

    function updateNetworkedGunState(prefix, inputOrGun, interval, throttleGroup)
        prefix = normalizeNetworkedGunPrefix(prefix)
        if not prefix then return false end

        local gun
        if _type(inputOrGun) == "table" and _type(inputOrGun.isValid) == "function" and
            _type(inputOrGun.get) == "function" then
            if inputOrGun:isValid() then
                gun = inputOrGun:get()
            end
        elseif inputOrGun ~= nil and isValid(inputOrGun) then
            gun = inputOrGun
        end

        if gun ~= nil and isValid(gun) then
            updateNetworkedVariable(prefix .. "Ready", gun:acfReady(), interval, throttleGroup)
            updateNetworkedVariable(prefix .. "AmmoType", gun:acfAmmoType() or "", interval, throttleGroup)
            updateNetworkedVariable(prefix .. "AmmoCount", gun:acfMagRounds() or 0, interval, throttleGroup)
            updateNetworkedVariable(prefix .. "AmmoCountMax", gun:acfMagSize() or 0, interval, throttleGroup)
            return true
        end

        updateNetworkedVariable(prefix .. "Ready", false, interval, throttleGroup)
        updateNetworkedVariable(prefix .. "AmmoType", "", interval, throttleGroup)
        updateNetworkedVariable(prefix .. "AmmoCount", 0, interval, throttleGroup)
        updateNetworkedVariable(prefix .. "AmmoCountMax", 0, interval, throttleGroup)
        return false
    end

    function updateNetworkedPropArmorState(prefix, inputOrProp, interval, throttleGroup)
        prefix = normalizeNetworkedPropPrefix(prefix)
        if not prefix then return false end

        local prop
        if _type(inputOrProp) == "table" and _type(inputOrProp.isValid) == "function" and
            _type(inputOrProp.get) == "function" then
            if inputOrProp:isValid() then
                prop = inputOrProp:get()
            end
        elseif inputOrProp ~= nil and isValid(inputOrProp) then
            prop = inputOrProp
        end

        if prop ~= nil and isValid(prop) then
            updateNetworkedVariable(prefix .. "Armor", prop:acfPropArmor() or 0, interval, throttleGroup)
            return true
        end

        updateNetworkedVariable(prefix .. "Armor", 0, interval, throttleGroup)
        return false
    end

    function safeWireCreate(entI, entO, inputName, outputName)
        local ok = pcall(_wire.create, entI, entO, inputName, outputName)
        return ok
    end

    function safeWireDelete(ent, inputName)
        local ok = pcall(_wire.delete, ent, inputName)
        return ok
    end

    _G.safeWireCreate = safeWireCreate
    _G.safeWireDelete = safeWireDelete

    hook.add("input", NETVAR_PREFIX .. "_Input", doWireInput)

    hook.add("ClientInitialized", NETVAR_PREFIX .. "_Init", function(ply)
        addInitializedNetworkTarget(ply)
        if not targetIncludesPlayer(NetworkTargets, ply) then return end

        for _, entry in _pairs(NetworkedVariables) do
            if sendNetworkedEntrySnapshot(ply, entry) then
                updateSendStats(entry, _timer_curtime())
            end
        end
    end)

if CLIENT then
    local clientInputProto = {}
    clientInputProto.__index = clientInputProto

    function clientInputProto:get()
        return InputValues[self.name]
    end

    function clientInputProto:isValid()
        return InputValidity[self.name] == true
    end

    function clientInputProto:addTrigger(fn)
        if _type(fn) ~= "function" then return nil end

        local name = self.name
        local list = InputTriggers[name]
        if list == nil then
            list = {}
            InputTriggers[name] = list
        end

        local idx = #list + 1
        list[idx] = fn
        return idx
    end

    function clientInputProto:removeTrigger(idx)
        if _type(idx) ~= "number" then return false end

        local list = InputTriggers[self.name]
        if not list or not list[idx] then return false end

        _table_remove(list, idx)
        if #list == 0 then
            InputTriggers[self.name] = nil
        end

        return true
    end

    function clientInputProto:clearTriggers()
        InputTriggers[self.name] = nil
    end

    local function ensureClientInputObject(name)
        local input = _rawget(Inputs, name)
        if not input then
            input = _setmetatable({ name = name, networked = true }, clientInputProto)
            _rawset(Inputs, name, input)
        end

        return input
    end

    if getmetatable(Inputs) == nil then
        _setmetatable(Inputs, {
            __index = function(tbl, key)
                if _type(key) ~= "string" or key == "" then
                    return nil
                end

                return ensureClientInputObject(key)
            end
        })
    end

    local function fireClientInputTriggers(name, value)
        local triggers = InputTriggers[name]
        if not triggers then return end

        for i = 1, #triggers do
            local fn = triggers[i]
            if fn then
                local ok, msg = _xpcall(fn, function(err)
                    return "[SS-SWH] client trigger error: " .. _tostring(err)
                end, value)
                if not ok then _print(msg) end
            end
        end
    end

    local function applyNetworkedValue(entry, value)
        local previousValue = entry.currentValue
        local storedValue = cloneValueIfTable(value)
        NetworkedVariableValues[entry.name] = cloneValueIfTable(value)
        entry.currentValue = storedValue

        if entry.sourceCode == NET_SOURCE_INPUT then
            ensureClientInputObject(entry.name)
            InputValues[entry.name] = cloneValueIfTable(value)
            InputValidity[entry.name] = true
            fireClientInputTriggers(entry.name, value)
        end

        if not deepValuesEqual(previousValue, value) then
            fireNetworkedVariableTriggers(entry.name, value, "[SS-SWH] client networked trigger error: ")
        end
    end

    net.receive(NETVAR_META, function()
        local id = net.readUInt(NETVAR_ID_BITS)
        local sourceCode = net.readUInt(1)
        local name = net.readString()
        local descriptor = readDescriptor()
        local value = readValueByDescriptor(descriptor)

        local entry = getOrCreateNetworkedVariableObject(name)

        entry.id = id
        if entry.previousId and entry.previousId ~= id then
            NetworkedVariableLookup[entry.previousId] = nil
        end
        entry.sourceCode = sourceCode
        entry.descriptor = descriptor
        entry.enabled = true
        entry.previousId = id
        entry = ensureNetworkedVariableObject(entry)

        _rawset(NetworkedVariables, name, entry)
        NetworkedVariableLookup[id] = entry
        NetworkedVariableIds[name] = id

        applyNetworkedValue(entry, value)
    end)

    net.receive(NETVAR_UPDATE, function()
        local id = net.readUInt(NETVAR_ID_BITS)
        local entry = NetworkedVariableLookup[id]
        if not entry or not entry.descriptor then return end

        local value = readValueByDescriptor(entry.descriptor)
        applyNetworkedValue(entry, value)
    end)

    net.receive(NETVAR_REMOVE, function()
        local id = net.readUInt(NETVAR_ID_BITS)
        local entry = NetworkedVariableLookup[id]
        if not entry then return end
        local sourceCode = entry.sourceCode

        NetworkedVariableLookup[id] = nil
        NetworkedVariableIds[entry.name] = nil
        NetworkedVariableValues[entry.name] = nil
        entry.id = nil
        entry.currentValue = nil
        entry.pendingValue = nil
        entry.descriptor = nil
        entry.enabled = false
        entry.sourceCode = nil
        entry.previousId = nil

        if sourceCode == NET_SOURCE_INPUT then
            InputValues[entry.name] = nil
            InputValidity[entry.name] = false
            Inputs[entry.name] = nil
        end
    end)

    function addNetworkedVariableTrigger(name, fn)
        if _type(name) ~= "string" or name == "" then return nil end
        local entry = getOrCreateNetworkedVariableObject(name)
        return entry:addTrigger(fn)
    end

    net.receive(NETVAR_DEBUG, function()
        _print(net.readString())
    end)
end
