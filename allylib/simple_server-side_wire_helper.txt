--@name Simple Server-Side Wire Helper
--@author Ally for Hire / Merydian9

--[[
Small Starfall helper for common chip plumbing:
- readable wire input/output declarations
- cached input values and input triggers
- cached output writes
- simple latest-value NetVars replication

Networking is intentionally one path:
server code registers or updates a NetVar, the helper keeps only the newest value,
and a capped flush sends sequential typed records to the selected clients.

Function index:
- `updateInputs(str)`: creates wire inputs and `Inputs.<Name>` objects.
- `updateOutputs(str)`: creates wire outputs and `Outputs.<Name>` objects.
- `syncInputCommands(enabled)`: mirrors input values into globals matching their names.
- `changed(value, keepChangedState, slot, epsilon)`: cached change detector.
- `resetChanged(slot)`: clears one change slot, or all slots when omitted.
- `addInputTrigger(name, fn)`: adds an input trigger without touching `Inputs` directly.
- `safeWireCreate(entI, entO, inputName, outputName)`: protected `wire.create`.
- `safeWireDelete(ent, inputName)`: protected `wire.delete`.
- `addDriverDetection(vehicleOrInput, fn, interval)`: polls a chair/entity input for its driver.
- `minSendTime(...)`: returns the helper's current send interval estimate.
- `networkTargetSet(target)`: sets explicit NetVar recipients; nil means `net.send(nil)`.
- `networkEnableDebug(enabled)`: toggles helper debug state.
- `networkStats()`: returns simple replication counters.
- `updateMaxSend(...)`: compatibility no-op in the simplified helper.
- `Inputs.<Name>:get()`: reads cached input, falling back to `wire.ports`.
- `Inputs.<Name>:isValid()`: checks cached/wired input validity.
- `Inputs.<Name>:addTrigger(fn)`: registers an input callback.
- `Inputs.<Name>:enableNetworking(enabled, interval)`: mirrors that input through NetVars.
- `Outputs.<Name>:update(value)`: cached output write.
- `NetVars.<Name>:get(fallback)`: reads replicated value with a valid default.
- `NetVars.<Name>:addTrigger(fn)`: runs when a replicated value changes.
- `NetVars.<Name>:update(value)`: updates a custom NetVar on the server.
- `NetVars.<Name>:setInterval(interval)`: caps that NetVar's send interval.
- `addNetworkedVariable(name, interval)`: registers a custom NetVar.
- `updateNetworkedVariable(name, value, interval)`: queues latest value for replication.
- `State.<Prefix>`: reads a replicated gun or prop armor state table with valid defaults.
- `addNetworkedGunState(prefix, interval)`: registers common ACF gun state vars.
- `updateNetworkedGunState(prefix, inputOrGun, interval)`: queues common ACF gun state.
- `addNetworkedPropArmorState(prefix, interval)`: registers common prop armor state.
- `updateNetworkedPropArmorState(prefix, inputOrProp, interval)`: queues prop armor state.
]]--

local function isVector(value)
    return type(value) == "Vector" or (type(value) == "table" and type(value.x) == "number" and type(value.y) == "number" and type(value.z) == "number")
end

local function isAngle(value)
    return type(value) == "Angle" or (type(value) == "table" and (
        type(value.p) == "number" or type(value.pitch) == "number"
    ) and (
        type(value.y) == "number" or type(value.yaw) == "number"
    ))
end

local function angleComponent(value, longName, shortName)
    return value[longName] or value[shortName] or 0
end

local function quantizeNumber(value)
    if value >= 0 then return math.floor(value * 1000 + 0.5) / 1000 end
    return -math.floor(-value * 1000 + 0.5) / 1000
end

local function clone(value)
    if isVector(value) then return Vector(value.x, value.y, value.z) end
    if isAngle(value) then
        return Angle(
            angleComponent(value, "pitch", "p"),
            angleComponent(value, "yaw", "y"),
            angleComponent(value, "roll", "r")
        )
    end
    if type(value) ~= "table" then return value end

    local out = {}
    for k, v in pairs(value) do out[k] = clone(v) end
    return out
end

local function quantize(value)
    if type(value) == "number" then return quantizeNumber(value) end
    if isVector(value) then return Vector(quantizeNumber(value.x), quantizeNumber(value.y), quantizeNumber(value.z)) end
    if isAngle(value) then
        return Angle(
            quantizeNumber(angleComponent(value, "pitch", "p")),
            quantizeNumber(angleComponent(value, "yaw", "y")),
            quantizeNumber(angleComponent(value, "roll", "r"))
        )
    end
    if type(value) ~= "table" then return value end

    local out = {}
    for k, v in pairs(value) do out[k] = quantize(v) end
    return out
end

local function same(a, b)
    if a == b and type(a) ~= "table" then return true end
    if isVector(a) and isVector(b) then return a.x == b.x and a.y == b.y and a.z == b.z end
    if isAngle(a) and isAngle(b) then
        return angleComponent(a, "pitch", "p") == angleComponent(b, "pitch", "p")
            and angleComponent(a, "yaw", "y") == angleComponent(b, "yaw", "y")
            and angleComponent(a, "roll", "r") == angleComponent(b, "roll", "r")
    end
    if type(a) ~= type(b) then return false end
    if type(a) ~= "table" then return a == b end

    for k, v in pairs(a) do
        if not same(v, b[k]) then return false end
    end
    for k in pairs(b) do
        if a[k] == nil then return false end
    end
    return true
end

local function hasMethod(value, methodName)
    if value == nil then return false end
    local ok, method = pcall(function() return value[methodName] end)
    return ok and type(method) == "function"
end

local function defaultEntityValue()
    local value = chip()
    if hasMethod(value, "getPos") and hasMethod(value, "getForward") then return value end
    return DefaultEntityValue
end

local function isEntityDefaultName(name)
    name = string.lower(tostring(name or ""))
    if string.find(name, "ready$")
        or string.find(name, "ammotype$")
        or string.find(name, "ammocount$")
        or string.find(name, "ammocountmax$")
        or string.find(name, "magsize$")
        or string.find(name, "armor$")
        or string.find(name, "armormax$") then return false end

    return string.find(name, "driver")
        or string.find(name, "chair")
        or string.find(name, "parent")
        or string.find(name, "owner")
        or string.find(name, "entity")
        or string.match(name, "^gun%d*$")
        or name == "base"
end

local function defaultNetVarValue(name)
    name = string.lower(tostring(name or ""))
    if isEntityDefaultName(name) then return defaultEntityValue() end
    if string.find(name, "filter") or string.find(name, "list") or string.find(name, "data") or string.find(name, "ids") or string.find(name, "positions") or string.find(name, "contacts") or string.find(name, "targetstate") or string.find(name, "table") then return {} end
    if string.find(name, "pos") or string.find(name, "vel") or string.find(name, "vector") or string.find(name, "aim") then return Vector() end
    if string.find(name, "ang") or string.find(name, "angle") then return Angle() end
    if string.find(name, "active") or string.find(name, "enabled") or string.find(name, "ready") or string.find(name, "lock") then return false end
    if string.find(name, "name") or string.find(name, "text") or string.find(name, "status") or string.find(name, "type") then return "" end
    if string.find(name, "count") or string.find(name, "amount") or string.find(name, "size") or string.find(name, "armor") or string.find(name, "state") or string.find(name, "fov") or string.find(name, "id") then return 0 end
    return 0
end

local function defaultForTypeCode(code)
    if code == 1 then return false end
    if code == 2 then return 0 end
    if code == 3 then return "" end
    if code == 4 then return Vector() end
    if code == 5 then return Angle() end
    if code == 6 then return defaultEntityValue() end
    if code == 7 or code == 8 then return {} end
    return 0
end

local function validDefaultEntity(value)
    local ok, valid = pcall(isValid, value)
    return ok and valid == true
end

local function inferNetVarTypeCode(value)
    if type(value) == "boolean" then return 1 end
    if type(value) == "number" then return 2 end
    if type(value) == "string" then return 3 end
    if isVector(value) then return 4 end
    if isAngle(value) then return 5 end
    if validDefaultEntity(value) then return 6 end
    if type(value) ~= "table" then return 3 end

    local n = 0
    for key in pairs(value) do
        if type(key) ~= "number" or key < 1 or key % 1 ~= 0 then return 8 end
        if key > n then n = key end
    end
    for i = 1, n do
        if value[i] == nil then return 8 end
    end
    return 7
end

local function netVar(name)
    local entry = rawget(NetVars, name)
    if entry then return entry end

    entry = setmetatable({ name = name, interval = 0, dirty = false }, netVarProto)
    rawset(NetVars, name, entry)
    return entry
end

if getmetatable(NetVars) == nil then
    setmetatable(NetVars, {
        __index = function(_, key)
            if type(key) ~= "string" then return nil end
            return netVar(key)
        end
    })
end

function netVarProto:get(fallback)
    local value = NetVarValues[self.name]
    if value ~= nil then return value end
    if fallback ~= nil then return fallback end
    if self.typeCode ~= nil then return defaultForTypeCode(self.typeCode) end
    return defaultNetVarValue(self.name)
end

function netVarProto:addTrigger(fn)
    if type(fn) ~= "function" then return nil end
    local list = NetworkedVariableTriggers[self.name] or {}
    NetworkedVariableTriggers[self.name] = list
    list[#list + 1] = fn
    return #list
end

function netVarProto:removeTrigger(idx)
    local list = NetworkedVariableTriggers[self.name]
    if not list or not list[idx] then return false end
    table.remove(list, idx)
    return true
end

function netVarProto:clearTriggers()
    NetworkedVariableTriggers[self.name] = nil
end

function netVarProto:changed(keepChangedState, slot, epsilon)
    return changed(self:get(), keepChangedState, slot or ("__netvar:" .. self.name), epsilon)
end

function netVarProto:resetChanged(slot)
    return resetChanged(slot or ("__netvar:" .. self.name))
end

function netVarProto:update(value)
    if not SERVER then return false end
    return updateNetworkedVariable(self.name, value)
end

function netVarProto:setInterval(interval)
    if not SERVER then return false end
    return setNetworkedVariableInterval(self.name, interval)
end

local function fireNetVarTriggers(name, value)
    local list = NetworkedVariableTriggers[name]
    if not list then return end
    for i = 1, #list do
        xpcall(list[i], function(err) print("[SS-SWH] NetVar trigger error: ", err) end, value, name)
    end
end

local function splitPorts(str)
    local names, types = {}, {}
    if type(str) ~= "string" then return names, types end

    for token in string.gmatch(str, "[^,%s]+") do
        local name, portType = string.match(token, "^([^:]+):?(.*)$")
        if name and name ~= "" then
            names[#names + 1] = name
            types[#types + 1] = portType ~= "" and portType or "NORMAL"
        end
    end
    return names, types
end

function inputProto:enableNetworking(enabled, interval)
    if not SERVER then return false end
    local entry = addNetworkedVariable(self.name, interval)
    entry.input = true
    entry.enabled = enabled ~= false
    self.networked = entry.enabled
    if entry.enabled and InputValues[self.name] ~= nil then
        updateNetworkedVariable(self.name, InputValues[self.name], interval)
    end
    return entry.enabled
end

function changed(value, keepChangedState, slot, epsilon)
    slot = slot or "__default"
    local previous = ChangedSlots[slot]
    local didChange

    if epsilon and type(value) == "number" and type(previous) == "number" then
        didChange = math.abs(value - previous) > epsilon
    else
        didChange = not same(value, previous)
    end

    if keepChangedState ~= false then ChangedSlots[slot] = clone(value) end
    return didChange
end

function resetChanged(slot)
    if slot ~= nil then
        ChangedSlots[slot] = nil
    else
        ChangedSlots = {}
    end
end

local function vehicleFromSource(source)
    if not source then return nil end

    if type(source) == "table" and source.get then
        local ok, value = pcall(source.get, source)
        if ok and validEntity(value) then return value end
    end

    if SERVER and type(source) == "table" and source.name and wire.ports then
        local value = wire.ports[source.name]
        if validEntity(value) then return value end
    end

    if validEntity(source) then return source end
    return nil
end

local function checkDriverDetector(detector, force)
    local vehicle = vehicleFromSource(detector.source)
    if not validEntity(vehicle) then vehicle = nil end

    local driver = nil
    if vehicle then
        local ok, result = pcall(function()
            return vehicle:getDriver()
        end)
        if ok and validEntity(result) then driver = result end
    end

    if not force and driver == detector.driver and vehicle == detector.vehicle then return end

    local previous = detector.driver
    detector.vehicle = vehicle
    detector.driver = driver
    networkDebug("driver detector vehicle=", vehicle, " driver=", driver, " previous=", previous)
    xpcall(detector.fn, function(err) print("[SS-SWH] driver detector error: ", err) end, driver, vehicle, previous)
end

local function installDriverHook()
    if DriverHookInstalled then return end
    DriverHookInstalled = true
    hook.add("think", "SSSWH_DriverDetection", function()
        local now = CurTime()
        for _, detector in pairs(DriverDetectors) do
            if now >= detector.nextCheck then
                detector.nextCheck = now + detector.interval
                checkDriverDetector(detector, false)
            end
        end
    end)
end

function addDriverDetection(vehicleOrInput, fn, interval)
    if type(fn) ~= "function" then return nil end
    local detector = {
        source = vehicleOrInput,
        fn = fn,
        interval = interval or DriverDetectionInterval or 0.5,
        nextCheck = 0,
        driver = nil,
        vehicle = nil
    }
    detector.getDriver = function() return detector.driver end
    detector.getVehicle = function() return detector.vehicle end
    detector.check = function() checkDriverDetector(detector, true) end
    detector.remove = function()
        for i = #DriverDetectors, 1, -1 do
            if DriverDetectors[i] == detector then table.remove(DriverDetectors, i) end
        end
    end
    DriverDetectors[#DriverDetectors + 1] = detector
    networkDebug("added driver detector source=", vehicleOrInput, " interval=", detector.interval)
    installDriverHook()
    checkDriverDetector(detector, true)
    return detector
end

function updateInputs(str)
    local names, types = splitPorts(str)
    for i = 1, #names do
        Inputs[names[i]] = setmetatable({ name = names[i] }, inputProto)
    end
    if SERVER then wire.adjustInputs(names, types) end
    return names, types, Inputs
end

function updateOutputs(str)
    local names, types = splitPorts(str)
    for i = 1, #names do
        Outputs[names[i]] = setmetatable({ name = names[i] }, outputProto)
    end
    if SERVER then wire.adjustOutputs(names, types) end
    return names, types, Outputs
end

function addInputTrigger(name, fn)
    if not Inputs[name] then Inputs[name] = setmetatable({ name = name }, inputProto) end
    return Inputs[name]:addTrigger(fn)
end

if SERVER then
    local DirtyOrder = DirtyOrder or {}
    local DirtyLookup = DirtyLookup or {}
    local NextFlush = 0
    local NextStatePoll = 0

    local function markDirty(entry)
        if DirtyLookup[entry.name] then return end
        DirtyLookup[entry.name] = true
        DirtyOrder[#DirtyOrder + 1] = entry
    end

    function addNetworkedVariable(name, interval)
        if type(name) ~= "string" or name == "" then return nil end
        local entry = netVar(name)
        entry.enabled = true
        if interval ~= nil then entry.interval = tonumber(interval) or 0 end
        return entry
    end

    function setNetworkedVariableInterval(name, interval)
        local entry = addNetworkedVariable(name)
        if not entry then return false end
        entry.interval = tonumber(interval) or 0
        return true
    end

    function updateNetworkedVariable(name, value, interval)
        if value == nil then return false end
        local entry = addNetworkedVariable(name, interval)
        if not entry or entry.enabled == false then return false end

        value = quantize(value)
        if same(value, entry.value) and not entry.dirty then return false end

        entry.typeCode = inferNetVarTypeCode(value)
        entry.value = value
        NetVarValues[name] = value
        entry.dirty = true
        markDirty(entry)
        fireNetVarTriggers(name, value)
        networkDebug("queued netvar ", name, " value=", value)
        NetworkReplicationStats.variablesQueued = (NetworkReplicationStats.variablesQueued or 0) + 1
        return true
    end

    local TYPE_NIL = 0
    local TYPE_BOOL = 1
    local TYPE_NUMBER = 2
    local TYPE_STRING = 3
    local TYPE_VECTOR = 4
    local TYPE_ANGLE = 5
    local TYPE_ENTITY = 6
    local TYPE_ARRAY = 7
    local TYPE_MAP = 8

    local function valueTypeCode(value)
        if value == nil then return TYPE_NIL end
        if type(value) == "boolean" then return TYPE_BOOL end
        if type(value) == "number" then return TYPE_NUMBER end
        if type(value) == "string" then return TYPE_STRING end
        if validEntity(value) then return TYPE_ENTITY end
        if isVector(value) then return TYPE_VECTOR end
        if isAngle(value) then return TYPE_ANGLE end
        if type(value) ~= "table" then return TYPE_STRING end

        local n = 0
        for k in pairs(value) do
            if type(k) ~= "number" or k < 1 or k % 1 ~= 0 then return TYPE_MAP end
            if k > n then n = k end
        end
        for i = 1, n do
            if value[i] == nil then return TYPE_MAP end
        end
        return TYPE_ARRAY
    end

    local function writeValue(value)
        local code = valueTypeCode(value)
        net.writeUInt(code, 4)

        if code == TYPE_NIL then
            return
        elseif code == TYPE_BOOL then
            net.writeBool(value == true)
        elseif code == TYPE_NUMBER then
            net.writeDouble(value)
        elseif code == TYPE_STRING then
            net.writeString(tostring(value))
        elseif code == TYPE_VECTOR then
            net.writeVector(value)
        elseif code == TYPE_ANGLE then
            net.writeAngle(value)
        elseif code == TYPE_ENTITY then
            net.writeEntity(value)
        elseif code == TYPE_ARRAY then
            net.writeUInt(#value, 16)
            for i = 1, #value do writeValue(value[i]) end
        elseif code == TYPE_MAP then
            local count = 0
            for _ in pairs(value) do count = count + 1 end
            net.writeUInt(count, 16)
            for k, v in pairs(value) do
                writeValue(k)
                writeValue(v)
            end
        end
    end

    local function validRecipient(target)
        if target == nil then return true end
        local ok, valid = pcall(isValid, target)
        return ok and valid == true
    end

    local function sendToTargets()
        if NetworkTargets == nil or NetworkTargets == false then
            net.send(nil)
            return true
        end

        if type(NetworkTargets) ~= "table" then
            if not validRecipient(NetworkTargets) then return false end
            net.send(NetworkTargets)
            return true
        end

        local targets = {}
        for _, target in pairs(NetworkTargets) do
            if validRecipient(target) then
                targets[#targets + 1] = target
            end
        end

        if #targets <= 0 then return false end
        net.send(targets)
        return true
    end

    local function sendRecords(records)
        if #records <= 0 then return end
        networkDebug("sending records count=", #records, " target=", NetworkTargets)
        net.start(NET_MESSAGE)
        net.writeUInt(#records, 16)
        for i = 1, #records do
            net.writeString(records[i][1])
            writeValue(records[i][2])
        end
        if not sendToTargets() then return end
        NetworkReplicationStats.packetsSent = (NetworkReplicationStats.packetsSent or 0) + 1
        NetworkReplicationStats.variablesSent = (NetworkReplicationStats.variablesSent or 0) + #records
    end

    local function flush(force)
        if not force and #DirtyOrder <= 0 then return end
        local now = CurTime()
        if not force and now < NextFlush then return end
        NextFlush = now + NetworkFlushInterval

        local records, remaining = {}, {}
        for i = 1, #DirtyOrder do
            local entry = DirtyOrder[i]
            DirtyLookup[entry.name] = nil

            if entry.dirty and entry.enabled ~= false then
                local due = force or not entry.lastSent or entry.interval <= 0 or now - entry.lastSent >= entry.interval
                if due and #records < NetworkMaxRecordsPerFlush then
                    entry.dirty = false
                    entry.lastSent = now
                    records[#records + 1] = { entry.name, entry.value }
                else
                    remaining[#remaining + 1] = entry
                    DirtyLookup[entry.name] = true
                    NetworkReplicationStats.variablesDeferred = (NetworkReplicationStats.variablesDeferred or 0) + 1
                end
            end
        end

        DirtyOrder = remaining
        sendRecords(records)
    end

    function addNetworkedVariableTrigger(name, fn)
        return NetVars[name]:addTrigger(fn)
    end

    local function addStatePoller(kind, prefix, interval)
        local existing = NetworkedStatePollers[prefix]
        local names = stateNames(prefix)
        NetworkedStatePollers[prefix] = {
            kind = kind,
            prefix = prefix,
            names = names,
            interval = interval or NetworkFlushInterval,
            nextUpdate = existing and existing.nextUpdate or 0
        }
        NextStatePoll = 0
    end

    function addNetworkedGunState(prefix, interval, enableReloadTime)
        prefix = normalizeGunPrefix(prefix)
        local names = stateNames(prefix)

        addNetworkedVariable(names.Name, 0)
        addNetworkedVariable(names.Ready, interval)
        if enableReloadTime then
            addNetworkedVariable(names.ReloadTime, interval)
            addNetworkedVariable(names.ReloadProgress, 0)
            addNetworkedVariable(names.MaxReloadTime, 0)

        addNetworkedVariable(names.AmmoType, interval)
        addNetworkedVariable(names.AmmoCount, interval)
        addNetworkedVariable(names.AmmoCountMax, interval)
        addNetworkedVariable(names.MagSize, interval)

    local function resolveEntity(value)
        if type(value) == "table" and value.get then value = value:get() end
        local valueType = type(value)
        if value == nil or valueType == "number" or valueType == "string" or valueType == "boolean" or valueType == "table" then return nil end
        if isValid(value) then return value end
        return nil
    end

    local function acfValue(ent, methodName, fallback)
        if not ent then return fallback end
        local method = ent[methodName]
        if not method then return fallback end
        local value = method(ent)
        if value ~= nil then return value end
        return fallback
    end

    local function deferStatePoll(prefix, interval)
        local poller = NetworkedStatePollers[prefix]
        if not poller then return end
        poller.nextUpdate = CurTime() + (interval or poller.interval)
    end

    local function staticStateChanged(prefix, ent)
        local state = NetworkedStateStatic[prefix]
        if not state then
            state = {}
            NetworkedStateStatic[prefix] = state
        end

        if state.entity ~= ent then
            state.entity = ent
            return true
        end

        return false
    end

    local function markStaticStateDirty(prefix)
        local state = NetworkedStateStatic[prefix]
        if state then state.entity = nil end
    end

    function updateNetworkedGunState(prefix, inputOrGun, interval)
        prefix = normalizeGunPrefix(prefix)
        local names = stateNames(prefix)
        StateKinds[prefix] = "gun"
        local gun = resolveEntity(inputOrGun)
        local updateStatic = staticStateChanged(prefix, gun)
        if updateStatic then
            updateNetworkedVariable(names.Name, acfValue(gun, "acfName", ""), 0)
        end
        updateNetworkedVariable(names.Ready, acfValue(gun, "acfReady", false), interval)
        if NetVars[names.ReloadTime] then
            local reloadTime = acfValue(gun, "acfReloadTime", 0)
            updateNetworkedVariable(names.ReloadTime, reloadTime, interval)
            updateNetworkedVariable(names.ReloadProgress, acfValue(gun, "acfReloadProgress", 0), interval)
            if reloadTime ~= 0 then
                updateNetworkedVariable(names.MaxReloadTime, acfValue(gun, "acfReloadTimeMax", 0), interval)
            end
        end
        updateNetworkedVariable(names.AmmoType, acfValue(gun, "acfAmmoType", ""), interval)
        updateNetworkedVariable(names.AmmoCount, acfValue(gun, "acfMagRounds", 0), interval)
        if updateStatic then
            updateNetworkedVariable(names.AmmoCountMax, acfValue(gun, "acfTotalAmmoCount", 0), interval)
            updateNetworkedVariable(names.MagSize, acfValue(gun, "acfMagSize", 0), interval)
        end
        deferStatePoll(prefix, interval)
        return gun ~= nil
    end

    function updateNetworkedPropArmorState(prefix, inputOrProp, interval)
        prefix = normalizePropArmorPrefix(prefix)
        local names = stateNames(prefix)
        StateKinds[prefix] = "propArmor"
        local prop = resolveEntity(inputOrProp)
        local updateStatic = staticStateChanged(prefix, prop)
        updateNetworkedVariable(names.Armor, acfValue(prop, "acfPropArmor", 0), interval)
        if updateStatic then updateNetworkedVariable(names.ArmorMax, acfValue(prop, "acfPropArmorMax", 0), interval) end
        deferStatePoll(prefix, interval)
        return prop ~= nil
    end

    hook.add("input", "SSSWH_Input", function(name, value)
        InputValues[name] = clone(value)
        InputValidity[name] = wire.self():isWired(name)
        if SyncInputCommandsEnabled then _G[name] = InputValues[name] end

        local entry = rawget(NetVars, name)
        if entry and entry.input and entry.enabled then updateNetworkedVariable(name, value) end

        if NetworkedStatePollers[name] then markStaticStateDirty(name) end

        local list = InputTriggers[name]
        if list then
            for i = 1, #list do
                xpcall(list[i], function(err) print("[SS-SWH] input trigger error: ", err) end, value, name)
            end
        end
    end)

    local function pollNetworkedStates()
        local now = CurTime()
        if now < NextStatePoll then return end

        local nextPoll
        for _, poller in pairs(NetworkedStatePollers) do
            if now >= poller.nextUpdate then
                poller.nextUpdate = now + poller.interval
                if poller.kind == "gun" then
                    updateNetworkedGunState(poller.prefix, InputValues[poller.prefix], poller.interval)
                elseif poller.kind == "propArmor" then
                    updateNetworkedPropArmorState(poller.prefix, InputValues[poller.prefix], poller.interval)
                end
            end

            if nextPoll == nil or poller.nextUpdate < nextPoll then nextPoll = poller.nextUpdate end
        end
        NextStatePoll = nextPoll or now + NetworkFlushInterval
    end

    hook.add("think", "SSSWH_NetFlush", function()
        pollNetworkedStates()
        flush(false)
    end)

    hook.add("ClientInitialized", "SSSWH_ClientInit", function(ply)
        local records = {}
        for name, value in pairs(NetVarValues) do records[#records + 1] = { name, clone(value) } end
        if #records <= 0 then return end
        local oldTargets = NetworkTargets
        NetworkTargets = ply
        sendRecords(records)
        NetworkTargets = oldTargets
    end)

        if not same(oldValue, value) then fireNetVarTriggers(name, value) end
        return true
    end

    local TYPE_NIL = 0
    local TYPE_BOOL = 1
    local TYPE_NUMBER = 2
    local TYPE_STRING = 3
    local TYPE_VECTOR = 4
    local TYPE_ANGLE = 5
    local TYPE_ENTITY = 6
    local TYPE_ARRAY = 7
    local TYPE_MAP = 8

    local function readValue()
        local code = net.readUInt(4)

        if code == TYPE_NIL then return nil end
        if code == TYPE_BOOL then return net.readBool() end
        if code == TYPE_NUMBER then return net.readDouble() end
        if code == TYPE_STRING then return net.readString() end
        if code == TYPE_VECTOR then return net.readVector() end
        if code == TYPE_ANGLE then return net.readAngle() end
        if code == TYPE_ENTITY then return net.readEntity() end

        if code == TYPE_ARRAY then
            local out = {}
            local count = net.readUInt(16)
            for i = 1, count do out[i] = readValue() end
            return out
        end

        if code == TYPE_MAP then
            local out = {}
            local count = net.readUInt(16)
            for i = 1, count do out[readValue()] = readValue() end
            return out
        end

        return nil
    end

    local function usableInputValue(name, value, typeCode)
        if typeCode == TYPE_ENTITY or isEntityDefaultName(name) then
            if hasMethod(value, "getPos") and hasMethod(value, "getForward") then return value end
            return defaultEntityValue()
        end

        if value ~= nil then return value end
        return defaultNetVarValue(name)
    end

    net.receive(NET_MESSAGE, function()
        local count = net.readUInt(16)
        for i = 1, count do
            local name = net.readString()
            local value = readValue()
            local typeCode = inferNetVarTypeCode(value)
            value = usableInputValue(name, value, typeCode)
            local oldValue = NetVarValues[name]
            local entry = netVar(name)
            entry.typeCode = typeCode
            NetVarValues[name] = clone(value)
            InputValues[name] = clone(value)
            InputValidity[name] = true
            if not same(oldValue, value) then fireNetVarTriggers(name, value) end

            local list = InputTriggers[name]
            if list and not same(oldValue, value) then
                for n = 1, #list do
                    xpcall(list[n], function(err) print("[SS-SWH] client input trigger error: ", err) end, value, name)
                end
            end
        end
    end)
