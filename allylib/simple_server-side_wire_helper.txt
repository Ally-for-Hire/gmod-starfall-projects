--@name Simple Server-Side Wire Helper
--@author Ally for Hire / Merydian9

--[[
Small utility that turns human-readable wire I/O strings into actual ports and
provides helper tables (`Inputs`, `Outputs`, `InputValues`) for quick access.

Optimized pass goals:
- lower parse overhead (single-pass parser, no token table)
- safer trigger execution (isolated callback failures)
- lightweight API additions while preserving compatibility

Function index:
- `updateInputs(str)`: creates or refreshes wire inputs from a compact port string.
- `updateOutputs(str)`: creates or refreshes wire outputs from a compact port string.
- `syncInputCommands(enabled)`: mirrors input values into globals matching their input names.
- `changed(value, keepChangedState, slot, epsilon)`: change detector with optional persistent dirty state.
- `resetChanged(slot)`: clears `changed(...)` state for one slot or all slots.
- `addInputTrigger(name, fn)`: registers a trigger callback on a named input.
- `safeWireCreate(entI, entO, inputName, outputName)`: protected `wire.create(...)`.
- `safeWireDelete(ent, inputName)`: protected `wire.delete(...)`.
- `Inputs.<Name>:get()`: reads the cached input value.
- `Inputs.<Name>:addTrigger(fn)`: registers a trigger callback on that input object.
- `Inputs.<Name>:removeTrigger(idx)`: removes a previously added trigger by index.
- `Inputs.<Name>:clearTriggers()`: removes all triggers for that input.
- `Outputs.<Name>:update(value)`: writes a cached output value to Wire.
- `InputValues.<Name>`: latest cached input payload for direct reads.
]]--

-- Version 1.6.0

--[[
Usage:
    updateInputs("Base:ENTITY Fire:NORMAL")
    updateOutputs("Ready:NORMAL Aim:VECTOR")
    Inputs.Fire:addTrigger(function(val) if val > 0 then Outputs.Ready:update(1) end end)
    Outputs.Aim:update(Vector())
    local baseEnt = InputValues.Base
]]--

if CLIENT then return end

Inputs, Outputs, InputValues = Inputs or {}, Outputs or {}, InputValues or {}
local InputTriggers = InputTriggers or {}

local LastInputNames, LastInputTypes = LastInputNames or {}, LastInputTypes or {}
local LastOutputNames, LastOutputTypes = LastOutputNames or {}, LastOutputTypes or {}
local LastOutputValues = LastOutputValues or {}
local ChangedState = ChangedState or {}
local SyncInputCommandsEnabled = SyncInputCommandsEnabled or false
local SyncedInputCommandNames = SyncedInputCommandNames or {}
local SyncedInputCommandBackup = SyncedInputCommandBackup or {}

local _type = type
local _pairs = pairs
local _setmetatable = setmetatable
local _xpcall = xpcall
local _tostring = tostring
local _print = print
local _table_remove = table.remove
local _debug = debug
local _math_abs = math.abs
local _G = _G
local _wire = wire
local _wire_ports = wire.ports
local _string_gmatch = string.gmatch
local _string_match = string.match

local ReusableInputKeep = ReusableInputKeep or {}

-- Input helper prototype ---------------------------------------------------
local inputProto = {}
inputProto.__index = inputProto

function inputProto:get()
    return InputValues[self.name]
end

function inputProto:addTrigger(fn)
    if _type(fn) ~= "function" then return nil end

    local name = self.name
    local list = InputTriggers[name]
    if list == nil then
        list = {}
        InputTriggers[name] = list
    end

    local idx = #list + 1
    list[idx] = fn
    return idx
end

function inputProto:removeTrigger(idx)
    if _type(idx) ~= "number" then return false end

    local list = InputTriggers[self.name]
    if not list or not list[idx] then return false end

    _table_remove(list, idx) -- keep trigger execution order deterministic
    if #list == 0 then
        InputTriggers[self.name] = nil
    end

    return true
end

function inputProto:clearTriggers()
    InputTriggers[self.name] = nil
end

-- Output helper prototype --------------------------------------------------
local outputProto = {}
outputProto.__index = outputProto

local function isSequentialTable(tbl)
    local count = 0
    for key in _pairs(tbl) do
        if _type(key) ~= "number" or key < 1 or key % 1 ~= 0 then
            return false, 0
        end

        count = count + 1
    end

    return true, count
end

local function cloneOutputValue(value)
    if _type(value) ~= "table" then
        return value
    end

    local clone = {}
    for key, innerValue in _pairs(value) do
        clone[key] = cloneOutputValue(innerValue)
    end

    return clone
end

local function outputValuesEqual(a, b)
    if a == b and _type(a) ~= "table" then
        return true
    end

    local typeA = _type(a)
    if typeA ~= _type(b) then
        return false
    end

    if typeA == "vector" then
        return a.x == b.x and a.y == b.y and a.z == b.z
    end

    if typeA == "angle" then
        return a.pitch == b.pitch and a.yaw == b.yaw and a.roll == b.roll
    end

    if typeA ~= "table" then
        return a == b
    end

    local sequentialA, countA = isSequentialTable(a)
    local sequentialB, countB = isSequentialTable(b)
    if sequentialA ~= sequentialB or countA ~= countB then
        return false
    end

    if sequentialA then
        for i = 1, countA do
            if not outputValuesEqual(a[i], b[i]) then
                return false
            end
        end

        return true
    end

    for key, valueA in _pairs(a) do
        if not outputValuesEqual(valueA, b[key]) then
            return false
        end
    end

    for key in _pairs(b) do
        if a[key] == nil then
            return false
        end
    end

    return true
end

function outputProto:update(value)
    local name = self.name
    local lastValue = LastOutputValues[name]
    if lastValue ~= nil and outputValuesEqual(value, lastValue) then
        return false
    end

    LastOutputValues[name] = cloneOutputValue(value)
    _wire_ports[name] = value
    return true
end

-- Parse helpers ------------------------------------------------------------
-- single-pass parser: "A:TYPE B C:TYPE" -> names/types arrays
local function splitPortString(portString)
    local names, types = {}, {}

    if _type(portString) ~= "string" then
        return names, types
    end

    local n = 0
    for token in _string_gmatch(portString, "[^,%s]+") do
        local name, portType = _string_match(token, "^([^:]+):?(.*)$")
        if name and name ~= "" then
            n = n + 1
            names[n] = name
            types[n] = (portType ~= nil and portType ~= "") and portType or "NORMAL"
        end
    end

    return names, types
end

-- Internal wire input hook -------------------------------------------------
local function onTriggerError(err)
    return "[SS-SWH] trigger error: " .. _tostring(err)
end

local function syncInputGlobal(name, value)
    if not SyncInputCommandsEnabled then return end

    if not SyncedInputCommandNames[name] then
        SyncedInputCommandNames[name] = true
        SyncedInputCommandBackup[name] = _G[name]
    end

    _G[name] = value
end

local function unsyncInputGlobal(name)
    if not SyncedInputCommandNames[name] then return end

    _G[name] = SyncedInputCommandBackup[name]
    SyncedInputCommandBackup[name] = nil
    SyncedInputCommandNames[name] = nil
end

local function doWireInput(name, value)
    -- Keep legacy behavior: ignore nil update payloads
    if value == nil then return end

    InputValues[name] = value
    syncInputGlobal(name, value)

    local triggers = InputTriggers[name]
    if not triggers then return end

    for i = 1, #triggers do
        local fn = triggers[i]
        if fn then
            local ok, msg = _xpcall(fn, onTriggerError, value)
            if not ok then _print(msg) end
        end
    end
end

local function samePortLayout(namesA, typesA, namesB, typesB)
    local len = #namesA
    if len ~= #namesB or len ~= #typesA or len ~= #typesB then return false end

    for i = 1, len do
        if namesA[i] ~= namesB[i] or typesA[i] ~= typesB[i] then
            return false
        end
    end

    return true
end

local function copyPortLayout(srcNames, srcTypes, dstNames, dstTypes)
    for i = 1, #dstNames do
        dstNames[i] = nil
        dstTypes[i] = nil
    end

    for i = 1, #srcNames do
        dstNames[i] = srcNames[i]
        dstTypes[i] = srcTypes[i]
    end
end

local function clearSet(tbl)
    for key in _pairs(tbl) do
        tbl[key] = nil
    end
end

local function resolveChangedSlot(explicitSlot)
    if explicitSlot ~= nil then return explicitSlot end

    if _debug and _debug.getinfo then
        local info = _debug.getinfo(3, "Sl") -- caller of changed/resetChanged
        if info then
            return (_tostring(info.short_src) .. ":" .. _tostring(info.currentline))
        end
    end

    return "__default_slot"
end

local function valuesEqual(a, b, epsilon)
    if a == b then return true end

    if epsilon and _type(a) == "number" and _type(b) == "number" then
        return _math_abs(a - b) <= epsilon
    end

    return false
end

-- Public API ---------------------------------------------------------------
function syncInputCommands(enabled)
    local turnOn = enabled == true

    if turnOn == SyncInputCommandsEnabled then
        return SyncInputCommandsEnabled
    end

    SyncInputCommandsEnabled = turnOn

    if SyncInputCommandsEnabled then
        for name, value in _pairs(InputValues) do
            syncInputGlobal(name, value)
        end
    else
        for name in _pairs(SyncedInputCommandNames) do
            unsyncInputGlobal(name)
        end
    end

    return SyncInputCommandsEnabled
end

function changed(value, keepChangedState, slot, epsilon)
    local key = resolveChangedSlot(slot)
    local state = ChangedState[key]

    if not state then
        ChangedState[key] = { last = value, pending = value, dirty = false }
        return true
    end

    if not valuesEqual(value, state.last, epsilon) then
        state.pending = value
        state.dirty = true
    end

    if not state.dirty then
        return false
    end

    if not keepChangedState then
        state.last = state.pending
        state.dirty = false
    end

    return true
end

function resetChanged(slot)
    if slot == nil then
        for _, state in _pairs(ChangedState) do
            if state.dirty then
                state.last = state.pending
                state.dirty = false
            end
        end
        return true
    end

    local key = resolveChangedSlot(slot)
    local state = ChangedState[key]
    if not state then return false end

    if state.dirty then
        state.last = state.pending
        state.dirty = false
    end

    return true
end

function updateInputs(str)
    local names, types = splitPortString(str)

    local changed = not samePortLayout(names, types, LastInputNames, LastInputTypes)
    if not changed then
        return names, types, Inputs, false
    end

    local keep = ReusableInputKeep
    clearSet(keep)
    for i = 1, #names do
        keep[names[i]] = true
    end

    for key in _pairs(InputTriggers) do
        if not keep[key] then InputTriggers[key] = nil end
    end

    for key in _pairs(InputValues) do
        if not keep[key] then
            InputValues[key] = nil
            unsyncInputGlobal(key)
        end
    end

    for key in _pairs(Inputs) do
        Inputs[key] = nil
    end

    for i = 1, #names do
        local name = names[i]
        Inputs[name] = _setmetatable({ name = name }, inputProto)
    end

    copyPortLayout(names, types, LastInputNames, LastInputTypes)
    _wire.adjustInputs(names, types)
    return names, types, Inputs, true
end

function updateOutputs(str)
    local names, types = splitPortString(str)

    local changed = not samePortLayout(names, types, LastOutputNames, LastOutputTypes)
    if not changed then
        return names, types, Outputs, false
    end

    for key in _pairs(Outputs) do
        Outputs[key] = nil
    end

    for key in _pairs(LastOutputValues) do
        LastOutputValues[key] = nil
    end

    for i = 1, #names do
        local name = names[i]
        Outputs[name] = _setmetatable({ name = name }, outputProto)
    end

    copyPortLayout(names, types, LastOutputNames, LastOutputTypes)
    _wire.adjustOutputs(names, types)
    return names, types, Outputs, true
end

function addInputTrigger(name, fn)
    local inputObj = Inputs[name]
    if not inputObj then return nil end
    return inputObj:addTrigger(fn)
end

function safeWireCreate(entI, entO, inputName, outputName)
    local ok = pcall(_wire.create, entI, entO, inputName, outputName)
    return ok
end

function safeWireDelete(ent, inputName)
    local ok = pcall(_wire.delete, ent, inputName)
    return ok
end

hook.add("input", "SS-SWH_input", doWireInput)
