--@name Lowlag Radar Track
--@author Ally for Hire / Merydian9
--@include allylib/simple_server-side_wire_helper.txt
--@include allylib/helpful_hud_functions.txt
--@shared

require("allylib/simple_server-side_wire_helper.txt")
require("allylib/helpful_hud_functions.txt")

OUTPUT_UPDATE_CAP = 0.05
FOLLOW_LAST_VELOCITY_TIME = 3
SOUND_VOLUME = 35
CHIP = chip()

if SERVER then
    updateInputs("Radar:ENTITY Position:ARRAY Velocity:ARRAY IDs:ARRAY Owner:ARRAY HitPos:VECTOR Lock:NORMAL")
    updateOutputs("GimbalPos:VECTOR TargetPos:VECTOR TargetVelocity:VECTOR TargetID:NORMAL TargetName:STRING")
    addNetworkedVariable("TrackData")
    
    ActivelyTracking = false
    TargetInArray = false
    LastOutputUpdate = 0
    LastSeen = 0
    SearchAttempts = 0
    TrackRevision = 0
    InternalInfo = {}

    local function publishTrackData()
        updateNetworkedVariable("TrackData", {
            pos = InternalInfo[1] or Vector(),
            id = InternalInfo[3] or 0,
            name = InternalInfo[4] or "Merydian9",
            inArray = TargetInArray == true,
            active = ActivelyTracking == true,
            revision = TrackRevision
        })
    end
    
    Inputs.Radar:addTrigger(function(ent)
        if ent ~= nil and isValid(ent) then
            local successes = 0
            successes = successes + (safeWireCreate(CHIP, ent, "Position", "Position") and 1 or 0)
            successes = successes + (safeWireCreate(CHIP, ent, "Velocity", "Velocity") and 1 or 0)
            successes = successes + (safeWireCreate(CHIP, ent, "Owner", "Owner") and 1 or 0)
            successes = successes + (safeWireCreate(CHIP, ent, "IDs", "IDs") and 1 or 0)
            
            if successes == 4 then return end
        end
        
        safeWireDelete(CHIP, "Position")
        safeWireDelete(CHIP, "Velocity")
        safeWireDelete(CHIP, "IDs")
        safeWireDelete(CHIP, "Owner")
    end)
    
    Inputs.HitPos:addTrigger(function(vec)
        -- Updating outputs is lag-heavy, so we want to limit it when possible
        if ActivelyTracking then return end
        if (timer.curtime() - LastOutputUpdate) < OUTPUT_UPDATE_CAP then return end
        
        Outputs.GimbalPos:update(vec)
        LastOutputUpdate = timer.curtime()
    end)
    
    local function logic(vec)
        if not ActivelyTracking then return end
        if (timer.curtime() - LastOutputUpdate) < OUTPUT_UPDATE_CAP then return end
        
        local targPos = Inputs.Position:get()
        local targVel = Inputs.Velocity:get()
        local targID = Inputs.IDs:get()
        local targName = Inputs.Owner:get()
        
        -- if the arrays are miss-matched we dip
        if (#targPos + #targVel + #targID + #targName) ~= #targPos * 4 then return end
        
        local searchIndex = InternalInfo[5]
        
        -- if it's at the same place in the array we can skip searching for it
        -- otherwise we need to find it
        if targID[searchIndex] == InternalInfo[3] then
            InternalInfo = {targPos[searchIndex], targVel[searchIndex], targID[searchIndex], targName[searchIndex], searchIndex}
            LastSeen = timer.curtime()
            SearchAttempts = 0
            TargetInArray = true
        else
            TargetInArray = false
            for i = 1, #targID do
                if targID[i] ~= InternalInfo[3] then continue end
                
                InternalInfo = {targPos[i], targVel[i], targID[i], targName[i], i}
                LastSeen = timer.curtime()
                SearchAttempts = 0
                TargetInArray = true
                break
            end
        end
        
        -- InternalInfo = {pos, vel, id, name, arrayindex}
        Outputs.TargetPos:update(InternalInfo[1])
        Outputs.TargetVelocity:update(InternalInfo[2])
        Outputs.TargetID:update(InternalInfo[3])
        Outputs.TargetName:update(InternalInfo[4])
        TrackRevision = TrackRevision + 1
        publishTrackData()
        
        local searchPosition = InternalInfo[1] + InternalInfo[2] * 0.1
        local dt = (timer.curtime() - LastSeen)
        if not TargetInArray and dt < FOLLOW_LAST_VELOCITY_TIME then
            if SearchAttempts % 2 == 0 then
                searchPosition = InternalInfo[1] + InternalInfo[2] * dt
            end
            
            SearchAttempts = SearchAttempts + 1
        end
        
        Outputs.GimbalPos:update(searchPosition)
        LastOutputUpdate = timer.curtime()
    end
    
    Inputs.Position:addTrigger(logic)
    timer.create("LRT_downtimeUpdateLoop", 0.1, 0, function() logic(Inputs.Position:get()) end)
    
    Inputs.Lock:addTrigger(function(num)
        if num == 0 then return end
        
        if ActivelyTracking then
            ActivelyTracking = false
            TargetInArray = false
            TrackRevision = TrackRevision + 1
            publishTrackData()
            
            return
        end
        
        -- Grab these from the inputs
        local targPos = Inputs.Position:get()
        if #targPos == 0 then return end -- if we can't lock anything, avoid extra work
        local targVel = Inputs.Velocity:get()
        local targID = Inputs.IDs:get()
        local targName = Inputs.Owner:get()
        local hitPos = Inputs.HitPos:get()
        local targInfo = {targPos[1], targVel[1], targID[1], targName[1], 1}
        
        local minDistance = targPos[1]:getDistance(hitPos)
        local testDistance = 0

        for i = 2, #targPos do
            testDistance = targPos[i]:getDistance(hitPos)
            
            if testDistance < minDistance then
                targInfo = {targPos[i], targVel[i], targID[i], targName[i], i}
                minDistance = testDistance
            end
        end
        
        InternalInfo = targInfo
        ActivelyTracking = true
        TargetInArray = true
        TrackRevision = TrackRevision + 1
        publishTrackData()
    end)
end

if CLIENT then
    TargetPos = Vector()
    TargetID = 0
    TargetName = "Merydian9"
    TargetRange = 0
    ActivelyTracking = false
    TargetInArray = false
    NewTargetPeriod = false
    InitSounds = true
    P = player()
    
    FONT_MAIN = render.createFont("Default", 16, 400, nil, nil, true)
    COLOR_MAIN = Color(50, 255, 50)
    LOCK_INDICATOR = initRT("Tracking Indicator")
    LOST_INDICATOR = initRT("Lost Track Indicator")
    resx, resy = render.getResolution()
    resx = resx / 2
    resy = resy / 2
    
    SOUND_track = nil
    SOUND_lock = nil
    SOUND_dropTrack = nil

    local function ensureSounds()
        if not InitSounds then return end

        SOUND_track = sound.create(P, "acf_extra/airfx/hominglockaquired1.wav")
        SOUND_track:setSoundLevel(SOUND_VOLUME)
        SOUND_dropTrack = sound.create(P, "acf_extra/airfx/hominglocksequence1.wav")
        SOUND_dropTrack:setSoundLevel(SOUND_VOLUME)
        SOUND_lock = sound.create(P, "acf_extra/airfx/satellite_target.wav")
        SOUND_lock:setSoundLevel(SOUND_VOLUME)
        InitSounds = false
    end

    local function refreshTrackedTarget()
        local data = NetworkedVariables.TrackData:get() or {}
        local previousActive = ActivelyTracking

        ensureSounds()
        TargetPos = data.pos or Vector()
        TargetID = data.id or 0
        TargetName = data.name or "Merydian9"
        TargetInArray = data.inArray == true
        ActivelyTracking = data.active == true

        if not ActivelyTracking then
            NewTargetPeriod = true
            SOUND_lock:stop()
            SOUND_track:stop()

            if previousActive then
                SOUND_dropTrack:stop()
                SOUND_dropTrack:play()
            end

            return
        end

        if not previousActive then
            NewTargetPeriod = true
            SOUND_dropTrack:stop()
        else
            NewTargetPeriod = false
        end

        if TargetInArray then
            SOUND_track:play()
            SOUND_lock:stop()
        else
            SOUND_track:stop()
            SOUND_lock:play()
        end
    end

    hook.add("drawhud", "ST_mainDrawLoop", function()
        if not ActivelyTracking or NewTargetPeriod then return end
        
        local c = TargetPos:toScreen()
        if NewTargetPeriod then
            c.x = resx
            c.y = resy
        end
        local x = c.x
        local y = c.y
        local vis = c.visible

        -- if not visible, don't render it
        if not vis then return end
        color(COLOR_MAIN)
        
        if TargetInArray then
            drawCenterRT(x, y, LOCK_INDICATOR, 2)
            drawCenterRT(x, y, LOST_INDICATOR, 2)
        else
            drawCenterRT(x, y, LOST_INDICATOR, 2)
        end
        
        font(FONT_MAIN)
        textSimple(x, y - 30, TargetName, TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER)
        textSimple(x, y + 30, "DIST " .. TargetRange .. "hu", TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER)
        textSimple(x, y + 46, TargetID, TEXT_ALIGN.CENTER, TEXT_ALIGN.CENTER)
    end)
    
    hook.add("renderoffscreen", "LRT_drawStaticElements", function()
        local x, y = 0, 0
        
        selectClearRT(LOCK_INDICATOR)    
            local m = Matrix()
            m:translate(Vector(512, 512, 0))
            m:rotate(Angle(0, 45, 0))
            render.pushMatrix(m)
            
            color(COL_BLACK)
            centerRectOutline(x, y, 58, 58, 4)
            
            color(COL_WHITE)
            centerRectOutline(x, y, 56, 56, 2)
            
            render.popMatrix()
            
        selectClearRT(LOST_INDICATOR)
            local halfLength = 26
            local distOuter = 13
            local distInner = 15
            
            local m = Matrix()
            m:translate(Vector(512, 512, 0))
            m:rotate(Angle(0, 45, 0))
            render.pushMatrix(m)

            color(COL_BLACK)
            line(x + halfLength, y + halfLength, x + distOuter, y + halfLength, 4)
            line(x + halfLength, y + halfLength, x + halfLength, y + distOuter, 4)
            
            line(x + halfLength, y - halfLength, x + distOuter, y - halfLength, 4)
            line(x + halfLength, y - halfLength, x + halfLength, y - distOuter, 4)
            
            line(x - halfLength, y + halfLength, x - distOuter, y + halfLength, 4)
            line(x - halfLength, y + halfLength, x - halfLength, y + distOuter, 4)
            
            line(x - halfLength, y - halfLength, x - distOuter, y - halfLength, 4)
            line(x - halfLength, y - halfLength, x - halfLength, y - distOuter, 4)
            
            color(COL_WHITE)
            line(x + halfLength, y + halfLength, x + distInner, y + halfLength, 2)
            line(x + halfLength, y + halfLength, x + halfLength, y + distInner, 2)
            
            line(x - halfLength, y + halfLength, x - distInner, y + halfLength, 2)
            line(x - halfLength, y + halfLength, x - halfLength, y + distInner, 2)
            
            line(x + halfLength, y - halfLength, x + distInner, y - halfLength, 2)
            line(x + halfLength, y - halfLength, x + halfLength, y - distInner, 2)
            
            line(x - halfLength, y - halfLength, x - distInner, y - halfLength, 2)
            line(x - halfLength, y - halfLength, x - halfLength, y - distInner, 2)
            
            render.popMatrix()
            
        hook.remove("renderoffscreen", "LRT_drawStaticElements")
    end)

    NetworkedVariables.TrackData:addTrigger(refreshTrackedTarget)
    
    timer.create("LRT_targetRangeUpdateLoop", 0.1, 0, function()
        TargetRange = math.round(TargetPos:getDistance(P:getPos()))
    end)
end
