--@include allylib/curve-fitting_ballistics_algorithm.txt
--@server

local curtime = timer.curtime
require("allylib/simple_server-side_wire_helper.txt")
require("allylib/curve-fitting_ballistics_algorithm.txt")

local ZERO_VEC = Vector()

ACTIVE_INTERVAL = 0.05
IDLE_INTERVAL   = 0.2
LOAD_DIST  = 10000
LOAD_PITCH = nil
LOAD_ENABLED = LOAD_PITCH ~= nil
LOAD_ANG = LOAD_ENABLED and Angle(LOAD_PITCH, 0, 0) or nil
AUTO_FIRE_MAX_TARGET_AGE = 0.5
AUTO_FIRE_LOS_CONE_DEG = 1
AUTO_FIRE_LOS_INTERVAL = 0.1
AUTO_FIRE_LOS_POS_EPS = 64
AUTO_FIRE_HOLD_TIME = 0.12

NextThink  = 0
Unlocked   = false
LastGun    = nil
TurretOverrideWired = false
LastLosTime = 0
LastLosPos = Vector()
LastLosResult = false
LastAutoFireUntil = 0

local function clampValue(value, minValue, maxValue)
    if value < minValue then return minValue end
    if value > maxValue then return maxValue end
    return value
end

local function getBasePos(hit, tpos)
    if tpos:getLength() > 0 then
        return tpos
    end
    return hit
end

local function getLoadAimPos(gun, base, targetPos, loadAng)
    local origin = gun:getMassCenterW() or gun:getPos()
    local baseAng = base:getAngles()
    local toTarget = targetPos - origin
    local yaw = toTarget:getAngle().y
    local ang = Angle(baseAng.p - loadAng.p, yaw, 0)
    return origin + ang:getForward() * LOAD_DIST
end

local function getGunOrigin(gun)
    if not gun or not gun:isValid() then return nil end
    return gun:getMassCenterW() or gun:getPos()
end

local function posDeltaLenSq(a, b)
    local dx = a[1] - b[1]
    local dy = a[2] - b[2]
    local dz = a[3] - b[3]
    return dx * dx + dy * dy + dz * dz
end

local function hasLineOfSight(gun, base, targetPos, now)
    local origin = getGunOrigin(gun)
    if not origin then return false end
    if (now - LastLosTime) < AUTO_FIRE_LOS_INTERVAL
    and posDeltaLenSq(targetPos, LastLosPos) <= (AUTO_FIRE_LOS_POS_EPS * AUTO_FIRE_LOS_POS_EPS) then
        return LastLosResult
    end

    local tr = trace.line(origin, targetPos, function(ent)
        if ent == nil then return false end
        if ent == gun or ent == base then return false end

        local entOwner = ent:getOwner()
        if entOwner and entOwner == owner() then
            return false
        end

        return true
    end)

    local result = tr and tr.Hit == false or false
    LastLosTime = now
    LastLosPos = targetPos
    LastLosResult = result
    return result
end

local function shouldAutoFire(gun, base, aimPos, targetPos, now)
    if not gun or not gun:isValid() then return false end
    if not gun:acfReady() then return false end
    if not targetPos or targetPos:getLength() <= 0 then return false end
    if not hasLineOfSight(gun, base, targetPos, now) then return false end

    local origin = getGunOrigin(gun)
    if not origin then return false end

    local toAim = aimPos - origin
    local dist = toAim:getLength()
    if dist <= 0 then return false end

    local aimDir = toAim / dist
    local gunDir = gun:getForward()
    local dot = clampValue(gunDir:dot(aimDir), -1, 1)
    local angularError = math.acos(dot)
    return angularError <= math.rad(AUTO_FIRE_LOS_CONE_DEG)
end

local function main()
    local now = curtime()
    local manualFire = (M1 or 0) ~= 0

    local gun    = Gun or I.Gun
    local base   = Base
    local hit    = HitPos or ZERO_VEC
    local tpos   = TargetPos or ZERO_VEC
    local tvel   = TargetVel or ZERO_VEC
    local hasTrackedPos = tpos:getLength() > 0
    local autoFireEnabled = (AutoFire or 0) ~= 0
    local manual = (M2 or 0) ~= 0
    local turretActiveOverride = (TurretActiveOverride or 0) ~= 0
    local turretActive = Unlocked or (TurretOverrideWired and not turretActiveOverride and (Active or 0) == 1)
    local gunValid = gun and gun:isValid()
    local baseValid = base and base:isValid()
    local ready = gunValid and (LOAD_ENABLED or autoFireEnabled) and gun:acfReady() or false

    -- keep C-FBA in sync
    if gunValid and gun ~= LastGun and CFBA_SetGun then
        CFBA_SetGun(gun)
        LastGun = gun
    end

    local rawBasePos = getBasePos(hit, tpos)
    local manualBasePos = hit:getLength() > 0 and hit or rawBasePos
    local solvedPredPos = rawBasePos
    local solvedAimPos = rawBasePos
    local solvedTta = 0
    local solvedTargetAge = nil
    local idle = (not turretActive) and (not manual) and (not manualFire) and (not autoFireEnabled)
    local activeControl =
        turretActive and not manual and
        gunValid and
        baseValid and
        hasTrackedPos

    local loadActive = LOAD_ENABLED and turretActive and not manual and not ready
    local dynamicAim = activeControl or loadActive or manual

    if now < NextThink and not dynamicAim then
        Outputs.Fire:update((manualFire or now < LastAutoFireUntil) and 1 or 0)
        return
    end

    local shouldSolve =
        (activeControl or loadActive) and
        gunValid and
        baseValid and
        hasTrackedPos and
        CFBA_SolveIntercept

    if CFBA_ResetTargetTracking and not shouldSolve then
        CFBA_ResetTargetTracking()
    end

    if shouldSolve then
        solvedPredPos, solvedAimPos, solvedTta, solvedTargetAge = CFBA_SolveIntercept(base, tpos, tvel)
    end

    local targetPredPos = rawBasePos
    local aimPos = rawBasePos
    local tta    = 0

    if activeControl then
        targetPredPos = solvedPredPos
        aimPos = solvedAimPos
        tta = solvedTta
    end

    if loadActive then
        targetPredPos = solvedPredPos
        local loadPos = getLoadAimPos(gun, base, targetPredPos, LOAD_ANG)
        if loadPos then
            aimPos = loadPos
        end
        tta = 0
    elseif manual then
        targetPredPos = manualBasePos
        aimPos = manualBasePos
        tta = 0
    elseif not activeControl and not idle then
        if gunValid and baseValid and rawBasePos:getLength() > 0 and CFBA_SolveStatic then
            local staticPredPos, staticAimPos, staticTta = CFBA_SolveStatic(base, rawBasePos)
            aimPos = staticAimPos
            targetPredPos = staticPredPos
            tta = staticTta
        end
    end

    local autoFireNow =
        autoFireEnabled
        and activeControl
        and solvedTta > 0
        and solvedTargetAge ~= nil
        and solvedTargetAge < AUTO_FIRE_MAX_TARGET_AGE
        and shouldAutoFire(gun, base, aimPos, targetPredPos, now)

    if autoFireNow then
        LastAutoFireUntil = now + AUTO_FIRE_HOLD_TIME
    end
    local fire = (manualFire or autoFireNow or now < LastAutoFireUntil) and 1 or 0

    Outputs.TurretActive:update(turretActive and 1 or 0)
    Outputs.HAimPos:update(aimPos)
    Outputs.VAimPos:update(aimPos)
    Outputs.TargetPredPos:update(targetPredPos)
    Outputs.TTA:update(tta)
    Outputs.Elevation:update(0)
    Outputs.Fire:update(fire)

    if dynamicAim then
        NextThink = now
    elseif idle then
        NextThink = now + IDLE_INTERVAL
    else
        NextThink = now + ACTIVE_INTERVAL
    end
end

updateInputs("Lock:NORMAL Active:NORMAL TurretActiveOverride:NORMAL M1:NORMAL M2:NORMAL Gun:ENTITY Base:ENTITY HitPos:VECTOR TargetPos:VECTOR TargetVel:VECTOR AutoFire:NORMAL")
updateOutputs("TurretActive:NORMAL HAimPos:VECTOR VAimPos:VECTOR TargetPredPos:VECTOR TTA:NORMAL Elevation:NORMAL Fire:NORMAL")
syncInputCommands(true)

Inputs.Lock:addTrigger(function(v)
    if v == 0 then return end
    Unlocked = not Unlocked
end)

Inputs.Active:addTrigger(function()
    Unlocked = false
end)

Inputs.TurretActiveOverride:addTrigger(function()
    TurretOverrideWired = true
end)

hook.add("think", "SF_MGC", main)
