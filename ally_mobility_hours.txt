--[[
Tracked/wheeled ACF mobility controller.
Inputs: Engine, MainGearbox, Base, Pod wirelink.
Outputs: Throttle, LeftWeldLatch, RightWeldLatch.
GEARING_TYPE: 1 = RPM, 2 = speed, 3 = hybrid.
]]--

require("allylib/simple_server-side_wire_helper.txt")

syncInputCommands(true) -- init-only toggle; do not call per-tick
updateInputs("Engine:ENTITY MainGearbox:ENTITY Base:ENTITY Pod:WIRELINK")
updateOutputs("Throttle:NORMAL LeftWeldLatch:NORMAL RightWeldLatch:NORMAL")

EngineStateDirty = true

local function markEngineDirty()
    EngineStateDirty = true
end

Inputs.Engine:addTrigger(markEngineDirty)
Inputs.MainGearbox:addTrigger(markEngineDirty)

--- Config
SPEEDCONV = 1 / 17.6
MAX_YAW_RATE = 240
BRAKE_SPEED_GATE = 30
THINK_DT = 0.1

GEARING_TYPE          = 3
MAX_BRAKE_POWER       = 125
SCALE_BRAKES          = false
WHEEL_DIAMETER        = 28
FINAL_RATIO           = 0.18
IDLE_THROTTLE         = 40
DISABLE_NEUTRAL_STEER = false

HAS_WHEELS = false
WHEEL_COAST_WITHOUT_THROTTLE = true
INVERTED_TRANSFERS = -1
INLINE_TRANSFERS = true
AUTO_DETECT_INLINE_TRANSFERS = true

YAW_DEADZONE = 15
COUNTERSTEER_STRENGTH = 2 -- legacy P gain (kept for compatibility)
COUNTERSTEER_MIN_BRAKE = 35
COUNTERSTEER_KI = 0
COUNTERSTEER_KD = 0.05 -- small D term for easier damping tuning
ACTUAL_COUNTERSTEER_THRESHOLD = 0.5
ACTUAL_COUNTERSTEER_ENGAGEMENT = 1.0 -- 0 disables it; keep this low because 1 is a full reverse bite

INVERT_REVERSE_STEER = false
CLUTCH_START = 0.8

--- Internal Variables
CLUTCH_TURN = 1
CLUTCH_BRAKE = 1
GearCount = 0
GEAR_UP_BUFFER = 1.05
GEAR_DOWN_BUFFER = 0.95
BiteRPM, ReleaseRPM = 0, 0
PowerMin, PowerMax = 0, 0
UpshiftRPM, DownshiftRPM = 0, 0

MainWL, LeftWL, RightWL = nil, nil, nil
ActiveEngine = nil
EngineReady = false

GearSpeeds = {}
GearSlots = {}
ReverseGearSlots = {}
ReverseGearSpeeds = {}
ReverseGears = 0
IsAutogearing = false
CanNeutralSteer = false
DetectedDualClutch = false

OutputGear = 1
OutputGearSlot = 1
OutputThrottle = 0
OutputRightGear, OutputLeftGear = 1, 1
OutputRightClutch, OutputLeftClutch = CLUTCH_BRAKE, CLUTCH_BRAKE
OutputRightBrake, OutputLeftBrake = MAX_BRAKE_POWER, MAX_BRAKE_POWER
LeftTransfer, RightTransfer = nil, nil

CounterSteerI = 0
CounterSteerPrevError = 0

--- Functions
local function writeOutputs()
    Outputs.Throttle:update(OutputThrottle)
    Outputs.LeftWeldLatch:update(OutputLeftBrake > 0 and 1 or 0)
    Outputs.RightWeldLatch:update(OutputRightBrake > 0 and 1 or 0)

    if not MainWL then return end

    MainWL["Gear"] = OutputGear

    if CanNeutralSteer then
        if not LeftWL or not RightWL then return end

        LeftWL["Left Brake"] = OutputLeftBrake
        LeftWL["Right Brake"] = OutputLeftBrake
        RightWL["Left Brake"] = OutputRightBrake
        RightWL["Right Brake"] = OutputRightBrake

        LeftWL["Left Clutch"] = OutputLeftClutch
        LeftWL["Right Clutch"] = OutputLeftClutch
        RightWL["Left Clutch"] = OutputRightClutch
        RightWL["Right Clutch"] = OutputRightClutch

        if INLINE_TRANSFERS then
            LeftWL["Gear"] = OutputLeftGear
            RightWL["Gear"] = OutputRightGear
        else
            LeftWL["Left Gear"] = OutputLeftGear
            LeftWL["Right Gear"] = OutputLeftGear
            RightWL["Left Gear"] = OutputRightGear
            RightWL["Right Gear"] = OutputRightGear
        end
    elseif DetectedDualClutch then
        MainWL["Left Brake"] = OutputLeftBrake
        MainWL["Right Brake"] = OutputRightBrake
        MainWL["Left Clutch"] = OutputLeftClutch
        MainWL["Right Clutch"] = OutputRightClutch
    else
        MainWL["Brake"] = OutputLeftBrake
        MainWL["Clutch"] = OutputLeftClutch
    end
end

local function updateEngineVariables()
    EngineStateDirty = false
    MainWL, LeftWL, RightWL = nil, nil, nil
    ActiveEngine = nil
    LeftTransfer, RightTransfer = nil, nil
    CanNeutralSteer = false
    DetectedDualClutch = false
    EngineReady = false
    for i = 1, #GearSpeeds do GearSpeeds[i] = nil end
    for i = 1, #GearSlots do GearSlots[i] = nil end
    for i = 1, #ReverseGearSlots do ReverseGearSlots[i] = nil end
    for i = 1, #ReverseGearSpeeds do ReverseGearSpeeds[i] = nil end

    if not isValid(MainGearbox) then
        EngineReady = false
        return
    end

    local gearLinks = MainGearbox:acfLinks()

    if isValid(Engine) then
        ActiveEngine = Engine
    else
        local bestPowerMax = -1
        for i = 1, #gearLinks do
            local link = gearLinks[i]
            if isValid(link) then
                local class = link:getClass()
                if class ~= "prop_physics" and class:find("engine", 1, true) then
                    local powerMax = link:acfPowerbandMax()
                    if powerMax > bestPowerMax then
                        ActiveEngine = link
                        bestPowerMax = powerMax
                    end
                end
            end
        end
    end

    if not isValid(ActiveEngine) then
        return
    end

    PowerMin = ActiveEngine:acfPowerbandMin()
    PowerMax = ActiveEngine:acfPowerbandMax()

    BiteRPM = PowerMin * CLUTCH_START
    ReleaseRPM = PowerMin

    UpshiftRPM = PowerMax * 0.95
    DownshiftRPM = PowerMin * 0.85

    local totalGears = MainGearbox:acfNumGears()
    local useReciprocalRatios = false

    for i = 1, totalGears do
        local ratio = MainGearbox:acfGearRatio(i)
        if ratio ~= 0 and math.abs(ratio) > 1 then
            useReciprocalRatios = true
            break
        end
    end

    -- Driveline speed: wheel RPM = engine RPM / overall reduction, speed = wheel RPM * tire circumference.
    -- ACF2-style ratios are already inverse reductions; ACF3-style ratios need to be inverted first.
    local commonVal = (PowerMax * FINAL_RATIO / 60) * WHEEL_DIAMETER * math.pi

    GearCount = 0
    ReverseGears = 0

    for i = 1, totalGears do
        local ratio = MainGearbox:acfGearRatio(i)
        if ratio ~= 0 then
            local ratioAbs = math.abs(ratio)
            local speedAtPower = math.abs(commonVal * (useReciprocalRatios and (1 / ratioAbs) or ratioAbs) * SPEEDCONV)

            if ratio < 0 then
                local slot = ReverseGears + 1
                while slot > 1 and ReverseGearSpeeds[slot - 1] > speedAtPower do
                    ReverseGearSlots[slot] = ReverseGearSlots[slot - 1]
                    ReverseGearSpeeds[slot] = ReverseGearSpeeds[slot - 1]
                    slot = slot - 1
                end
                ReverseGearSlots[slot] = i
                ReverseGearSpeeds[slot] = speedAtPower
                ReverseGears = ReverseGears + 1
            else
                local slot = GearCount + 1
                while slot > 1 and GearSpeeds[slot - 1] > speedAtPower do
                    GearSlots[slot] = GearSlots[slot - 1]
                    GearSpeeds[slot] = GearSpeeds[slot - 1]
                    slot = slot - 1
                end
                GearSlots[slot] = i
                GearSpeeds[slot] = speedAtPower
                GearCount = GearCount + 1
            end
        end
    end

    for i = 1, ReverseGears do
        GearSpeeds[GearCount + i] = ReverseGearSpeeds[i]
    end

    OutputGearSlot = math.clamp(OutputGearSlot, 1, math.max(GearCount + ReverseGears, 1))
    OutputGear = (OutputGearSlot <= GearCount and GearSlots[OutputGearSlot])
        or ReverseGearSlots[OutputGearSlot - GearCount]
        or GearSlots[1]
        or 1
    IsAutogearing = GearCount > 2

    local transfers = 0
    for i = 1, #gearLinks do
        local link = gearLinks[i]
        if not isValid(link) or link:getClass() == "prop_physics" then continue end

        local type = link:acfType()
        if type == "Transfer Case" or type == "Transfer" then
            transfers = transfers + 1
            local y = MainGearbox:worldToLocal(link:getPos()).y
            if y * INVERTED_TRANSFERS < 0 then RightTransfer = link end
            if y * INVERTED_TRANSFERS > 0 then LeftTransfer = link end
        end
    end

    MainWL = MainGearbox:getWirelink()
    if isValid(LeftTransfer) then
        LeftWL = LeftTransfer:getWirelink()
    end
    if isValid(RightTransfer) then
        RightWL = RightTransfer:getWirelink()
    end

    if AUTO_DETECT_INLINE_TRANSFERS and LeftWL and RightWL then
        -- Inline transfer cases use one "Gear" input. Dual-side boxes expose "Left Gear" and "Right Gear".
        if pcall(function() LeftWL["Gear"] = 1 end) and pcall(function() RightWL["Gear"] = 1 end) then
            INLINE_TRANSFERS = true
        elseif pcall(function() LeftWL["Left Gear"] = 1 end) and pcall(function() LeftWL["Right Gear"] = 1 end)
            and pcall(function() RightWL["Left Gear"] = 1 end) and pcall(function() RightWL["Right Gear"] = 1 end) then
            INLINE_TRANSFERS = false
        end
    end

    CanNeutralSteer = transfers >= 2 and LeftWL ~= nil and RightWL ~= nil

    if not CanNeutralSteer and MainWL then
        DetectedDualClutch = pcall(function() MainWL["Left Brake"] = MAX_BRAKE_POWER end)
            and pcall(function() MainWL["Right Brake"] = MAX_BRAKE_POWER end)
            and pcall(function() MainWL["Left Clutch"] = CLUTCH_BRAKE end)
            and pcall(function() MainWL["Right Clutch"] = CLUTCH_BRAKE end)
    end

    EngineReady = MainWL ~= nil and isValid(ActiveEngine) and (transfers < 2 or CanNeutralSteer)
end

PrevEngine = nil
PrevMainGearbox = nil

local function think()
    -- 1) Refresh cached ACF/wirelink data only when entities change or a cached entity disappears.
    local needsRefresh = EngineStateDirty or Engine ~= PrevEngine or MainGearbox ~= PrevMainGearbox
    if not needsRefresh then
        needsRefresh = not isValid(MainGearbox) or not isValid(ActiveEngine) or MainWL == nil or not EngineReady
        if not needsRefresh and CanNeutralSteer then
            needsRefresh = not isValid(LeftTransfer) or not isValid(RightTransfer) or LeftWL == nil or RightWL == nil
        end
    end

    if needsRefresh then
        updateEngineVariables()
        PrevEngine = Engine
        PrevMainGearbox = MainGearbox
    end

    -- 2) Fail closed: no throttle, first gear, full brake/clutch.
    if not MainGearbox or not Base or not Pod or not isValid(MainGearbox) or not isValid(Base) or not EngineReady then
        OutputGearSlot = 1
        OutputGear = (GearSlots[1] or ReverseGearSlots[1] or 1)
        OutputLeftGear = 1
        OutputRightGear = 1
        OutputThrottle = 0
        OutputLeftBrake = MAX_BRAKE_POWER
        OutputRightBrake = MAX_BRAKE_POWER
        OutputLeftClutch = CLUTCH_BRAKE
        OutputRightClutch = CLUTCH_BRAKE
        CounterSteerI = 0
        CounterSteerPrevError = 0
        writeOutputs()
        return
    end

    -- 3) Read driver intent and vehicle state.
    local vel = Base:getVelocity()
    local speed = math.sqrt(vel.x * vel.x + vel.y * vel.y) * SPEEDCONV
    local rpm = ActiveEngine:acfRPM()
    local driveDir = (Pod["W"] or 0) - (Pod["S"] or 0)
    local steerSign = INVERTED_TRANSFERS < 0 and 1 or -1
    local steerDir = ((Pod["D"] or 0) - (Pod["A"] or 0)) * steerSign
    local neutralSteerEnabled = CanNeutralSteer and not HAS_WHEELS and not DISABLE_NEUTRAL_STEER
    local brakeSteerEnabled = not HAS_WHEELS and not neutralSteerEnabled
    local hasDriveInput = driveDir ~= 0
    local hasSteerInput = steerDir ~= 0
    local steerOnlyDrive = brakeSteerEnabled and not hasDriveInput and hasSteerInput
    if steerOnlyDrive then
        driveDir = 1
    end
    local hasInput = hasDriveInput or (neutralSteerEnabled and hasSteerInput) or steerOnlyDrive

    -- 4) Start each tick from the plain "hold still" output state, then loosen it as needed.
    OutputLeftGear = 1
    OutputRightGear = 1

    if not hasInput and speed < 1 and rpm < BiteRPM then
        OutputThrottle = IDLE_THROTTLE
        OutputLeftBrake = MAX_BRAKE_POWER
        OutputRightBrake = MAX_BRAKE_POWER
        OutputLeftClutch = CLUTCH_BRAKE
        OutputRightClutch = CLUTCH_BRAKE
        writeOutputs()
        return
    end

    local brakePower = MAX_BRAKE_POWER
    if SCALE_BRAKES then
        if IsAutogearing and GearSpeeds[GearCount] and speed > 1 then
            brakePower = MAX_BRAKE_POWER * math.min(GearSpeeds[GearCount] / 2 / speed, 1)
        elseif speed > 1 then
            brakePower = MAX_BRAKE_POWER * math.min(BRAKE_SPEED_GATE / speed, 1)
        end
    end

    local baseBrakePower = brakePower
    if HAS_WHEELS and WHEEL_COAST_WITHOUT_THROTTLE and not hasDriveInput and speed > 1 then
        baseBrakePower = 0
    end

    OutputLeftBrake = baseBrakePower
    OutputRightBrake = baseBrakePower
    OutputLeftClutch = CLUTCH_BRAKE
    OutputRightClutch = CLUTCH_BRAKE

    if hasInput then
        OutputThrottle = 100
        local clutchRelease = 1 - math.clamp((rpm - BiteRPM) / (ReleaseRPM - BiteRPM), 0, 1)
        OutputLeftBrake = 0
        OutputRightBrake = 0
        OutputLeftClutch = clutchRelease
        OutputRightClutch = clutchRelease
    else
        OutputThrottle = IDLE_THROTTLE
    end

    -- 5) Choose a logical gear slot, then map that slot to the real ACF gear number.
    -- GearSpeeds[slot] is the theoretical speed at PowerMax. So:
    -- predicted RPM in a candidate slot = PowerMax * current speed / slot speed.
    if driveDir == 0 then
        if OutputGearSlot > GearCount then
            driveDir = -1
        else
            driveDir = 1
        end
    end

    if not IsAutogearing then
        if neutralSteerEnabled then
            OutputGearSlot = 1
        elseif driveDir < 0 and ReverseGears > 0 then
            OutputGearSlot = GearCount + 1
        else
            OutputGearSlot = 1
        end
    else
        local minSlot = 1
        local maxSlot = GearCount

        if driveDir < 0 and not neutralSteerEnabled and ReverseGears > 0 then
            minSlot = GearCount + 1
            maxSlot = GearCount + ReverseGears
        end

        if OutputGearSlot < minSlot or OutputGearSlot > maxSlot then
            OutputGearSlot = minSlot
        end

        local slotSpeed = GearSpeeds[OutputGearSlot] or 0
        local prevSlotSpeed = GearSpeeds[OutputGearSlot - 1] or 0
        local nextSlotSpeed = GearSpeeds[OutputGearSlot + 1] or 0
        local predictedRPM = slotSpeed > 0 and PowerMax * speed / slotSpeed or rpm
        local canUpshift = OutputGearSlot < maxSlot and (nextSlotSpeed == 0 or PowerMax * speed / nextSlotSpeed > DownshiftRPM)
        local canDownshift = OutputGearSlot > minSlot and (prevSlotSpeed == 0 or PowerMax * speed / prevSlotSpeed < UpshiftRPM)

        if GEARING_TYPE == 1 then
            if rpm > UpshiftRPM and canUpshift then
                OutputGearSlot = OutputGearSlot + 1
            elseif rpm < DownshiftRPM and canDownshift then
                OutputGearSlot = OutputGearSlot - 1
            end
        elseif GEARING_TYPE == 2 then
            if canUpshift and speed > slotSpeed * GEAR_UP_BUFFER then
                OutputGearSlot = OutputGearSlot + 1
            elseif canDownshift and speed < prevSlotSpeed * GEAR_DOWN_BUFFER then
                OutputGearSlot = OutputGearSlot - 1
            end
        elseif GEARING_TYPE == 3 then
            if canUpshift and (speed > slotSpeed * GEAR_UP_BUFFER or rpm > UpshiftRPM) then
                OutputGearSlot = OutputGearSlot + 1
            elseif canDownshift and predictedRPM < DownshiftRPM then
                OutputGearSlot = OutputGearSlot - 1
            end
        end
    end

    OutputGear = (OutputGearSlot <= GearCount and GearSlots[OutputGearSlot])
        or ReverseGearSlots[OutputGearSlot - GearCount]
        or GearSlots[1]
        or 1

    -- 6) Steering: tracked builds brake/clutch one side; split transfers can also reverse a side.
    if neutralSteerEnabled then
        if not hasDriveInput and hasSteerInput then
            if steerDir < 0 then
                OutputLeftGear = 2
                OutputRightGear = 1
            else
                OutputLeftGear = 1
                OutputRightGear = 2
            end
        else
            if driveDir < 0 then
                OutputLeftGear = 2
                OutputRightGear = 2
            end

            if steerDir < 0 then
                OutputLeftBrake = brakePower
                OutputRightBrake = 0
                OutputLeftClutch = CLUTCH_TURN
            elseif steerDir > 0 then
                OutputLeftBrake = 0
                OutputRightBrake = brakePower
                OutputRightClutch = CLUTCH_TURN
            end
        end
    elseif brakeSteerEnabled then
        if CanNeutralSteer and driveDir < 0 and ReverseGears == 0 then
            OutputLeftGear = 2
            OutputRightGear = 2
        end

        if hasSteerInput and (hasDriveInput or speed > 1 or steerOnlyDrive) and
            (CanNeutralSteer or DetectedDualClutch) then
            local turnDir = steerDir
            if driveDir < 0 and INVERT_REVERSE_STEER then
                turnDir = -turnDir
            end

            if turnDir > 0 then
                OutputRightBrake = math.max(OutputRightBrake, brakePower)
                OutputRightClutch = math.max(OutputRightClutch, CLUTCH_TURN)
            else
                OutputLeftBrake = math.max(OutputLeftBrake, brakePower)
                OutputLeftClutch = math.max(OutputLeftClutch, CLUTCH_TURN)
            end
        end
    end

    -- 7) Straight-line yaw correction for tracked vehicles only.
    if not HAS_WHEELS and hasDriveInput and not hasSteerInput then
        local yawRate = Base:getAngleVelocity().z
        local yawAbs = math.abs(yawRate)

        if yawAbs > YAW_DEADZONE then
            local err = yawAbs
            CounterSteerI = CounterSteerI + err * THINK_DT

            local pidNorm = math.min(math.max(
                (err / MAX_YAW_RATE) * COUNTERSTEER_STRENGTH
                + (CounterSteerI / MAX_YAW_RATE) * COUNTERSTEER_KI
                + ((err - CounterSteerPrevError) / THINK_DT / MAX_YAW_RATE) * COUNTERSTEER_KD,
                0
            ), 1)
            local actualAssist = 0
            if CanNeutralSteer and ACTUAL_COUNTERSTEER_ENGAGEMENT > 0 and ACTUAL_COUNTERSTEER_THRESHOLD < 1 then
                actualAssist = math.clamp((pidNorm - ACTUAL_COUNTERSTEER_THRESHOLD) / (1 - ACTUAL_COUNTERSTEER_THRESHOLD), 0, 1)
            end

            local counterSteerWeight = math.min(math.max(COUNTERSTEER_MIN_BRAKE, pidNorm * brakePower), brakePower) * (1 - actualAssist)

            CounterSteerPrevError = err

            local direction = (driveDir ~= 0 and driveDir or 1) * steerSign

            if yawRate * direction < 0 then
                OutputLeftBrake = math.max(OutputLeftBrake, counterSteerWeight)
                if actualAssist > 0 then
                    OutputLeftGear = OutputLeftGear == 1 and 2 or 1
                    OutputLeftClutch = math.max(OutputLeftClutch, 1 - actualAssist * math.clamp(ACTUAL_COUNTERSTEER_ENGAGEMENT, 0, 1))
                else
                    OutputLeftClutch = math.max(OutputLeftClutch, CLUTCH_TURN)
                end
            else
                OutputRightBrake = math.max(OutputRightBrake, counterSteerWeight)
                if actualAssist > 0 then
                    OutputRightGear = OutputRightGear == 1 and 2 or 1
                    OutputRightClutch = math.max(OutputRightClutch, 1 - actualAssist * math.clamp(ACTUAL_COUNTERSTEER_ENGAGEMENT, 0, 1))
                else
                    OutputRightClutch = math.max(OutputRightClutch, CLUTCH_TURN)
                end
            end
        else
            CounterSteerI = 0
            CounterSteerPrevError = 0
        end
    else
        CounterSteerI = 0
        CounterSteerPrevError = 0
    end

    -- 8) One write point at the end of the loop keeps wire side effects predictable.
    writeOutputs()
end

--- Hooks
timer.create("AMH_mainLogicLoop", THINK_DT, 0, think)
