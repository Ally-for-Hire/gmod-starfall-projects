/*
--- Overview
Plug-and-play mobility brain for tracked or wheeled ACF vehicles. It manages autogearing, clutch modulation, braking, and counter-steer logic so you can concentrate on driving instead of spinning out.

--- Wiring
-- Inputs
1. "MainGearbox" -> main gearbox (typically the largest gearbox on the vehicle).
2. "Engine" -> the engine you want monitored; pick the one with the largest powerband if you have multiple different types.
3. "Base" -> vehicle baseplate entity.
4. "Pod" -> pod controller
-- Outputs
1. "Throttle" <- all of your engines' throttle inputs.

Once those links are set, the chip begins driving the gearbox and throttle outputs automatically.

--- Customizing
Every tunable sits in the "Customizable Variables" block and includes inline descriptions.

Key notes:
- "GEARING_TYPE": "1" uses RPM-based shifts, "2" uses speed gates, and "3" blends both. Hybrid is the recommended default.
- "WHEEL_DIAMETER" and "FINAL_RATIO" should match your physical wheel diameter and combined gearbox ratios so the shift map stays accurate. Example: (1/1.5) * (1/1.4).
- Flip "showDebug" on to review gear-speed targets; nudge "WHEEL_DIAMETER" if the speeds feel off.
- Counter-steer defaults are tuned for general use. Only tweak them for exceptionally unusual drivetrains.
- "clutchStart" comes from Cheezus' clutch curve; leave it at "0.8" unless you have a specific low-RPM behaviour to chase.

Need a hand tailoring it to a particular rig? Ping @allyfh and I'll walk you through it.
*/

require("allylib/simple_server-side_wire_helper.txt")

syncInputCommands(true) -- init-only toggle; do not call per-tick
updateInputs("Engine:ENTITY MainGearbox:ENTITY Base:ENTITY Pod:WIRELINK")
updateOutputs("Throttle:NORMAL LeftWeldLatch:NORMAL RightWeldLatch:NORMAL")

--- Constants
SPEEDCONV = 1 / 17.6
MAX_YAW_RATE = 240
BRAKE_SPEED_GATE = 30
THINK_DT = 0.05

--- Customizable Variables
-- Gearing and Braking Setup
GEARING_TYPE          = 3
MAX_BRAKE_POWER       = 100
SCALE_BRAKES          = false
WHEEL_DIAMETER        = 28
FINAL_RATIO           = 0.18
IDLE_THROTTLE         = 20
DISABLE_NEUTRAL_STEER = false

-- Vehicle Setup 
HAS_WHEELS = false
DUAL_CLUTCH = false
INVERTED_TRANSFERS = 1
INLINE_TRANSFERS = true

-- Countersteering Setup
YAW_DEADZONE = 15
COUNTERSTEER_STRENGTH = 1.5 -- legacy P gain (kept for compatibility)
COUNTERSTEER_KI = 0
COUNTERSTEER_KD = 0.05 -- small D term for easier damping tuning

-- Steering Setup
INVERT_STEER = false
INVERT_REVERSE_STEER = false

-- Misc
SHOW_DEBUG = false
CLUTCH_START = 0.8

--- Internal Variables
CLUTCH_TURN = 1
CLUTCH_BRAKE = 1
GearCount = 0
GEAR_UP_BUFFER = 1.05
GEAR_DOWN_BUFFER = 0.95
BiteRPM, ReleaseRPM = 0, 0
PowerMin, PowerMax = 0, 0
UpshiftRPM, DownshiftRPM = 0, 0

MainWL, LeftWL, RightWL = nil, nil, nil
ActiveEngine = nil
EngineReady = false

GearLinks, GearSpeeds = {}, {}
ReverseGears = 0
IsAutogearing = false
HasSplitTransfers = false
CanNeutralSteer = false
Acf3GearRatios = true

OutputGear = 1
OutputThrottle = 0
OutputRightGear, OutputLeftGear = 1, 1
OutputRightClutch, OutputLeftClutch = CLUTCH_BRAKE, CLUTCH_BRAKE
OutputRightBrake, OutputLeftBrake = MAX_BRAKE_POWER, MAX_BRAKE_POWER
LeftTransfer, RightTransfer = nil, nil

LastGear, LastThrottle = nil, nil
LastLBrake, LastRBrake = nil, nil
LastLClutch, LastRClutch = nil, nil
LastLGear, LastRGear = nil, nil
LastLWeld, LastRWeld = nil, nil

CounterSteerI = 0
CounterSteerPrevError = 0

--- Functions
local applyOutputs

local function applyBrakeClutch(LBrake, RBrake, LClutch, RClutch)
    OutputRightClutch = RClutch
    OutputLeftClutch = LClutch
    OutputRightBrake = RBrake
    OutputLeftBrake = LBrake
end

local function writeWireIfChanged(wl, key, value, lastValue)
    if not wl then return lastValue end
    if lastValue ~= value then
        wl[key] = value
        return value
    end
    return lastValue
end

local function writeWirePairIfChanged(wl, keyA, keyB, value, lastValue)
    if not wl then return lastValue end
    if lastValue ~= value then
        wl[keyA] = value
        wl[keyB] = value
        return value
    end
    return lastValue
end

local function writePortIfChanged(key, value, lastValue)
    if lastValue ~= value then
        local out = Outputs[key]
        if out then
            out:update(value)
        else
            wire.ports[key] = value
        end
        return value
    end
    return lastValue
end

local function resetOutputCache()
    LastGear, LastThrottle = nil, nil
    LastLBrake, LastRBrake = nil, nil
    LastLClutch, LastRClutch = nil, nil
    LastLGear, LastRGear = nil, nil
    LastLWeld, LastRWeld = nil, nil
end

local function resetCounterSteer()
    CounterSteerI = 0
    CounterSteerPrevError = 0
end

local function applySafeState()
    OutputGear = 1
    OutputLeftGear = 1
    OutputRightGear = 1
    OutputThrottle = 0
    applyBrakeClutch(MAX_BRAKE_POWER, MAX_BRAKE_POWER, CLUTCH_BRAKE, CLUTCH_BRAKE)
    resetCounterSteer()
    applyOutputs()
end

local function findLinkedEngine(links)
    local bestEngine = nil
    local bestPowerMax = -1

    for i = 1, #links do
        local link = links[i]
        if isValid(link) then
            local class = link:getClass()
            if class ~= "prop_physics" and class:find("engine", 1, true) then
                local powerMax = link:acfPowerbandMax()
                if powerMax > bestPowerMax then
                    bestEngine = link
                    bestPowerMax = powerMax
                end
            end
        end
    end

    return bestEngine
end

local function selectGear(rpm, speed, driveDir, neutralSteerEnabled)
    if not IsAutogearing then
        if neutralSteerEnabled then
            return 1
        end
        if driveDir < 0 and ReverseGears > 0 then
            return GearCount + 1
        end
        return 1
    end

    local minGear = 1
    local maxGear = GearCount

    if driveDir < 0 and not neutralSteerEnabled and ReverseGears > 0 then
        minGear = GearCount + 1
        maxGear = GearCount + ReverseGears
    end

    local newGear = OutputGear
    if newGear < minGear or newGear > maxGear then
        newGear = minGear
    end

    if GEARING_TYPE == 1 then
        if rpm > UpshiftRPM and newGear < maxGear then
            newGear = newGear + 1
        elseif rpm < DownshiftRPM and newGear > minGear then
            newGear = newGear - 1
        end
    elseif GEARING_TYPE == 2 then
        if newGear < maxGear and speed > GearSpeeds[newGear] * GEAR_UP_BUFFER then
            newGear = newGear + 1
        elseif newGear > minGear and speed < GearSpeeds[newGear - 1] * GEAR_DOWN_BUFFER then
            newGear = newGear - 1
        end
    elseif GEARING_TYPE == 3 then
        if newGear < maxGear and (speed > GearSpeeds[newGear] * GEAR_UP_BUFFER or rpm > UpshiftRPM) then
            newGear = newGear + 1
        elseif newGear > minGear and (speed < GearSpeeds[newGear - 1] * GEAR_DOWN_BUFFER and rpm < PowerMin) then
            newGear = newGear - 1
        end
    end

    return newGear
end

function applyOutputs()
    LastThrottle = writePortIfChanged("Throttle", OutputThrottle, LastThrottle)

    local lweld = OutputLeftBrake > 0 and 1 or 0
    local rweld = OutputRightBrake > 0 and 1 or 0

    LastLWeld = writePortIfChanged("LeftWeldLatch", lweld, LastLWeld)
    LastRWeld = writePortIfChanged("RightWeldLatch", rweld, LastRWeld)

    if not MainWL then return end

    LastGear = writeWireIfChanged(MainWL, "Gear", OutputGear, LastGear)

    if CanNeutralSteer then
        if not LeftWL or not RightWL then return end

        LastLBrake = writeWirePairIfChanged(LeftWL, "Left Brake", "Right Brake", OutputLeftBrake, LastLBrake)
        LastRBrake = writeWirePairIfChanged(RightWL, "Left Brake", "Right Brake", OutputRightBrake, LastRBrake)

        LastLClutch = writeWirePairIfChanged(LeftWL, "Left Clutch", "Right Clutch", OutputLeftClutch, LastLClutch)
        LastRClutch = writeWirePairIfChanged(RightWL, "Left Clutch", "Right Clutch", OutputRightClutch, LastRClutch)

        if INLINE_TRANSFERS then
            LastLGear = writeWireIfChanged(LeftWL, "Gear", OutputLeftGear, LastLGear)
            LastRGear = writeWireIfChanged(RightWL, "Gear", OutputRightGear, LastRGear)
        else
            LastLGear = writeWirePairIfChanged(LeftWL, "Left Gear", "Right Gear", OutputLeftGear, LastLGear)
            LastRGear = writeWirePairIfChanged(RightWL, "Left Gear", "Right Gear", OutputRightGear, LastRGear)
        end
    elseif DUAL_CLUTCH then
        LastLBrake = writeWireIfChanged(MainWL, "Left Brake", OutputLeftBrake, LastLBrake)
        LastRBrake = writeWireIfChanged(MainWL, "Right Brake", OutputRightBrake, LastRBrake)

        LastLClutch = writeWireIfChanged(MainWL, "Left Clutch", OutputLeftClutch, LastLClutch)
        LastRClutch = writeWireIfChanged(MainWL, "Right Clutch", OutputRightClutch, LastRClutch)
    else
        LastLBrake = writeWireIfChanged(MainWL, "Brake", OutputLeftBrake, LastLBrake)
        LastLClutch = writeWireIfChanged(MainWL, "Clutch", OutputLeftClutch, LastLClutch)
    end
end

local function updateEngineVariables()
    MainWL, LeftWL, RightWL = nil, nil, nil
    ActiveEngine = nil
    LeftTransfer, RightTransfer = nil, nil
    CanNeutralSteer = false
    HasSplitTransfers = false
    EngineReady = false
    GearLinks, GearSpeeds = {}, {}
    resetOutputCache()

    if not isValid(MainGearbox) then
        EngineReady = false
        return
    end

    GearLinks = MainGearbox:acfLinks()

    if isValid(Engine) then
        ActiveEngine = Engine
    else
        ActiveEngine = findLinkedEngine(GearLinks)
    end

    if not isValid(ActiveEngine) then
        return
    end

    PowerMin = ActiveEngine:acfPowerbandMin()
    PowerMax = ActiveEngine:acfPowerbandMax()

    BiteRPM = PowerMin * CLUTCH_START
    ReleaseRPM = PowerMin

    UpshiftRPM = PowerMax * 0.95
    DownshiftRPM = PowerMin * 0.85

    GearCount = MainGearbox:acfNumGears()
    ReverseGears = 0
    for i = 1, GearCount do
        local ratio = MainGearbox:acfGearRatio(i)
        if ratio < 0 then
            ReverseGears = ReverseGears + 1
        end
    end
    GearCount = GearCount - ReverseGears
    IsAutogearing = GearCount > 2

    Acf3GearRatios = math.abs(MainGearbox:acfGearRatio(1)) > 1
    local commonVal = (PowerMax * FINAL_RATIO / 60) * WHEEL_DIAMETER * math.pi

    if IsAutogearing then
        if SHOW_DEBUG then print("Gear Speeds:") end
        if Acf3GearRatios then
            for i = 1, GearCount + ReverseGears do
                GearSpeeds[i] = math.abs(commonVal * (1 / MainGearbox:acfGearRatio(i)) * SPEEDCONV)
                if SHOW_DEBUG then print("    " .. i .. " .. " .. GearSpeeds[i]) end
            end
        else
            for i = 1, GearCount + ReverseGears do
                GearSpeeds[i] = math.abs(commonVal * (MainGearbox:acfGearRatio(i)) * SPEEDCONV)
                if SHOW_DEBUG then print("    " .. i .. " .. " .. GearSpeeds[i]) end
            end
        end
    end

    local transfers = 0
    for i = 1, #GearLinks do
        local link = GearLinks[i]
        if not isValid(link) or link:getClass() == "prop_physics" then continue end

        local type = link:acfType()
        if type == "Transfer Case" or type == "Transfer" then
            transfers = transfers + 1
            local y = MainGearbox:worldToLocal(link:getPos()).y
            if y * INVERTED_TRANSFERS < 0 then RightTransfer = link end
            if y * INVERTED_TRANSFERS > 0 then LeftTransfer = link end
        end
    end

    MainWL = MainGearbox:getWirelink()
    if isValid(LeftTransfer) then
        LeftWL = LeftTransfer:getWirelink()
    end
    if isValid(RightTransfer) then
        RightWL = RightTransfer:getWirelink()
    end

    HasSplitTransfers = transfers >= 2
    CanNeutralSteer = HasSplitTransfers and LeftWL ~= nil and RightWL ~= nil
    EngineReady = MainWL ~= nil and isValid(ActiveEngine) and (not HasSplitTransfers or CanNeutralSteer)

    if SHOW_DEBUG then
        if Acf3GearRatios and IsAutogearing then print("This Vehicle Uses ACF3 Gears") elseif IsAutogearing then print("This vehicle Uses ACE Gears") end
        print("Detected Forward Gears .. " .. GearCount)
        print("Detected Reverse Gears .. " .. ReverseGears)
        print("Detected Transfers .. " .. transfers)
        if isValid(ActiveEngine) then print("Active Engine .. " .. ActiveEngine:getClass()) end
        if IsAutogearing then print("Autogearing Enabled") else print("Autogearing Disabled") end
        if CanNeutralSteer then print("The Vehicle Can Neutral Steer") else print("The Vehicle Cannot Neutral Steer") end
    end
end

PrevEngine = nil
PrevMainGearbox = nil

local function think()
    local needsRefresh = Engine ~= PrevEngine or MainGearbox ~= PrevMainGearbox
    if not needsRefresh then
        needsRefresh = not isValid(MainGearbox) or not isValid(ActiveEngine) or MainWL == nil or not EngineReady
        if not needsRefresh and CanNeutralSteer then
            needsRefresh = not isValid(LeftTransfer) or not isValid(RightTransfer) or LeftWL == nil or RightWL == nil
        end
    end

    if needsRefresh then
        updateEngineVariables()
        PrevEngine = Engine
        PrevMainGearbox = MainGearbox
    end

    if not MainGearbox or not Base or not Pod then
        applySafeState()
        return
    end

    if not isValid(MainGearbox) or not isValid(Base) or not EngineReady then
        applySafeState()
        return
    end

    local vel = Base:getVelocity()
    local speed = math.sqrt(vel.x * vel.x + vel.y * vel.y) * SPEEDCONV
    local rpm = ActiveEngine:acfRPM()
    local W = Pod["W"] or 0
    local A = Pod["A"] or 0
    local S = Pod["S"] or 0
    local D = Pod["D"] or 0
    local driveDir = W - S
    local steerDir = D - A
    if INVERT_STEER then
        steerDir = -steerDir
    end
    local neutralSteerEnabled = CanNeutralSteer and not HAS_WHEELS and not DISABLE_NEUTRAL_STEER
    local nonNeutralSteerEnabled = not HAS_WHEELS and not neutralSteerEnabled
    local hasDriveInput = driveDir ~= 0
    local hasSteerInput = steerDir ~= 0
    local steerOnlyDrive = nonNeutralSteerEnabled and not hasDriveInput and hasSteerInput
    if steerOnlyDrive then
        driveDir = 1
    end
    local hasInput = hasDriveInput or (neutralSteerEnabled and hasSteerInput) or steerOnlyDrive

    OutputLeftGear = 1
    OutputRightGear = 1

    if not hasInput and speed < 1 and rpm < BiteRPM then
        OutputThrottle = IDLE_THROTTLE
        applyBrakeClutch(MAX_BRAKE_POWER, MAX_BRAKE_POWER, CLUTCH_BRAKE, CLUTCH_BRAKE)
        applyOutputs()
        return
    end

    local brakePower = MAX_BRAKE_POWER
    if SCALE_BRAKES then
        if IsAutogearing and GearSpeeds[GearCount] and speed > 1 then
            brakePower = MAX_BRAKE_POWER * math.min(GearSpeeds[GearCount] / 2 / speed, 1)
        elseif speed > 1 then
            brakePower = MAX_BRAKE_POWER * math.min(BRAKE_SPEED_GATE / speed, 1)
        end
    end

    applyBrakeClutch(brakePower, brakePower, CLUTCH_BRAKE, CLUTCH_BRAKE)

    if hasInput then
        OutputThrottle = 100
        local clutchRelease = 1 - math.clamp((rpm - BiteRPM) / (ReleaseRPM - BiteRPM), 0, 1)
        applyBrakeClutch(0, 0, clutchRelease, clutchRelease)
    else
        OutputThrottle = IDLE_THROTTLE
    end

    if driveDir == 0 then
        if OutputGear > GearCount then
            driveDir = -1
        else
            driveDir = 1
        end
    end

    OutputGear = selectGear(rpm, speed, driveDir, neutralSteerEnabled)

    if neutralSteerEnabled then
        if not hasDriveInput and hasSteerInput then
            if steerDir < 0 then
                OutputLeftGear = 2
                OutputRightGear = 1
            else
                OutputLeftGear = 1
                OutputRightGear = 2
            end
        else
            if driveDir < 0 then
                OutputLeftGear = 2
                OutputRightGear = 2
            end

            if steerDir < 0 then
                applyBrakeClutch(brakePower, 0, CLUTCH_TURN, OutputRightClutch)
            elseif steerDir > 0 then
                applyBrakeClutch(0, brakePower, OutputLeftClutch, CLUTCH_TURN)
            end
        end
    elseif nonNeutralSteerEnabled then
        if CanNeutralSteer and driveDir < 0 and ReverseGears == 0 then
            OutputLeftGear = 2
            OutputRightGear = 2
        end

        if hasSteerInput and (hasDriveInput or speed > 1 or steerOnlyDrive) and (CanNeutralSteer or DUAL_CLUTCH) then
            local turnDir = steerDir
            if driveDir < 0 and INVERT_REVERSE_STEER then
                turnDir = -turnDir
            end

            if turnDir > 0 then
                OutputRightBrake = math.max(OutputRightBrake, brakePower)
                OutputRightClutch = math.max(OutputRightClutch, CLUTCH_TURN)
            else
                OutputLeftBrake = math.max(OutputLeftBrake, brakePower)
                OutputLeftClutch = math.max(OutputLeftClutch, CLUTCH_TURN)
            end
        end
    end

    if not HAS_WHEELS and hasDriveInput and not hasSteerInput then
        local yawRate = Base:getAngleVelocity().z
        local yawAbs = math.abs(yawRate)

        if yawAbs > YAW_DEADZONE then
            local err = yawAbs
            CounterSteerI = CounterSteerI + err * THINK_DT

            local p = (err / MAX_YAW_RATE) * COUNTERSTEER_STRENGTH
            local i = (CounterSteerI / MAX_YAW_RATE) * COUNTERSTEER_KI
            local d = ((err - CounterSteerPrevError) / THINK_DT / MAX_YAW_RATE) * COUNTERSTEER_KD

            local pidNorm = math.min(math.max(p + i + d, 0), 1)
            local counterSteerWeight = pidNorm * brakePower

            CounterSteerPrevError = err

            local direction = driveDir ~= 0 and driveDir or 1
            if INVERT_STEER then
                direction = -direction
            end

            if yawRate * direction < 0 then
                OutputLeftBrake = math.max(OutputLeftBrake, counterSteerWeight)
                OutputLeftClutch = math.max(OutputLeftClutch, CLUTCH_TURN)
            else
                OutputRightBrake = math.max(OutputRightBrake, counterSteerWeight)
                OutputRightClutch = math.max(OutputRightClutch, CLUTCH_TURN)
            end
        else
            resetCounterSteer()
        end
    else
        resetCounterSteer()
    end

    applyOutputs()
end

--- Hooks
timer.create("Main Logic Loop", THINK_DT, 0, think)
