--@include allylib/simple_server-side_wire_helper.txt
--@server

require("allylib/simple_server-side_wire_helper.txt")

updateInputs("Sensors:ARRAY")
updateOutputs("Position:ARRAY Velocity:ARRAY Owner:ARRAY IDs:ARRAY Times:ARRAY")

POLL_RATE = 0.02
PUBLISH_RATE = 0.05
CONTACT_TTL = 0.15
PRUNE_INTERVAL = 0.5
TIMER_NAME = "MerydianSensorSuite_Poll"
ZERO_VEC = Vector()

CachedRadars = {}
Contacts = {}
OutputsActive = false
OutputsInitialized = false
NextPrune = 0
NextPublish = 0
SnapshotDirty = true

local pollSensors

local function isRadar(ent)
    if not isValid(ent) then return false end

    local class = ent:getClass()
    return class == "ace_searchradar" or class == "ace_trackingradar" or class == "acf_radar"
end

local function clearOutputs()
    if OutputsInitialized and not OutputsActive then return end

    Outputs.Position:update({})
    Outputs.Velocity:update({})
    Outputs.Owner:update({})
    Outputs.IDs:update({})
    Outputs.Times:update({})
    OutputsActive = false
    OutputsInitialized = true
    SnapshotDirty = false
end

local function schedulePoll()
    timer.remove(TIMER_NAME)

    if #CachedRadars == 0 then
        clearOutputs()
        return
    end

    timer.create(TIMER_NAME, POLL_RATE, 1, pollSensors)
end

local function cacheSensors(arr)
    local nextRadars = {}

    if type(arr) == "table" then
        for i = 1, #arr do
            local ent = arr[i]
            if not isRadar(ent) then continue end
            if not ent:getWirelink() then continue end

            nextRadars[#nextRadars + 1] = ent
        end
    end

    CachedRadars = nextRadars
    NextPrune = 0
    pollSensors()
end

local function pruneRadars()
    local changed = false

    for i = #CachedRadars, 1, -1 do
        local ent = CachedRadars[i]
        if isRadar(ent) and ent:getWirelink() then
            continue
        end

        table.remove(CachedRadars, i)
        changed = true
    end

    return changed
end

local function publishContacts(time)
    local outPosition = {}
    local outVelocity = {}
    local outOwner = {}
    local outIDs = {}
    local outTimes = {}
    local sortedIDs = {}
    local count = 0

    for id in pairs(Contacts) do
        sortedIDs[#sortedIDs + 1] = id
    end

    table.sort(sortedIDs)

    for i = 1, #sortedIDs do
        local id = sortedIDs[i]
        local data = Contacts[id]
        count = i
        outPosition[count] = data[1]
        outVelocity[count] = data[2]
        outOwner[count] = data[3]
        outIDs[count] = id
        outTimes[count] = data[4] or time
    end

    if count == 0 then
        clearOutputs()
        return
    end

    Outputs.Position:update(outPosition)
    Outputs.Velocity:update(outVelocity)
    Outputs.Owner:update(outOwner)
    Outputs.IDs:update(outIDs)
    Outputs.Times:update(outTimes)
    OutputsActive = true
    OutputsInitialized = true
    SnapshotDirty = false
    NextPublish = time + PUBLISH_RATE
end

local function updateContact(id, pos, vel, ownerName, time)
    local existing = Contacts[id]
    if existing == nil then
        Contacts[id] = {pos, vel, ownerName, time}
        SnapshotDirty = true
        return
    end

    if existing[1] ~= pos or existing[2] ~= vel or existing[3] ~= ownerName then
        SnapshotDirty = true
    end

    existing[1] = pos
    existing[2] = vel
    existing[3] = ownerName
    existing[4] = time
end

local function pruneContacts(time)
    local removed = false

    for id, data in pairs(Contacts) do
        if (time - data[4]) <= CONTACT_TTL then continue end

        Contacts[id] = nil
        removed = true
    end

    if removed then
        SnapshotDirty = true
    end
end

pollSensors = function()
    local time = timer.curtime()

    if time >= NextPrune then
        pruneRadars()
        NextPrune = time + PRUNE_INTERVAL
    end

    if #CachedRadars == 0 then
        Contacts = {}
        clearOutputs()
        schedulePoll()
        return
    end

    for i = 1, #CachedRadars do
        local ent = CachedRadars[i]
        local wl = ent:getWirelink()
        if not wl then continue end

        local positions = wl.Position
        local ids = wl.ID or wl.IDs
        local velocities = wl.Velocity
        local owners = wl.Owner

        if type(positions) ~= "table" or type(ids) ~= "table" then
            continue
        end

        for index, pos in pairs(positions) do
            local id = ids[index]
            if id == nil or pos == nil then continue end

            updateContact(
                id,
                pos,
                type(velocities) == "table" and velocities[index] or ZERO_VEC,
                type(owners) == "table" and owners[index] or "",
                time
            )
        end
    end

    pruneContacts(time)

    if SnapshotDirty and time >= NextPublish then
        publishContacts(time)
    end

    schedulePoll()
end

Inputs.Sensors:addTrigger(cacheSensors)
cacheSensors(Inputs.Sensors:get())
