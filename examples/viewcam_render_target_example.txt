--@name Viewcam Render Target Example
--@author Merydian9
--@client

--[[
Purpose:
- Minimal viewcam render-target example drawing a second camera feed on the HUD.
- Good for learning the bare minimum renderView -> RT -> textured rect flow.

What To Learn:
- How to throttle `render.renderView()` work.
- How to present a live secondary camera inside normal HUD drawing.

[FUTURE]
- Today we would gate this more carefully on quota and resolution, and usually wrap the RT/material setup in
  a helper so examples can focus on the camera logic itself.
]]--

if(player() != owner()) then return end
enableHud(owner(), true)

Player = owner()
fps = 10
last = timer.curtime()

render.createRenderTarget("rt")
local mat = material.create("gmodscreenspace")
mat:setTextureRenderTarget("$basetexture", "rt")

hook.add("renderscene", "scene",function()
    local dt = (timer.curtime()-last)
    -- [FUTURE] Modern viewcam code should also gate on quota, not just FPS cadence.
    if dt<(1/fps) then return end

    render.selectRenderTarget("rt")
    render.renderView({
        origin = chip():getPos() + Vector(0, 0, 10),
        angles = Angle(),
        x = 0,
        y = 0,
        drawhud = true
    })
    
    last = timer.curtime()
end)


hook.add("drawhud", "hud", function()
    local x, y = render.getResolution()
    local x, y = x / 2, y / 2
    
    render.setMaterial(mat)
    render.setColor(Color(255, 255, 255, 255))
    render.drawTexturedRect(x + 256, y - 256, 512, 512)
end)
