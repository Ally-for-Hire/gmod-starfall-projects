--[[
Purpose:
- Tiny radial menu/input capture example.
- Useful for learning the cursor-toggle and draw loop parts of simple in-HUD menus.

What To Learn:
- How to toggle cursor ownership from a keybind.
- How to keep a basic menu centered and interactive on screen.

[FUTURE]
- Today we would recompute resolution on change, avoid global vars, and probably wrap cursor/menu state in a
  small local UI helper instead of open-coding it.
]]--

O = owner()
P = player()

if P != O then return end

require("../allylib/helpful_hud_functions.txt")
enableHud(O, true)

local open = false
local centerX = 0
local centerY = 0
local resx, resy = render.getGameResolution()
local x, y = resx / 2, resy / 2

local function setOpen(state)
    open = state
    input.enableCursor(open)
    -- [FUTURE] Modern UI helpers usually recenter from current resolution instead of boot-time cached values.
    if open then
        input.setCursorPos(x, y)
    end
end

hook.add("inputPressed", "radial_toggle", function(key)
    if key == KEY.M then
        setOpen(not open)
    end
end)

hook.add("drawhud", "radial_draw", function()
    color(COL_BLACK)
    circle(x, y, 8, 6)
    drawPlus(x, y, 22, 22, 22, 22, 6)
    
    color(COL_GREEN)
    drawPlus(x, y, 20, 20, 20, 20, 2)
    circle(x, y, 10, 2)
    
    if open then
        local cursx, cursy = input.getCursorPos()
        render.setColor(COL_BLACK)
        drawPlus(cursx, cursy, 10, 10, 10, 10, 1)
    end
end)
