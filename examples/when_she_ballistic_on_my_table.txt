TIME_STEP = game.getTickInterval()
GRAVITY = Vector2(0, physenv.getGravity()[3])
MAX_TIME = 4
INCHES_PER_METER = 39.3701
MIN_FORWARD_SPEED = INCHES_PER_METER * 10
MAX_RANGE_METERS = 5000
RANGE_STEP_METERS = 50

AtZeroBallistics = {}
MuzzleVel = 0
VisualHolos = {}

local function interpolatePointsByDistance(low, high, val)
    local weight = (val - low[4]) / (high[4] - low[4])
    return Vector(low[1][1] + weight * (high[1][1] - low[1][1]), low[1][2] + weight * (high[1][2] - low[1][2]))
end

local function generateBallisticsTable(inputGun, angle, dt, maxTime)
    -- Initialization of variables
    local curTime = 0
    local indices = 0
    local dragCoef = inputGun:acfDragCoef()
    local gunPos = inputGun:getMassCenterW()
    local output = {}
    
    -- Calculating the starting position of the shell
    local attachmentInfo = inputGun:getAttachment(inputGun:lookupAttachment("muzzle"))
    local muzzlePos = Vector(attachmentInfo[1], attachmentInfo[2], attachmentInfo[3])
    local muzzleDist = gunPos:getDistance(muzzlePos)
    local shellPos = Vector2(math.cos(angle), math.sin(angle)) * muzzleDist
    
    -- Calculating the starting velocity of the shell
    local muzzleVelUnits = inputGun:acfMuzzleVel() * INCHES_PER_METER
    local shellVel = Vector2(math.cos(angle), math.sin(angle)) * muzzleVelUnits
    
    while curTime < maxTime and shellVel[1] > MIN_FORWARD_SPEED do
        local drag = (dragCoef * shellVel:getLength()^2) / 40
        drag = shellVel:getNormalized() * (drag * dt)
        drag = Vector2(drag[1], drag[2]) * INCHES_PER_METER
        
        shellPos = shellPos + shellVel * dt
        shellVel = shellVel + (GRAVITY * dt) - drag
                
        curTime = curTime + dt
        indices = indices + 1
        
        output[indices] = {shellPos, shellVel, curTime, math.sqrt(shellPos[1]^2 + shellPos[2]^2)} -- 2D Position, 2D Velocity, Current Time, Distance from Start
    end

    return output
end

-- Aforementioned logic to be run here
local function runLogic(gunEntity)
    MuzzleVel = gunEntity:acfMuzzleVel()
    
    AtZeroBallistics = generateBallisticsTable(gunEntity, 0, TIME_STEP, MAX_TIME)
    wire.ports.At_Zero_Ballistics = AtZeroBallistics
    
    local rangedBallistics = {}
    if #AtZeroBallistics < 2 then
        wire.ports.Ranged_Ballistics = rangedBallistics
        return
    end
    
    -- This will generate a relative to the gun's position table of points on the ballistic path
    local range = 0
    for i = 2, #AtZeroBallistics do
        local prevData = AtZeroBallistics[i - 1]
        local data = AtZeroBallistics[i]

        while data[4] > range * INCHES_PER_METER do
            local middlePos = interpolatePointsByDistance(prevData, data, range * INCHES_PER_METER)

            rangedBallistics[#rangedBallistics + 1] = Vector2(middlePos[1], middlePos[2])
            
            range = range + RANGE_STEP_METERS
            if range > MAX_RANGE_METERS then
                break
            end
        end
        if range > MAX_RANGE_METERS then
            break
        end
    end
    
    wire.ports.Ranged_Ballistics = rangedBallistics
end

-- Attempt to run the logic if the conditions are met
local function attemptLogic(name, value)
    if value == nil or name ~= "Gun" or not value:isValid() or value:getClass() ~= "acf_gun" or value:acfMuzzleVel() == MuzzleVel then
        return
    end
    
    runLogic(value)
end

-- Try to update the logic every 0.2s
timer.create("ballistics_retry", 0.1, 0, function()
    attemptLogic("Gun", wire.ports.Gun)
end)

-- Try to update the logic everytime the input is updated
hook.add("input", "ballistics_wire_input", attemptLogic)

-- Wire ports 
wire.adjustPorts({Gun = "entity"}, {At_Zero_Ballistics = "table", Ranged_Ballistics = "table"})
