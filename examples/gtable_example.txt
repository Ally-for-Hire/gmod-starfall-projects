--[[
Purpose:
- Minimal example of the GTABLE replicated-table library.
- Shows how multiple chips can share per-chip state and react to updates.

What To Learn:
- Basic GTABLE lifecycle: create, set local state, subscribe to changes, and pull a snapshot.
- The shape of a small replicated-state workflow without any HUD or wire concerns.

[FUTURE]
- Modern multi-writer GTABLE usage should assume versioned merges and tombstones, not naive overwrite semantics.
- If you expand this pattern, add explicit key ownership and packet-budget awareness.
]]--

-- pull in the GTABLE library
require("lib/gtable_library.txt")
-- [FUTURE] In larger examples we would wrap library setup in a local factory/helper instead of open-coded globals.

local PREFIX = "GTABLE_DEMO"
local gt = GTABLE:new(PREFIX)

-- Set our own entry (keys are per-chip by default)
gt:setSelf({
    id = chip():entIndex(),
    time = timer.curtime()
})

-- React to every update that is not from us
gt:onChange(function(tbl, key, value, source)
    -- ignore the callback for our own writes
    if key == chip():entIndex() then return end

    if key ~= nil then
        print(string.format("[%s] key=%s source=%s value=%s", PREFIX, tostring(key), tostring(source), tostring(value)))
    else
        print(string.format("[%s] bulk change source=%s size=%d", PREFIX, tostring(source), gt:count()))
    end
end, true)

-- ask the network for the current snapshot
gt:pull()
