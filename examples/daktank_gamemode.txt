-- daktank point cost detection
hook.add( "OnPhysgunReload", "Gamemode_DakPointCost", function(physgun,ply) 
    local trace = ply:getEyeTrace()
    local Ent = trace.Entity
    holograms.removeAll()
    SoundHolo = holograms.create(player():getEyePos(), Angle(0,0,0), "models/hunter/blocks/cube025x025x025.mdl", Vector(0))
    if isValid(Ent) and winner == "" then
        local ContChildren = Ent:getChildren()
        points = 0
        Gate = nil
        Core = nil
        if(playerTeam == 1) then
            points = team1Points
        else
            points = team2Points
        end
            
        for k,v in pairs(ContChildren) do
            if(v:getClass()=="gmod_wire_gate")then Gate = v; break end
        end       
        if(not isValid(Gate))then
            SoundHolo:emitSound("ambient/alarms/klaxon1.wav")
            return false
        end
        
        for k,v in pairs(Gate:getChildren()) do
            if(v:getClass()=="dak_tankcore") then Core = v; break end
        end       
        if(not isValid(Core) or Core:getMass()<=1 or Core:getMass() > points)then
            SoundHolo:emitSound("ambient/alarms/klaxon1.wav")
            Ent:remove()
            Gate:remove()
            return false
        else
            SoundHolo:emitSound("Item.Materialize")
            playerTeam = playersSorted[ply:getName()]
            if(playerTeam == 1) then
                team1Points = team1Points - Core:getMass()
            else
                team2Points = team2Points - Core:getMass()
            end
        end
    end
end)

-- daktank weapon spawning handler
hook.add( "PlayerSay", "Gamemode_WeaponSpawning", function(ply, text, plyteam, isdead)
    if not isdead then
        SoundHolo = holograms.create(player():getEyePos(), Angle(0,0,0), "models/hunter/blocks/cube025x025x025.mdl", Vector(0))
        textin = string.split(text, "!")
        options = {["pz"] = {"dak_panzerschreck",7},["jav"] = {"dak_javelin",20},["ptrs"] = {"dak_ptrs41",2}}
        
        if( textin[1] == "" ) then
            command = string.split(string.lower(text)," ")
            if command[1] == "!wep" then
                local weapon = options[command[2]]
                if (weapon != nil) then
                    local player = string.split(ply:getName()," ")[1]
                    local playerTeam = playersSorted[ply:getName()]
                    if( playerTeam == 1 ) then
                        if weapon[2] < team1Points then
                            team1Points = team1Points - weapon[2]
                            SoundHolo:emitSound("Item.Materialize")
                            concmd("ulx give " .. player .. " " .. weapon[1]) 
                        else
                            SoundHolo:emitSound("ambient/alarms/klaxon1.wav")
                        end
                    else
                        if weapon[2] < team2Points then
                            team2Points = team2Points - weapon[2]
                            SoundHolo:emitSound("Item.Materialize")
                            concmd("ulx give " .. player .. " " .. weapon[1]) 
                        else
                            SoundHolo:emitSound("ambient/alarms/klaxon1.wav")
                        end
                    end
                else
                    SoundHolo:emitSound("ambient/alarms/warningbell1.wav")
                end
            end
        end
    end
end)

-- server code, handles all logic
if SERVER then
    function arrayLen(array) 
        local size = 0
        for _ in pairs(array) do size = size + 1 end
        return size
    end
    caps = {{Vector(10861, 8612, -1882),3,{}},{Vector(7550, -4925, -2061),3,{}},{Vector(-9033, 9634, -2193),3,{}},{Vector(-10809, -7097, -2191),3,{}}}
    caps = {{Vector(-9172, 9459, -2193),3,{}},{Vector(8912, -10506, -2191),3,{}}}
    local pointSize = 1500
    
    team1 = {}
    team2 = {}  
    playersSorted = {}
    team1Points = 5
    team2Points = 5
    winner = ""
    teamPoints = {team1Points,team2Points}
    pointCapPercents = {}
    for c,v in pairs(caps) do 
        pointCapPercents[c] = {0,0,0}
    end
    teams = {team1,team2,team1Points,team2Points,playersSorted} 
    local players = find.allPlayers()
    local teamColours = {Color(255, 0, 0),Color(0,0,255),Color(155,155,155)}
    teamSize = arrayLen(players) / 2 
    print(Color(0,255,0),"Teams of " .. teamSize .. " Players each chosen")
    
    for _,v in pairs(players) do
        retry = true
        while retry do
            teamPick = math.random(1,2)
            chosenTeam = teams[teamPick]
            if arrayLen(chosenTeam)<teamSize then
                retry = false
                table.insert(chosenTeam,#chosenTeam,v)
                playersSorted[v:getName()] = teamPick
            end          
        end
    end
    
    
    timer.create("Gamemode_CapCheck",1,0,function()
        if winner == "" then
            team1GainPoints = 0
            team2GainPoints = 0
            for c,v in pairs(caps) do
                if v[2] == 1 then
                    team1GainPoints = team1GainPoints + 1
                elseif v[2] == 2 then
                    team2GainPoints = team2GainPoints + 1
                end
            
                local players = find.allPlayers()
                nearbyPlayers = {0,0}
                for _,p in pairs(players) do
                    if(v[1]:getDistance(p:getPos()) < pointSize/2 and not isValid(p:getVehicle()) and p:isAlive()) then
                        -- if player is on point execute below code
                        local playerTeam = playersSorted[p:getName()]
                        nearbyPlayers[playerTeam] = nearbyPlayers[playerTeam] + 1
                        local players = find.allPlayers()
                        local enemyTeamOnPoint = 0
                        for _,k in pairs(players) do
                            if v[1]:getDistance(k:getPos()) < pointSize/2 and not isValid(k:getVehicle()) and k:isAlive() and playersSorted[k:getName()] != playerTeam then
                                enemyTeamOnPoint = 1
                            end
                        end
                        if(v[2]!=playerTeam and enemyTeamOnPoint == 0) then
                            pointCapPercents[c][playerTeam] = pointCapPercents[c][playerTeam] + 1
                            caps[c][3] = pointCapPercents[c][playerTeam]
                            if( pointCapPercents[c][playerTeam] >= 15 ) then 
                                caps[c][2] = playerTeam 
                                pointCapPercents[c][playerTeam] = 0
                            end 
                        end
                    end
                end
                v[3] = pointCapPercents[c]          
                for j,k in pairs(nearbyPlayers) do
                    if k == 0 then
                        pointCapPercents[c][j] = math.max(pointCapPercents[c][j] - 0.3,0)
                    else
                        
                    end
                end
            end
            if(team1GainPoints > team2GainPoints) then
                team1Points = team1Points + 0.25 * team1GainPoints
            elseif(team2GainPoints > team1GainPoints) then
                team2Points = team2Points + 0.25 * team2GainPoints
            end
            if(team1Points > 75) then winner = "Blue Team"
            elseif(team2Points > 75) then winner = "Red Team" end
        end
    end)
    
    timer.create( "Gamemode_NetInterface", 1, 0, function() -- redox wrote this
        net.start("Gamemode_CapInformation")
        net.writeTable(caps)
        net.send(find.allPlayers())
        teams = {team1,team2,team1Points,team2Points,playersSorted,winner}
        net.start("Gamemode_TeamInformation")
        net.writeTable(teams)
        net.send(find.allPlayers())
    end)
    timer.create( "Gamemode_HoloUpdate", 1, 0, function()
        holograms.removeAll()
        for k, m in ipairs( caps ) do
            local capPos = m[1]
            local capHolo = holograms.create(capPos, Angle(0,0,90), "models/sprops/geometry/t_fdisc_24.mdl", Vector(1))
            capHolo:setSize(Vector(pointSize,1,pointSize))
            capHolo:setMaterial("models/wireframe")
            capHolo:setColor(teamColours[m[2]])
        end
    end)
end

            else 
                fontSize = 26
