--[[
Purpose:
- Tiny reusable base pattern for chair/driver-aware HUD chips.
- Demonstrates one of the older approaches to sharing seat occupancy with a client HUD.

What To Learn:
- Basic chair input handling and driver state mirroring.
- A minimal server/client event flow around entering and leaving a vehicle.

[FUTURE]
- Today this would be built on a helper that owns driver tracking and owner-view gating instead of storing
  mutable shared globals directly on the root file.
]]--

INPUTS = {}
SHARED = {}

INPUTS.Chair = nil
SHARED.Driver = nil
SHARED.O = owner()
SHARED.P = player()
SHARED.ENTID = chip():entIndex()

if SERVER then
    wire.adjustInputs({"Chair"}, {"Entity"}, {"Vehicle Chair"})
    
    -- [FUTURE] A modern helper could abstract seat-enter/seat-leave hooks into one reusable driver-state service.
    local function enteredVehicleHandler(ply, vehicle, num)
        if vehicle ~= INPUTS.Chair then return end
        
        SHARED.Driver = ply
        net.start(SHARED.ENTID .. " Driver Update")
            net.writeEntity(ply)
        net.send()
    end
    
    local function leftVehicleHandler(ply, vehicle)
        if vehicle ~= INPUTS.Chair then return end
        
        net.start(SHARED.ENTID .. " Driver Update")
            net.writeEntity(SHARED.Driver)
        net.send()
        SHARED.Driver = nil
    end
    
    local function wireHandler(name, value)
        if value == nil then return end
        
        if name == "Chair" and isValid(value) then 
            INPUTS.Chair = value 
        end
    end

    local function clientInitializedHandler(entity)
        net.start(SHARED.ENTID .. " Initial Wire Sync")
            -- Put things that will be wired at the start here
        net.send()
    end

    hook.add("ClientInitialized", "ClientInitialized", clientInitialiedHandler)
    hook.add("PlayerEnteredVehicle", SHARED.ENTID .. " PlayerEnteredVehicle", enteredVehicleHandler)
    hook.add("PlayerLeaveVehicle", SHARED.ENTID .. " PlayerLeaveVehicle", leftVehicleHandler)
    hook.add("Input", SHARED.ENTID .. " Input", wireHandler)
end

    local function driverUpdateHandler()
        if SHARED.P ~= SHARED.Driver and SHARED.Driver ~= nil then return end
            
        local newDriver = net.readEntity()
        if newDriver == SHARED.Driver then
            enableHud(SHARED.Driver, false)
            SHARED.Driver = nil
            return
        end
        
        SHARED.Driver = newDriver
        enableHud(SHARED.Driver, true)
    end

    end

    end

    net.receive(SHARED.ENTID .. " Wire Update", wireHandler)
    net.receive(SHARED.ENTID .. " Initial Wire Sync", initialWireHandler)
end
