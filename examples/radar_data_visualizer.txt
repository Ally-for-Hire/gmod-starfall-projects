--[[
Purpose:
- Visualizes raw radar-style track arrays and their freshness on a HUD.
- Serves as a compact example of stale-track handling, extrapolation, and array-aligned data flow.

What To Learn:
- How to compare incoming position/velocity arrays and only push updates when the dataset really changes.
- How to extrapolate for short outages without pretending stale data is fresh forever.

[FUTURE]
- If scaled up today, this would want packet chunking, explicit schema versioning, and probably the same cache
  merge patterns used in the newer tracksync work.
]]--

local UPDATE_INTERVAL = 0.05
local POSITION_EPS = 1
local VELOCITY_EPS = 1
local STALE_TRACK_TIME = 1.5
local MAX_EXTRAPOLATION = 0.35
local NET_ID_BITS = 16

if SERVER then
    local positions = {}
    local velocities = {}
    local owners = {}
    local ids = {}
    local lastUpdate = 0
    local posEpsSq = POSITION_EPS * POSITION_EPS
    local velEpsSq = VELOCITY_EPS * VELOCITY_EPS

    -- [FUTURE] Modern tracking systems usually hide this comparison logic behind a cache/helper module.
    local function datasetChanged(newPositions, newVelocities, newOwners, newIds)
        local count = #newPositions
        if count ~= #positions then return true end

        for i = 1, count do
            if newIds[i] ~= ids[i] then return true end
            if newOwners[i] ~= owners[i] then return true end
            if vectorDeltaLenSq(newPositions[i], positions[i]) > posEpsSq then return true end
            if vectorDeltaLenSq(newVelocities[i], velocities[i]) > velEpsSq then return true end
        end

        return false
    end

    local function storeSnapshot(newPositions, newVelocities, newOwners, newIds)
        positions = {}
        velocities = {}
        owners = {}
        ids = {}

        for i = 1, #newPositions do
            positions[i] = newPositions[i]
            velocities[i] = newVelocities[i]
            owners[i] = newOwners[i]
            ids[i] = newIds[i]
        end
    end

    local function handleInput()
        local now = timer.curtime()
        if (now - lastUpdate) < UPDATE_INTERVAL then return end

        local newPositions = wire.ports.Position or {}
        local newVelocities = wire.ports.Velocity or {}
        local newOwners = wire.ports.Owner or {}
        local newIds = wire.ports.IDs or {}
        local count = #newPositions

        if not (count == #newVelocities and count == #newOwners and count == #newIds) then return end
        if not datasetChanged(newPositions, newVelocities, newOwners, newIds) then return end

        lastUpdate = now
        storeSnapshot(newPositions, newVelocities, newOwners, newIds)

        net.start("updateValues")
            net.writeInt(count, NET_ID_BITS)
            net.writeFloat(now)
            for i = 1, count do
                net.writeVector(positions[i])
                net.writeVector(velocities[i])
                net.writeString(owners[i] or "")
                net.writeInt(ids[i] or 0, NET_ID_BITS)
            end
        net.send(owner(), true)
    end

    hook.add("input", "Update Inputs", handleInput)
    wire.adjustPorts({Position = "array", Velocity = "array", Owner = "array", IDs = "array"}, {})
end

if CLIENT then
    if player() ~= owner() then return end

    local box_size = 40
    local targetBoxTextureSize = 64
    local type = 2
    local posEpsSq = POSITION_EPS * POSITION_EPS
    local velEpsSq = VELOCITY_EPS * VELOCITY_EPS

    local tracks = {}
    local drawOrder = {}
    local serverTime = 0
    local lastUpdate = 0

    local col_black = Color(0, 0, 0)
    local col_black_noalpha = Color(0, 0, 0, 0)
    local col_white = Color(255, 255, 255)
    local col_main = col_white

    local font_main = render.createFont("FontAwesome", 16, 800, nil, nil, nil, true)

    local drawRect = render.drawRect
    local drawLine = render.drawLine
    local drawSimpleText = render.drawSimpleText
    local drawPoly = render.drawPoly
    local setColor = render.setColor
    local setFont = render.setFont
    local setRenderTargetTexture = render.setRenderTargetTexture
    local drawTexturedRect = render.drawTexturedRect
    local selectRenderTarget = render.selectRenderTarget
    local createRenderTarget = render.createRenderTarget
    local clear = render.clear
    local getResolution = render.getResolution

    local function drawThickLine(x1, y1, x2, y2, thickness)
        local dx = x2 - x1
        local dy = y2 - y1
        local len = math.sqrt(dx * dx + dy * dy)
        if len <= 0 then return end

        local scale = thickness * 0.5 / len
        local perpX = dy * scale
        local perpY = -dx * scale

        drawPoly({
            {x = x1 + perpX, y = y1 + perpY},
            {x = x2 + perpX, y = y2 + perpY},
            {x = x2 - perpX, y = y2 - perpY},
            {x = x1 - perpX, y = y1 - perpY}
        })
    end

    local function drawRenderTarget(x, y, name, sizeX, sizeY)
        if sizeY == nil then sizeY = sizeX end
        setRenderTargetTexture(name)
        drawTexturedRect(x - sizeX * 0.5, y - sizeY * 0.5, sizeX, sizeY)
        setRenderTargetTexture()
    end

    local function updateVariables()
        local count = net.readInt(NET_ID_BITS)
        serverTime = net.readFloat()
        lastUpdate = timer.curtime()

        local seen = {}
        for i = 1, count do
            local pos = net.readVector()
            local vel = net.readVector()
            local ownerName = net.readString()
            local id = net.readInt(NET_ID_BITS)
            local track = tracks[id]

            if not track then
                track = {}
                tracks[id] = track
            end

            local changed = (not track.pos)
                or vectorDeltaLenSq(track.pos, pos) > posEpsSq
                or vectorDeltaLenSq(track.vel, vel) > velEpsSq
                or track.owner ~= ownerName

            track.pos = pos
            track.vel = vel
            track.owner = ownerName
            track.id = id
            track.lastServerTime = serverTime
            if changed then
                track.lastChangeTime = serverTime
            elseif not track.lastChangeTime then
                track.lastChangeTime = serverTime
            end

            drawOrder[i] = id
            seen[id] = true
        end

        for i = count + 1, #drawOrder do
            drawOrder[i] = nil
        end

        for id, track in pairs(tracks) do
            if not seen[id] and (serverTime - (track.lastServerTime or serverTime)) > STALE_TRACK_TIME then
                tracks[id] = nil
            end
        end
    end

    local function getDisplayTrackPos(track, now)
        local age = now - (track.lastServerTime or now)
        if age < 0 then
            age = 0
        end

        local extrapolation = math.min(age + UPDATE_INTERVAL * 0.5, MAX_EXTRAPOLATION)
        local staleAge = now - (track.lastChangeTime or now)
        if staleAge > STALE_TRACK_TIME then
            extrapolation = 0
        end

        return track.pos + track.vel * extrapolation, staleAge
    end

    local function renderHud()
        setColor(col_main)
        setFont(font_main)

        local now = serverTime + (timer.curtime() - lastUpdate)
        for i = 1, #drawOrder do
            local id = drawOrder[i]
            local track = tracks[id]
            if track and track.pos and track.vel then
                local displayPos, staleAge = getDisplayTrackPos(track, now)
                local coords = displayPos:toScreen()

                if coords.visible then
                    drawRenderTarget(coords.x, coords.y, "targetBox", box_size, box_size)
                    drawSimpleText(coords.x + box_size * 0.5 + 5, coords.y - box_size * 0.5, track.owner or "", TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
                    drawSimpleText(coords.x + box_size * 0.5 + 5, coords.y + box_size * 0.5, math.round(track.vel:getLength() / 17.6) .. "KM/H", TEXT_ALIGN.LEFT, TEXT_ALIGN.BOTTOM)

                    if staleAge > 0.5 then
                        drawLine(coords.x - 15, coords.y - 15, coords.x + 15, coords.y + 15)
                        drawLine(coords.x + 15, coords.y - 15, coords.x - 15, coords.y + 15)
                        render.drawRectOutline(coords.x - 15, coords.y - 15, 30, 30, 1)
                    end
                end
            end
        end
    end

    local function createStaticElements()
        local x, y = targetBoxTextureSize, targetBoxTextureSize
        x = x * 0.5
        y = y * 0.5

        createRenderTarget("targetBox")
        selectRenderTarget("targetBox")
        clear(col_black_noalpha)
        setFont(font_main)

        if type == 1 then
            setColor(col_black)
            drawThickLine(x + box_size * 0.5 + 2, y + box_size * 0.5, x + box_size * 0.5 - 17, y + box_size * 0.5, 5)
            drawThickLine(x + box_size * 0.5, y + box_size * 0.5 + 2, x + box_size * 0.5, y + box_size * 0.5 - 17, 5)
            drawThickLine(x - box_size * 0.5 - 2, y + box_size * 0.5, x - box_size * 0.5 + 17, y + box_size * 0.5, 5)
            drawThickLine(x - box_size * 0.5, y + box_size * 0.5 + 2, x - box_size * 0.5, y + box_size * 0.5 - 17, 5)

            drawThickLine(x + box_size * 0.5 + 2, y - box_size * 0.5, x + box_size * 0.5 - 17, y - box_size * 0.5, 5)
            drawThickLine(x + box_size * 0.5, y - box_size * 0.5 - 2, x + box_size * 0.5, y - box_size * 0.5 + 17, 5)
            drawThickLine(x - box_size * 0.5 - 2, y - box_size * 0.5, x - box_size * 0.5 + 17, y - box_size * 0.5, 5)
            drawThickLine(x - box_size * 0.5, y - box_size * 0.5 - 2, x - box_size * 0.5, y - box_size * 0.5 + 17, 5)

            setColor(col_main)
            drawThickLine(x + box_size * 0.5, y + box_size * 0.5, x + box_size * 0.5 - 15, y + box_size * 0.5, 2)
            drawThickLine(x + box_size * 0.5, y + box_size * 0.5, x + box_size * 0.5, y + box_size * 0.5 - 15, 2)
            drawThickLine(x - box_size * 0.5, y + box_size * 0.5, x - box_size * 0.5 + 15, y + box_size * 0.5, 2)
            drawThickLine(x - box_size * 0.5, y + box_size * 0.5, x - box_size * 0.5, y + box_size * 0.5 - 15, 2)

            drawThickLine(x + box_size * 0.5, y - box_size * 0.5, x + box_size * 0.5 - 15, y - box_size * 0.5, 2)
            drawThickLine(x + box_size * 0.5, y - box_size * 0.5, x + box_size * 0.5, y - box_size * 0.5 + 15, 2)
            drawThickLine(x - box_size * 0.5, y - box_size * 0.5, x - box_size * 0.5 + 15, y - box_size * 0.5, 2)
            drawThickLine(x - box_size * 0.5, y - box_size * 0.5, x - box_size * 0.5, y - box_size * 0.5 + 15, 2)
        elseif type == 2 then
            setColor(col_black)
            drawThickLine(x - 2, y, x + 22, y, 5)
            drawThickLine(x + 20, y - 7, x + 20, y + 7, 5)
            drawRect(x - 4, y - 4, 8, 8)

            setColor(col_main)
            drawThickLine(x, y, x + 20, y, 2)
            drawThickLine(x + 20, y - 5, x + 20, y + 5, 2)
            drawRect(x - 2.5, y - 2.5, 5, 5)
        end

        selectRenderTarget()
        hook.remove("renderoffscreen", "Create Rendertargets")
    end

    hook.add("drawhud", "Visualize Information", renderHud)
    hook.add("renderoffscreen", "Create Rendertargets", createStaticElements)
    net.receive("updateValues", updateVariables)
end
